84  bool KTTSGenerator::Run()
85  {
86  // Create, signal, and destroy the egg header
87  KTEggHeader* newHeader = CreateEggHeader();
88 
89  KTDEBUG(genlog, "Created header:\n"
90  << "\tFilename: " << newHeader->GetFilename() << '\n'
91  << "\tAqcuisition Mode: " << newHeader->GetAcquisitionMode() << '\n'
92  << "\tNumber of Channels: " << newHeader->GetNChannels() << '\n'
93  << "\tSlice Size: " << newHeader->GetSliceSize() << '\n'
94  << "\tRecord Size: " << newHeader->GetRecordSize() << '\n'
95  << "\tRun Duration: " << newHeader->GetRunDuration() << " ms" << '\n'
96  << "\tAcquisition Rate: " << newHeader->GetAcquisitionRate() << " Hz \n"
97  << "\tTimestamp: " << newHeader->GetTimestamp() << '\n'
98  << "\tDescription: " << newHeader->GetDescription() << '\n'
99  << "\tRun Type: " << newHeader->GetRunType() << '\n'
100  << "\tRun Source: " << newHeader->GetRunSource() << '\n'
101  << "\tFormat Mode: " << newHeader->GetFormatMode());
102 
103  fHeaderSignal(newHeader);
104  delete newHeader;
105 
106  // Loop over slices
107  // The local copy of the data shared pointer is created and destroyed in each iteration of the loop
108  for (fSliceCounter = 0; fSliceCounter < fNSlices; ++fSliceCounter)
109  {
110  KTDataPtr newData = CreateNewData();
111 
112  if (! AddSliceHeader(*newData.get()))
113  {
114  KTERROR(genlog, "Something went wrong while adding the slice header");
115  return false;
116  }
117 
118  if (! AddEmptySlice(*newData.get()))
119  {
120  KTERROR(genlog, "Something went wrong while adding the empty slices");
121  return false;
122  }
123 
124  if (! newData->Has< KTTimeSeriesData >())
125  {
126  KTERROR(genlog, "New data does not contain time-series data!");
127  return false;
128  }
129 
130  GenerateTS(newData->Of< KTTimeSeriesData >());
131 
132  fDataSignal(newData);
133  }
134 
135  fDoneSignal();
136 
137  KTProcSummary* summary = new KTProcSummary();
138  summary->SetNSlicesProcessed(fSliceCounter);
139  summary->SetNRecordsProcessed((unsigned)ceil(double(fSliceCounter * fSliceSize) / fRecordSize));
140  summary->SetIntegratedTime(double(fSliceCounter * fSliceSize) * fBinWidth);
141  fSummarySignal(summary);
142  delete summary;
143 
144  return true;
145  }

180  bool KTTSGenerator::AddSliceHeader(KTData& data) const
181  {
182  KTSliceHeader& sliceHeader = data.Of< KTSliceHeader >().SetNComponents(1);
183  sliceHeader.SetSampleRate(1. / fBinWidth);
184  sliceHeader.SetSliceSize(fSliceSize);
185  sliceHeader.CalculateBinWidthAndSliceLength();
186  sliceHeader.SetTimeInRun(double(fSliceCounter * fSliceSize) * fBinWidth);
187  sliceHeader.SetSliceNumber(fSliceCounter);
188 
189  for (unsigned iComponent = 0; iComponent < fNChannels; ++iComponent)
190  {
191  sliceHeader.SetTimeStamp((TimeType)(sliceHeader.GetTimeInRun() * (double)NSEC_PER_SEC), iComponent); // TODO: change this to 1e3 when switch to usec is made
192  sliceHeader.SetAcquisitionID(0);
193  sliceHeader.SetRecordID(0);
194  }
195 
196  KTDEBUG(genlog, "Filled out slice header:\n"
197  << "\tSample rate: " << sliceHeader.GetSampleRate() << " Hz\n"
198  << "\tSlice size: " << sliceHeader.GetSliceSize() << '\n'
199  << "\tBin width: " << sliceHeader.GetBinWidth() << " s\n"
200  << "\tSlice length: " << sliceHeader.GetSliceLength() << " s\n"
201  << "\tTime in run: " << sliceHeader.GetTimeInRun() << " s\n"
202  << "\tSlice number: " << sliceHeader.GetSliceNumber() << '\n'
203  << "\tStart record number: " << sliceHeader.GetStartRecordNumber() << '\n'
204  << "\tStart sample number: " << sliceHeader.GetStartSampleNumber());
205 
206  return true;
207  }
