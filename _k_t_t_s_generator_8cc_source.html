82  Bool_t KTTSGenerator::Run()
83  {
84  // Create, signal, and destroy the egg header
85  KTEggHeader* newHeader = CreateEggHeader();
86 
87  KTDEBUG(genlog, "Created header:\n"
88  << "\tFilename: " << newHeader->GetFilename() << '\n'
89  << "\tAcuisition Mode: " << newHeader->GetAcquisitionMode() << '\n'
90  << "\tNumber of Channels: " << newHeader->GetNChannels() << '\n'
91  << "\tSlice Size: " << newHeader->GetSliceSize() << '\n'
92  << "\tRecord Size: " << newHeader->GetRecordSize() << '\n'
93  << "\tAcquisition Time: " << newHeader->GetAcquisitionTime() << " s" << '\n'
94  << "\tAcquisition Rate: " << newHeader->GetAcquisitionRate() << " Hz ");
95 
96  fHeaderSignal(newHeader);
97  delete newHeader;
98 
99  // Loop over slices
100  // The local copy of the data shared pointer is created and destroyed in each iteration of the loop
101  for (fSliceCounter = 0; fSliceCounter < fNSlices; fSliceCounter++)
102  {
103  shared_ptr< KTData > newData = CreateNewData();
104 
105  if (! AddSliceHeader(*newData.get()))
106  {
107  KTERROR(genlog, "Something went wrong while adding the slice header");
108  return false;
109  }
110 
111  if (! AddEmptySlice(*newData.get()))
112  {
113  KTERROR(genlog, "Something went wrong while adding the empty slices");
114  return false;
115  }
116 
117  if (! newData->Has< KTTimeSeriesData >())
118  {
119  KTERROR(genlog, "New data does not contain time-series data!");
120  return false;
121  }
122 
123  GenerateTS(newData->Of< KTTimeSeriesData >());
124 
125  fDataSignal(newData);
126  }
127 
128  fDoneSignal();
129 
130  KTProcSummary* summary = new KTProcSummary();
131  summary->SetNSlicesProcessed(fSliceCounter);
132  summary->SetNRecordsProcessed((UInt_t)ceil(Double_t(fSliceCounter * fSliceSize) / fRecordSize));
133  summary->SetIntegratedTime(Double_t(fSliceCounter * fSliceSize) * fBinWidth);
134  fSummarySignal(summary);
135  delete summary;
136 
137  return true;
138  }

167  Bool_t KTTSGenerator::AddSliceHeader(KTData& data) const
168  {
169  KTSliceHeader& sliceHeader = data.Of< KTSliceHeader >().SetNComponents(1);
