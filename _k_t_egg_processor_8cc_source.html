140  bool KTEggProcessor::ProcessEgg()
141  {
142  KTEggReader* reader = NULL;
143 
144  // Create egg reader and transfer information
145  if (fEggReaderType == kMonarchEggReader)
146  {
147 #ifdef USE_MONARCH
148  KTEggReaderMonarch* eggReaderMonarch = new KTEggReaderMonarch();
149  eggReaderMonarch->SetSliceSize(fSliceSize);
150  eggReaderMonarch->SetStride(fStride);
151  if (fTimeSeriesType == kRealTimeSeries)
152  eggReaderMonarch->SetTimeSeriesType(KTEggReaderMonarch::kRealTimeSeries);
153  else if (fTimeSeriesType == kFFTWTimeSeries)
154  eggReaderMonarch->SetTimeSeriesType(KTEggReaderMonarch::kFFTWTimeSeries);
155  reader = eggReaderMonarch;
156 #else
157  KTERROR(egglog, "Monarch is not enabled; please select another egg reader type");
158  return false;
159 #endif
160  }
161  else
162  {
163  KTEggReader2011* eggReader2011 = new KTEggReader2011();
164  reader = eggReader2011;
165  }
166 
167  KTEggHeader* header = reader->BreakEgg(fFilename);
168  if (header == NULL)
169  {
170  KTERROR(egglog, "Egg did not break");
171  return false;
172  }
173 
174  fHeaderSignal(header);
175 
176  KTINFO(egglog, "The egg file has been opened successfully and the header was parsed and processed;");
177  KTPROG(egglog, "Proceeding with slice processing");
178 
179  if (fCalculateNormalization) CalculateNormalization();
180 
181  if (fNSlices == 0) UnlimitedLoop(reader);
182  else LimitedLoop(reader);
183 
184  fEggDoneSignal();
185 
186  KTProcSummary* summary = new KTProcSummary();
187  summary->SetNSlicesProcessed(reader->GetNSlicesProcessed());
188  summary->SetNRecordsProcessed(reader->GetNRecordsProcessed());
189  summary->SetIntegratedTime(reader->GetIntegratedTime());
190  KTDEBUG(egglog, "Summary of processing:\n" <<
191  "\tSlices processed: " << summary->GetNSlicesProcessed() << '\n' <<
192  "\tRecords processed: " << summary->GetNRecordsProcessed() << '\n' <<
193  "\tIntegrated time: " << summary->GetIntegratedTime() << " s");
194  fSummarySignal(summary);
195  delete summary;
196 
197  return true;
