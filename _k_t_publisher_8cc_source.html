45  {
46  // Configure writers

48  for (KTPStoreNode::const_sorted_iterator it = itPair.first; it != itPair.second; it++)
49  {
50  KTPStoreNode subNode = KTPStoreNode(&(it->second));
51  if (! subNode.HasData("type"))
52  {
53  KTERROR(publog, "Unable to add writer: no writer type given");
54  return false;
55  }
56  string writerType = subNode.GetData("type");
57  string writerName = subNode.GetData("name", writerType);
58  KTWriter* newWriter = AddWriter(writerType, writerName);
59

62  KTERROR(publog, "Something went wrong while creating the writer.");

67  if (! newWriter->Configure(writerConfigNode))

116  {
117  PubMapIter pmIter = fPubMap.find(writerName);
118 
119  if (pmIter == fPubMap.end())
120  {
121  // Writer with writerName does not yet exist
122  // Create a new writer
123  KTWriter* newWriter = fPubFactory->Create(writerType);
124  if (newWriter == NULL)
125  {
126  // Failed to create the writer
127  KTERROR(publog, "Failed to create new writer <" << writerName << "> of type <" << writerType << ">");
128  return NULL;
129  }
130 
131  // Insert a new writer and data list into the publication map
132  std::pair< PubMapIter, Bool_t > newInsert = fPubMap.insert(PubMapValue(writerName, WriterAndDataList()));
133  if (! newInsert.second)
134  {
135  KTERROR(publog, "Failed to insert new writer into the publication map; writer name: " << writerName << "; writer type: " << writerType);
136  delete newWriter;

138  }
139  // Insertion was successful

141  // Hand off newWriter to the publication map
142  pmIter->second.fWriter = newWriter;

243  void KTPublisher::Publish(shared_ptr<KTBundle> bundle)
244  {
245  if (fStatus == kStopped) return;
246 
247  // Loop over all writers in the publication map
248  for (PubMapCIter pmIter = fPubMap.begin(); pmIter != fPubMap.end(); pmIter++)
249  {
250  KTWriter* writer = pmIter->second.fWriter;
251  // Loop over all of the data types that have been requested for publication for this writer
252  for (DataListIter dlIter = pmIter->second.fDataList.begin(); dlIter != pmIter->second.fDataList.end(); dlIter++)
253  {
254  const string dataName = *dlIter;
255  // Attempt to get data from the bundle with this name
256  KTWriteableData* data = bundle->GetData< KTWriteableData >(dataName);
257  if (data != NULL)

259  // Data exists and is writeable --- publish it!
