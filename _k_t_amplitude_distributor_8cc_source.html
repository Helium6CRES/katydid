354  bool KTAmplitudeDistributor::CreateDistributionsFromBuffer()
355  {
356  if (fBuffer.size() == 0)
357  {
358  KTERROR(adlog, "Buffer is empty, but buffer use was requested");
359  return false;
360  }
361 
362  KTDEBUG(adlog, "Initializing NULL distributions: " << fNComponents << " components with " << fNFreqBins << " frequency bins");
363  fDistributions->InitializeNull(fNComponents, fNFreqBins);
364  if (fDistributions->GetNComponents() != fNComponents)
365  {
366  KTERROR(adlog, "The distributions data doesn't have the correct number of components: " << fDistributions->GetNComponents());
367  return false;
368  }
369  if (fDistributions->GetNFreqBins() != fNFreqBins)
370  {
371  KTERROR(adlog, "The distributions data doesn't have the correct number of frequency bins: " << fDistributions->GetNFreqBins());
372  return false;
373  }
374 
375  if (fBuffer.size() > fNSlicesProcessed)
376  {
377  KTDEBUG(adlog, "Buffer is larger than the number of slices processed; resizing to " << fNSlicesProcessed);
378  fBuffer.resize(fNSlicesProcessed);
379  }
380 
381  double distMin, distMax, value;
382  //unsigned distBin;
383  for (unsigned iComponent = 0; iComponent < fNComponents; iComponent++)
384  {
385  for (unsigned iBin = fMinBin; iBin <= fMaxBin; iBin++)
386  {
387  KTWARN(adlog, "0 " << iComponent << " " << iBin << " " << fBuffer[0][iComponent][iBin]);
388  distMin = fBuffer[0][iComponent][iBin];
389  distMax = distMin;
390  KTERROR(adlog, distMin << " " << distMax << " buffer size: " << fBuffer.size());
391  for (unsigned iSpectrum = 1; iSpectrum < fBuffer.size(); iSpectrum++)
392  {
393  value = fBuffer[iSpectrum][iComponent][iBin];
394  if (value < distMin) distMin = value;
395  else if (value > distMax) distMax = value;
396  KTERROR(adlog, " " << distMin << " " << distMax);
397  }
398  if (! fDistributions->InitializeADistribution(iComponent, iBin, fDistNBins, distMin, distMax))
399  {
400  KTERROR(adlog, "There was a problem initializing a distribution: component " << iComponent << "; frequency bin " << iBin);
401  return false;
402  }
403  KTDEBUG(adlog, "Distribution initialized; filling in from buffer");
404  for (unsigned iSpectrum = 1; iSpectrum < fBuffer.size(); iSpectrum++)
405  {
406  value = fBuffer[iSpectrum][iComponent][iBin];
407  fDistributions->AddToDist(iBin, value, iComponent);
408  }
409  }
410  }
411 
412  // The condition (fUseBuffer && fBuffer.empty()) implies that if data has been taken, it's been transferred to distributions
413  fBuffer.clear();
414 
415  return true;
416  }
