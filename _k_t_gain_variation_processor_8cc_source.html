104  bool KTGainVariationProcessor::CoreGainVarCalc(KTFrequencySpectrumDataPolarCore& data, KTGainVariationData& newData)
105  {
106  if (fCalculateMinBin)
107  {
108  SetMinBin(data.GetSpectrumPolar(0)->FindBin(fMinFrequency));
109  }
110  else
111  {
112  fMinFrequency = data.GetSpectrumPolar(0)->GetBinCenter(fMinBin);
113  }
114  if (fCalculateMaxBin)
115  {
116  SetMaxBin(data.GetSpectrumPolar(0)->FindBin(fMaxFrequency));
117  }
118  else
119  {
120  fMaxFrequency = data.GetSpectrumPolar(0)->GetBinCenter(fMaxBin);
121  }
122  KTDEBUG(gvlog, "min frequency: " << fMinFrequency << "; max frequency: " << fMaxFrequency << "; min bin: " << fMinBin << "; max bin " << fMaxBin << "; input range max " << data.GetSpectrumPolar(0)->GetRangeMin() << "; input range min: " << data.GetSpectrumPolar(0)->GetRangeMax());
123 
124  unsigned nTotalBins = fMaxBin - fMinBin + 1;
125  unsigned nBinsPerFitPoint = nTotalBins / fNFitPoints; // integer division rounds down; there may be bins leftover unused
126 
127  KTDEBUG(gvlog, "Performing gain variation fits with " << fNFitPoints << " points, and " << nBinsPerFitPoint << " bins averaged per fit point.");
128 
129  unsigned nComponents = data.GetNComponents();
130 
131  //double sigmaNorm = 1. / double(nBinsPerFitPoint - 1);
132  for (unsigned iComponent=0; iComponent<nComponents; iComponent++)
133  {
134  const KTFrequencySpectrumPolar* spectrum = data.GetSpectrumPolar(iComponent);
135 
136  double* xVals = new double[fNFitPoints];
137  double* yVals = new double[fNFitPoints];
138 
139  // Calculate fit points
140 #pragma omp parallel for default(shared)
141  for (unsigned iFitPoint=0; iFitPoint < fNFitPoints; iFitPoint++)
142  {
143  unsigned fitPointStartBin = iFitPoint * nBinsPerFitPoint + fMinBin;
144  unsigned fitPointEndBin = fitPointStartBin + nBinsPerFitPoint;
145 
146  double leftEdge = spectrum->GetBinLowEdge(fitPointStartBin);
147  double rightEdge = spectrum->GetBinLowEdge(fitPointEndBin);
148  xVals[iFitPoint] = leftEdge + 0.5 * (rightEdge - leftEdge);
149 
150  double mean = 0.;
151  for (unsigned iBin=fitPointStartBin; iBin<fitPointEndBin; iBin++)
152  {
153  mean += (*spectrum)(iBin).abs();
154  }
155  mean /= (double)nBinsPerFitPoint;
156  yVals[iFitPoint] = mean;
157 
158  KTDEBUG(gvlog, "Fit point " << iFitPoint << " " << xVals[iFitPoint] << " " << yVals[iFitPoint]);
159  }
160 
161  if (fNormalize)
162  {
163  // Normalize the fit points to 1
164  double minYVal = yVals[0];
165  for (unsigned iFitPoint=1; iFitPoint < fNFitPoints; iFitPoint++)
166  {
167  if (yVals[iFitPoint] < minYVal) minYVal = yVals[iFitPoint];
168  }
169  for (unsigned iFitPoint=0; iFitPoint < fNFitPoints; iFitPoint++)
170  {
171  yVals[iFitPoint] = yVals[iFitPoint] / minYVal;
172  }
173  }
174 
175  KTSpline* spline = new KTSpline(xVals, yVals, fNFitPoints);
176  spline->SetXMin(fMinFrequency);
177  spline->SetXMax(fMaxFrequency);
178  //GainVariation* gainVarResult = CreateGainVariation(spline, spectrum->GetNBins(), spectrum->GetRangeMin(), spectrum->GetRangeMax());
179 
180  delete [] xVals;
181  delete [] yVals;
182 
183  newData.SetSpline(spline, iComponent);
184  //newData->SetGainVariation(gainVarResult, iComponent);
185  }
186  KTINFO(gvlog, "Completed gain variation calculation for " << nComponents);
187 
188  return true;
189  }
