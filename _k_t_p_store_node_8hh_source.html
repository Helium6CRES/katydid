166  template< typename XType >
167  XType KTPStoreNode::GetData(const std::string& dataName) const
168  {
169  KTPStoreNodeDataNotFound dnfException;
170  TreeNode::const_assoc_iterator it = fTree->find(dataName);
171  if (it == fTree->not_found())
172  {
173  KTERROR(utillog_psnode, "No subnode was found called <" << dataName << ">.");
174  throw dnfException;
175  }
176 
177  try
178  {
179  // get_value will only return data from this node (whereas get can return data from subnodes)
180  return it->second.get_value< XType >();
181  }
182  catch (boost::property_tree::ptree_bad_path& e)
183  {
184  KTERROR(utillog_psnode, "Subnode <" << dataName << "> did not contain data.");
185  throw dnfException;
186  }
187  catch (boost::property_tree::ptree_bad_data& e)
188  {
189  KTERROR(utillog_psnode, "Unable to convert to the specified type for parameter named <" << dataName << ">.");
190  throw dnfException;
191  }
192  KTERROR(utillog_psnode, "Unknown error while attempting to retrieve <" << dataName << ">.");
193  throw dnfException;
194  }
