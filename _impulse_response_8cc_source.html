120 Bool_t KTImpulseAnalysis::Analyze(KTFrequencySpectrumDataPolar& fsData)
121 {
122  KTFrequencySpectrumPolar* fs = fsData.GetSpectrumPolar(0);
123  UInt_t size = fs->size();
124  Double_t binWidth = fs->GetBinWidth();
125 
126  // Loop over all the bins in the FS
127  // Calculate the sum and keep the peak bin information
128  UInt_t peakBin = 0, previousPeakBin = 0;
129  Double_t peakBinValue = -1.;
130  Double_t previousPeakValue = -1.;
131  Double_t sum = 0.; // sum of squares, since we want to calculate the power
132  Double_t value;
133  for (UInt_t iBin=0; iBin < size; iBin++)
134  {
135  value = (*fs)(iBin).abs();
136  sum += value * value;
137  if (value > peakBinValue)
138  {
139  previousPeakValue = peakBinValue;
140  previousPeakBin = peakBin;
141  peakBinValue = value;
142  peakBin = iBin;
143  }
144  }
145 
146  Double_t peakFraction = peakBinValue * peakBinValue / sum;
147  Double_t leakagePeakBin = 1. - peakFraction;
148  Double_t peakThreeBinFraction = (peakBinValue*peakBinValue + (*fs)(peakBin-1).abs()*(*fs)(peakBin-1).abs() + (*fs)(peakBin+1).abs()*(*fs)(peakBin+1).abs()) / sum;
149  Double_t leakagePeakThreeBin = 1. - peakThreeBinFraction;
150  Double_t secondHighestBinRatio = previousPeakValue / peakBinValue;
151  //KTDEBUG(irlog, peakFraction << " " << leakagePeakBin << " " << peakThreeBinFraction << " " << leakagePeakThreeBin << " " << peakBinValue << " " << peakThreeBinValue << " " << sum);
152 
153  // Examine fractional peak width
154  Double_t fraction = 0.1;
155  UInt_t leftSideBin = peakBin, rightSideBin = peakBin;
156  if (peakBinValue > 0.)
157  {
158  Double_t fractionalPowerValue = sqrt(fraction) * peakBinValue;
159  while (leftSideBin > 0 && (*fs)(leftSideBin).abs() >= fractionalPowerValue)
160  {
161  leftSideBin--;
162  }
163  while (rightSideBin < size-1 && (*fs)(rightSideBin).abs() >= fractionalPowerValue)
164  {
165  rightSideBin++;
166  }
167  if (leftSideBin != 0 && (*fs)(leftSideBin).abs() < fractionalPowerValue) leftSideBin++; // if we didn't hit the left edge, we went one beyond the FWHM peak
168  if (rightSideBin != size-1 && (*fs)(rightSideBin).abs() < fractionalPowerValue) rightSideBin--; // if we didn't heit the right edge, we went one beyond the FWHM peak
169  }
170 
171  UInt_t fracWidthBins = rightSideBin - leftSideBin + 1;
172  Double_t fracWidth = Double_t(fracWidthBins) * binWidth;
173 
174  KTPROG(irlog, "Frequency of peak: " << fs->GetBinCenter(peakBin) << " Hz (bin # " << peakBin << ")");
175  KTPROG(irlog, "Leakage fraction (1 bin): " << leakagePeakBin);
176  KTPROG(irlog, "Leakage fraction (3 bin): " << leakagePeakThreeBin);
177  KTPROG(irlog, "Second-highest-bin ratio: " << secondHighestBinRatio << " (bin # " << previousPeakBin << ")");
178  KTPROG(irlog, fraction * 100. << "% width: " << fracWidth << " Hz (" << fracWidthBins << " bins)");
179 
180  return true;
181 }
