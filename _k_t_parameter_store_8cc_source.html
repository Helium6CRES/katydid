117  void KTParameterStore::PrintSubTree(const PStoreTree* tree, const string& addressOfTree, stringstream* printStream) const
118  {
119  for (PStoreTree::const_iterator it = tree->begin(); it != tree->end(); it++)
120  {
121  string addressOfNode;

123  else addressOfNode = it->first;
124 
125  if (it->second.get_value< string >().length() > 0)
