81  bool KTGainNormalization::Normalize(KTFrequencySpectrumDataPolar& fsData, KTGainVariationData& gvData)
82  {
83  if (fCalculateMinBin) SetMinBin(fsData.GetSpectrumPolar(0)->FindBin(fMinFrequency));
84  if (fCalculateMaxBin) SetMaxBin(fsData.GetSpectrumPolar(0)->FindBin(fMaxFrequency));
85 
86  unsigned nComponents = fsData.GetNComponents();
87  if (nComponents != gvData.GetNComponents())
88  {
89  KTERROR(gnlog, "Mismatch in the number of channels between the frequency spectrum data and the gain variation data! Aborting.");
90  return false;
91  }
92 
93  KTNormalizedFSDataPolar& newData = fsData.Of< KTNormalizedFSDataPolar >().SetNComponents(nComponents);
94 
95  for (unsigned iComponent=0; iComponent<nComponents; iComponent++)
96  {
97  KTFrequencySpectrumPolar* newSpectrum = Normalize(fsData.GetSpectrumPolar(iComponent), gvData.GetSpline(iComponent));
98  if (newSpectrum == NULL)
99  {
100  KTERROR(gnlog, "Normalization of spectrum " << iComponent << " failed for some reason. Continuing processing.");
101  continue;
102  }
103  else
104  {
105  KTDEBUG(gnlog, "Computed normalization; size: " << newSpectrum->size() << "; range: " << newSpectrum->GetRangeMin() << " - " << newSpectrum->GetRangeMax());
106  }
107  newData.SetSpectrum(newSpectrum, iComponent);
108  }
109  KTINFO(gnlog, "Completed gain normalization of " << nComponents << " frequency spectra (polar)");
110 
111  return true;

147  KTFrequencySpectrumPolar* KTGainNormalization::Normalize(const KTFrequencySpectrumPolar* frequencySpectrum, const KTSpline* spline)
148  {
149  unsigned nBins = fMaxBin - fMinBin + 1;
150  double freqMin = frequencySpectrum->GetBinLowEdge(fMinBin);
151  double freqMax = frequencySpectrum->GetBinLowEdge(fMaxBin) + frequencySpectrum->GetBinWidth();
152 
153  KTSpline::Implementation* splineImp = spline->Implement(nBins, freqMin, freqMax);
154 
155  unsigned nSpectrumBins = frequencySpectrum->size();
156  double freqSpectrumMin = frequencySpectrum->GetRangeMin();
157  double freqSpectrumMax = frequencySpectrum->GetRangeMax();
158 
159  KTDEBUG(gnlog, "Creating new FS for normalized data: " << nSpectrumBins << ", " << freqSpectrumMin << ", " << freqSpectrumMax);
160  KTFrequencySpectrumPolar* newSpectrum = new KTFrequencySpectrumPolar(nSpectrumBins, freqSpectrumMin, freqSpectrumMax);
161 
162  // First directly copy data that's outside the scaling range
163  unsigned iBin;
164 #pragma omp parallel default(shared)
165  {
166 #pragma omp for private(iBin)
167  for (iBin=0; iBin < fMinBin; iBin++)
168  {
169  (*newSpectrum)(iBin).set_polar((*frequencySpectrum)(iBin).abs(), (*frequencySpectrum)(iBin).arg());
170  }
171 #pragma omp for private(iBin)
172  for (iBin=fMaxBin+1; iBin < nSpectrumBins; iBin++)
173  {
174  (*newSpectrum)(iBin).set_polar((*frequencySpectrum)(iBin).abs(), (*frequencySpectrum)(iBin).arg());
175  }
176 
177  // Then scale the bins within the scaling range
178 #pragma omp for private(iBin)
179  for (iBin=fMinBin; iBin < fMaxBin+1; iBin++)
180  {
181  (*newSpectrum)(iBin).set_polar((*frequencySpectrum)(iBin).abs() / (*splineImp)(iBin - fMinBin), (*frequencySpectrum)(iBin).arg());
182  }
183  }
184 
185  spline->AddToCache(splineImp);
186 
187  return newSpectrum;
188  }
