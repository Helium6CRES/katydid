82  Bool_t KTCorrelator::CoreCorrelate(KTFrequencySpectrumDataPolarCore& data, KTCorrelationData& newData)
83  {
84  UInt_t iPair = 0;
85  for (PairVector::const_iterator iter = fPairs.begin(); iter != fPairs.end(); iter++)
86  {
87  UInt_t firstChannel = (*iter).first;
88  UInt_t secondChannel = (*iter).second;
89  KTFrequencySpectrumPolar* result = DoCorrelation(data.GetSpectrumPolar(firstChannel), data.GetSpectrumPolar(secondChannel));
90  if (result == NULL)
91  {
92  KTWARN(corrlog, "Something went wrong with the correlation of channels " << firstChannel << " and " << secondChannel);
93  }
94  else
95  {
96  newData.SetSpectrum(result, iPair);
97  newData.SetInputPair(firstChannel, secondChannel, iPair);
98  }
99  iPair++;
100  }
101 
102  KTINFO(corrlog, "Correlations complete; " << iPair << " channel-pairs correlated.");
103  return true;
104  }

131  KTFrequencySpectrumPolar* KTCorrelator::DoCorrelation(const KTFrequencySpectrumPolar* firstSpectrum, const KTFrequencySpectrumPolar* secondSpectrum)
132  {
133  // Performs cc(firstSpectrum) * secondSpectrum
134  KTFrequencySpectrumPolar* newSpect = new KTFrequencySpectrumPolar(*firstSpectrum);
135  newSpect->CConjugate();
136  (*newSpect) *= (*secondSpectrum);
137  return newSpect;
