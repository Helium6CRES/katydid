82  bool KTCorrelator::CoreCorrelate(KTFrequencySpectrumDataPolarCore& data, KTCorrelationData& newData)
83  {
84  unsigned iPair = 0;
85  for (PairVector::const_iterator iter = fPairs.begin(); iter != fPairs.end(); iter++)
86  {
87  unsigned firstChannel = (*iter).first;
88  unsigned secondChannel = (*iter).second;
89  KTFrequencySpectrumPolar* result = DoCorrelation(data.GetSpectrumPolar(firstChannel), data.GetSpectrumPolar(secondChannel));
90  if (result == NULL)

93  }
94  else
95  {
96  newData.SetSpectrum(result, iPair);

98  }
99  iPair++;

131  KTFrequencySpectrumPolar* KTCorrelator::DoCorrelation(const KTFrequencySpectrumPolar* firstSpectrum, const KTFrequencySpectrumPolar* secondSpectrum)
132  {
133  // Performs cc(firstSpectrum) * secondSpectrum
134  KTFrequencySpectrumPolar* newSpect = new KTFrequencySpectrumPolar(*firstSpectrum);
135  newSpect->CConjugate();
