176  inline double KTComplexFFTW::GetMinFrequency(double timeBinWidth) const
177  {
178  // There's one bin at the center, always: the DC bin.
179  // # of bins on the negative side is nFreqBins/2 (rounded down because of integer division).
180  // 0.5 is added to the # of bins because of the half of the DC bin on the negative frequency side.
181  return -GetFrequencyBinWidth(timeBinWidth) * (double(fSize/2) + 0.5);
182  }
183 
184  inline double KTComplexFFTW::GetMaxFrequency(double timeBinWidth) const
185  {
186  // There's one bin at the center, always: the DC bin.
187  // # of bins on the positive side is nFreqBins/2 if the number of bins is odd, and nFreqBins/2-1 if the number of bins is even (division rounded down because of integer division).
188  // 0.5 is added to the # of bins because of the half of the DC bin on the positive frequency side.
189  unsigned nBinsToSide = fSize / 2;
190  return GetFrequencyBinWidth(timeBinWidth) * (double(nBinsToSide*2 == fSize ? nBinsToSide - 1 : nBinsToSide) + 0.5);
191  }
