177  inline Double_t KTComplexFFTW::GetMinFrequency(Double_t timeBinWidth) const
178  {
179  // There's one bin at the center, always: the DC bin.
180  // # of bins on the negative side is nFreqBins/2 (rounded down because of integer division).
181  // 0.5 is added to the # of bins because of the half of the DC bin on the negative frequency side.
182  return -GetFrequencyBinWidth(timeBinWidth) * (Double_t(fSize/2) + 0.5);
183  }

185  inline Double_t KTComplexFFTW::GetMaxFrequency(Double_t timeBinWidth) const
186  {
187  // There's one bin at the center, always: the DC bin.
188  // # of bins on the positive side is nFreqBins/2 if the number of bins is odd, and nFreqBins/2-1 if the number of bins is even (division rounded down because of integer division).
189  // 0.5 is added to the # of bins because of the half of the DC bin on the positive frequency side.
190  UInt_t nBinsToSide = fSize / 2;
191  return GetFrequencyBinWidth(timeBinWidth) * (Double_t(nBinsToSide*2 == fSize ? nBinsToSide - 1 : nBinsToSide) + 0.5);
192  }
