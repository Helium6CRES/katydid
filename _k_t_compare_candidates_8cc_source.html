101  eventMatches[eventCounter] = eventMatches[eventCounter] + 1;
102  candidateMatches[candidateCounter] = candidateMatches[candidateCounter] + 1;
103  //KTDEBUG(cclog, "Match found:\n" <<
104  // "\tEvent " << eventCounter << "): " << truthIt->fStartRecord << ", " << truthIt->fStartSample << ", " << truthIt->fEndRecord << ", " << truthIt->fEndSample << '\n' <<

106  }
107  else if (comparison < 0)
108  {
109  //KTDEBUG(cclog, "Candidate before event:\n" <<
110  // "\tEvent: " << truthIt->fStartRecord << ", " << truthIt->fStartSample << ", " << truthIt->fEndRecord << ", " << truthIt->fEndSample << '\n' <<
111  // "\tCandidate: " << candIt->fStartRecord << ", " << candIt->fStartSample << ", " << candIt->fEndRecord << ", " << candIt->fEndSample << '\n');
112  if (fAssumeSparseCandidates)

116  candidateCounterStart = candidateCounter;
117  // move up to the next event as the place to start

119  candidateCounterStart++;
120  // if this is the last candidate, than there's no use in doing any more comparisons to events
121  if (candStartHere == candidates.end())
122  {
123  continueEventLoop = false;
124  // if we're here, the candidate loop will complete after this cycle
125  }
126  }
127  }
128  else // comparison > 0
129  {
130  //KTDEBUG(cclog, "Candidate after event:\n" <<
131  // "\tEvent: " << truthIt->fStartRecord << ", " << truthIt->fStartSample << ", " << truthIt->fEndRecord << ", " << truthIt->fEndSample << '\n' <<
132  // "\tCandidate: " << candIt->fStartRecord << ", " << candIt->fStartSample << ", " << candIt->fEndRecord << ", " << candIt->fEndSample << '\n');
133  // all remaining candidates are after the current truth event, so they can be skipped
134  break;
135  }
136  candidateCounter++;
137  }

139  }

168  }

171 
172  KTPROG(cclog, "Number of events: " << ccrData.GetNEvents());

209

213  fCCResultsSignal(dataPtr);

219  {

224

231  event.fEndRecord * eventRecordSize + event.fEndSample)
232  return 1;

236  return -1;
