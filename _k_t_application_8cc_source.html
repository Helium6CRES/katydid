73  void KTApplication::AddConfigOptionsToCLHandler(const KTParameterStore::PStoreTree* tree, const string& addressOfTree)
74  {
75  BOOST_FOREACH( const KTParameterStore::PStoreTree::value_type& treeNode, tree->get_child("") )
76  {
77  KTParameterStore::PStoreTree subtree = treeNode.second;
78  string addressOfNode;
79  if (addressOfTree.size() > 0) addressOfNode = addressOfTree + "." + treeNode.first;
80  else addressOfNode = treeNode.first;
81 
82  if (tree->get< string >(treeNode.first).length() > 0)
83  {
84  // Add this address to the CLHandler
85  string helpMsg = "Configuration option: " + addressOfNode;
86  fCLHandler->AddOption< string >("Config File Options", helpMsg, addressOfNode, false);
87  }
88 
89  // Recursively go down the hierarchy
90  AddConfigOptionsToCLHandler(&subtree, addressOfNode);
91  }
92  return;
