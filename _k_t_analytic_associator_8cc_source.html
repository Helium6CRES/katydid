69  bool KTAnalyticAssociator::CreateAssociateData(KTTimeSeriesData& tsData)
70  {
71  if (! fFullFFT.GetIsInitialized())
72  {
73  fFullFFT.InitializeFFT();

82 
83  KTFrequencySpectrumDataFFTW* fsData = NULL;
84  if (fSaveFrequencySpectrum)

87  }
88 
89  // New data to hold the time series of the analytic associate
90  KTAnalyticAssociateData& aaTSData = tsData.Of< KTAnalyticAssociateData >().SetNComponents(nComponents);
91 
92  // Calculate the analytic associates
93  for (unsigned iComponent = 0; iComponent < nComponents; iComponent++)
94  {
95  const KTTimeSeriesFFTW* nextInput = dynamic_cast< const KTTimeSeriesFFTW* >(tsData.GetTimeSeries(iComponent));
96  if (nextInput == NULL)
97  {
98  KTERROR(aalog, "Incorrect time series type: time series did not cast to KTTimeSeriesFFTW. Other types of time series data are not yet supported.");
99  return false;
100  }
101 
102  KTFrequencySpectrumFFTW* newFS = NULL;
103  KTTimeSeriesFFTW* newTS = NULL;
104  if (fSaveFrequencySpectrum)
105  {
106  newTS = CalculateAnalyticAssociate(nextInput, &newFS);
107  fsData->SetSpectrum(newFS, iComponent);
108  }
109  else
110  {
111  newTS = CalculateAnalyticAssociate(nextInput);
112  }
113 
114  if (newTS == NULL)
115  {
116  KTERROR(aalog, "Component <" << iComponent << "> did not transform correctly.");
117  return false;
118  }
119 
120  aaTSData.SetTimeSeries(newTS, iComponent);
121  }
122  KTINFO(aalog, "Analytic associate calculation (from TS) complete for " << nComponents << " components");
123 
124  return true;

202  KTTimeSeriesFFTW* KTAnalyticAssociator::CalculateAnalyticAssociate(const KTTimeSeriesFFTW* inputTS, KTFrequencySpectrumFFTW** outputFS)
203  {
204  // Forward FFT
205  KTFrequencySpectrumFFTW* freqSpec = fFullFFT.Transform(inputTS);
206  if (freqSpec == NULL)
207  {
208  KTERROR(aalog, "Something went wrong with the forward FFT on the time series.");
209  return NULL;
210  }
211 
212  // Calculate the analytic associate in frequency space
213  freqSpec->AnalyticAssociate();
214 
215  // reverse FFT
216  KTTimeSeriesFFTW* outputTS = fFullFFT.Transform(freqSpec);
217 
218  // copy the address of the frequency spectrum to outputFS if it's being kept; otherwise delete
219  if (fSaveFrequencySpectrum) outputFS = &freqSpec;
220  else delete freqSpec;
221 
222  if (outputTS == NULL)
223  {
224  KTERROR(aalog, "Something went wrong with the reverse FFT on the frequency spectrum.");
225  }
226 
227  return outputTS;
228  }
