119

157  Bool_t KTPolyphaseFilterBank::TransferHeaderInformation(const KTSliceHeader& oldHeader, KTSliceHeader& newHeader)
158  {
159  newHeader.SetNComponents(oldHeader.GetNComponents());
160  newHeader.SetSampleRate(oldHeader.GetSampleRate());
