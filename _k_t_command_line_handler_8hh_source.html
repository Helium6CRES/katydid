203  bool KTCommandLineHandler::AddOption(const std::string& aTitle, const std::string& aHelpMsg, const std::string& aLongOpt, char aShortOpt, bool aWarnOnDuplicate)
204  {
205  if (fAllOptionsLong.find(aLongOpt) != fAllOptionsLong.end())
206  {
207  if (aWarnOnDuplicate)
208  KTWARN(utillog_clh, "There is already an option called <" << aLongOpt << ">");
209  return false;
210  }
211  if (fAllOptionsShort.find(aShortOpt) != fAllOptionsShort.end())
212  {
213  if (aWarnOnDuplicate)
214  KTWARN(utillog_clh, "There is already a short option called <" << aShortOpt << ">");
215  return false;
216  }
217 
218  // option is okay at this point
219 
220  OptDescMapIt tIter = fProposedGroups.find(aTitle);
221  if (tIter == fProposedGroups.end())
222  {
223  tIter = CreateNewOptionGroup(aTitle);
224  }
225 
226  std::string tOptionName = aLongOpt;
227  fAllOptionsLong.insert(aLongOpt);
228  tOptionName += "," + std::string(&aShortOpt);
229  fAllOptionsShort.insert(aShortOpt);
230 
231  tIter->second->add_options()(tOptionName.c_str(), po::value< XValueType >(), aHelpMsg.c_str());
232 
233  return true;
