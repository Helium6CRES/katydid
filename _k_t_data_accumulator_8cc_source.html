86  bool KTDataAccumulator::CoreAddTSDataReal(KTTimeSeriesData& data, Accumulator& accDataStruct, KTTimeSeriesData& accData)
87  {
88  double remainingFrac = 1.;
89  if (accDataStruct.fCount >= fAccumulatorSize)
90  remainingFrac -= fAveragingFrac;
91  //KTDEBUG(avlog, "averaging frac: " << fAveragingFrac << " remaining frac: " << remainingFrac);
92 
93  unsigned nComponents = data.GetNComponents();
94 
95  if (accDataStruct.fCount == 0)
96  {
97  accData.SetNComponents(nComponents);
98  for (unsigned iComponent = 0; iComponent < nComponents; ++iComponent)
99  {
100  KTTimeSeriesReal* dataTS = static_cast< KTTimeSeriesReal* >(data.GetTimeSeries(iComponent));
101  KTTimeSeriesReal* newTS = new KTTimeSeriesReal(dataTS->GetNTimeBins(), dataTS->GetRangeMin(), dataTS->GetRangeMax());
102  (*newTS) *= 0.;
103  accData.SetTimeSeries(newTS, iComponent);
104  }
105  }
106 
107  ++accDataStruct.fCount;
108  ++accDataStruct.fSignalCount;
109 
110  if (nComponents != accData.GetNComponents())
111  {
112  KTERROR(avlog, "Numbers of components in the average and in the new data do not match");
113  return false;
114  }
115 
116  unsigned arraySize = data.GetTimeSeries(0)->GetNTimeBins();
117  if (arraySize != accData.GetTimeSeries(0)->GetNTimeBins())
118  {
119  KTERROR(avlog, "Sizes of arrays in the average and in the new data do not match");
120  return false;
121  }
122 
123  for (unsigned iComponent = 0; iComponent < nComponents; ++iComponent)
124  {
125  KTTimeSeriesReal* newTS = static_cast< KTTimeSeriesReal* >(data.GetTimeSeries(iComponent));
126  KTTimeSeriesReal* avTS = static_cast< KTTimeSeriesReal* >(accData.GetTimeSeries(iComponent));
127  for (unsigned iBin = 0; iBin < arraySize; iBin++)
128  {
129  //KTDEBUG(avlog, (*avTS)(iBin) << " " << (*newTS)(iBin) << " " << remainingFrac << " " << fAveragingFrac);
130  (*avTS)(iBin) = (*avTS)(iBin) * remainingFrac + (*newTS)(iBin) * fAveragingFrac;
131  //KTDEBUG(avlog, (*avTS)(iBin));
132  }
133  }
134 
135  return true;
136  }
