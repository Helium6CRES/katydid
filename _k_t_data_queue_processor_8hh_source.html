253  template< class XProcessorType >
254  bool KTDataQueueProcessorTemplate< XProcessorType >::ProcessQueue()
255  {
256  KTINFO(eqplog, "Beginning to process queue");
257  while (fStatus != kStopped)
258  {
259  KTDEBUG(eqplog, "processing . . .");
260  DataAndFunc daf;
261  if ((fQueue.*fPopFromQueue)(daf))
262  {
263  KTDEBUG(eqplog, "Data acquired for processing");
264  (static_cast<XProcessorType*>(this)->*(daf.fFuncPtr))(daf.fData);
265  if (daf.fData->fLastData) fStatus = kStopped;
266  }
267  else
268  {
269  if (fStatus == kContinue)
270  {
271  fStatus = kRunning;
272  }
273  else
274  {
275  fStatus = kStopped;
276  }
277  }
278  }
279  fQueueDoneSignal();
280  KTINFO(eqplog, "Queue processing has ended");
281  return true;
282  }
