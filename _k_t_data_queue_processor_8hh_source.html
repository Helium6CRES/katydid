255  template< class XProcessorType >
256  Bool_t KTDataQueueProcessorTemplate< XProcessorType >::ProcessQueue()
257  {
258  KTINFO(eqplog, "Beginning to process queue");
259  while (fStatus != kStopped)
260  {
261  KTDEBUG(eqplog, "processing . . .");
262  DataAndFunc daf;
263  if ((fQueue.*fPopFromQueue)(daf))
264  {
265  KTDEBUG(eqplog, "Data acquired for processing");
266  (static_cast<XProcessorType*>(this)->*(daf.fFuncPtr))(daf.fData);
267  if (daf.fData->fLastData) fStatus = kStopped;
268  }
269  else
270  {
271  if (fStatus == kContinue)
272  {
273  fStatus = kRunning;
274  }
275  else
276  {
277  fStatus = kStopped;
278  }
279  }
280  }
281  fQueueDoneSignal();
282  KTINFO(eqplog, "Queue processing has ended");
283  return true;
284  }
