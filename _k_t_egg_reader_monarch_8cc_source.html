145  KTDataPtr KTEggReaderMonarch::HatchNextSlice()
146  {
147  unsigned recordSize = fHeader.GetRecordSize();
148 
149  if (fMonarch == NULL)
150  {
151  KTERROR(eggreadlog, "Monarch file has not been opened");
152  return KTDataPtr();
153  }
154  if (fReadState.fStatus == MonarchReadState::kInvalid)
155  {
156  KTERROR(eggreadlog, "Read state status is <invalid>. Did you hatch the egg first?");
157  return KTDataPtr();
158  }
159 
160  if (fReadState.fStatus == MonarchReadState::kAtStartOfRun)
161  {
162  KTDEBUG(eggreadlog, "Reading first record");
163  // if we're at the beginning of the run, load the first records
164  if (! fMonarch->ReadRecord())
165  {
166  KTERROR(eggreadlog, "File appears to contain no slices.");
167  return KTDataPtr();
168  }
169  fReadState.fReadPtrOffset = 0;
170  fReadState.fReadPtrRecordOffset = 0;
171  fReadState.fSliceStartPtrOffset = 0;
172  fReadState.fAbsoluteRecordOffset = 0;
173  fSliceNumber = 0;
174  }
175  else
176  {
177  ++fSliceNumber;
178 
179  if (fReadState.fStatus == MonarchReadState::kReachedNextRecord)
180  {
181  KTDEBUG(eggreadlog, "Slice and record boundaries coincided; reading new record before creating new slice");
182  // if the slice boundary and record boundary coincide, we need to load the next record because the final position
183  // of the read pointer from the previous slice is at the sample after the end of the previous slice,
184  // which in this case would be the start of the next record.
185  if (! fMonarch->ReadRecord())
186  {
187  KTWARN(eggreadlog, "End of egg file reached after reading new records (or something else went wrong)");
188  return KTDataPtr();
189  }
190  ++(fReadState.fAbsoluteRecordOffset);
191  fReadState.fReadPtrOffset = 0;
192  ++(fReadState.fReadPtrRecordOffset);
193  fReadState.fStatus = MonarchReadState::kContinueReading;
194  }
195 
196  // shift the slice start pointer by the stride
197  // note that this pointer refers to the record in which the previous slice started
198  fReadState.fSliceStartPtrOffset += fStride;
199  unsigned sliceStartRecordOffset = 0; // how many records to shift to the start of the slice
200  while (fReadState.fSliceStartPtrOffset >= recordSize)
201  {
202  fReadState.fSliceStartPtrOffset -= recordSize;
203  ++sliceStartRecordOffset;
204  }
205 
206  // Calculate whether we need to move the read pointer to a different record by subtracing the number
207  // of records read in the last slice (fReadPtrRecordOffset)
208  // If this is 0, it doesn't need to be moved
209  // If it's > 0, then it needs to be reduced by 1 because of how the offset number in Monarch::ReadRecord is used (offset=0 will advance to the next record)
210  int readPtrRecordOffsetShift = int(sliceStartRecordOffset) - int(fReadState.fReadPtrRecordOffset);
211  if (readPtrRecordOffsetShift != 0)
212  {
213  // change the absolute record offset first because it should be done before the adjustment to Monarch::ReadRecord offset counting is made
214  fReadState.fAbsoluteRecordOffset += readPtrRecordOffsetShift;
215  if (readPtrRecordOffsetShift != 0)
216  {
217  --readPtrRecordOffsetShift;
218  KTDEBUG(eggreadlog, "Reading new record with offset " << readPtrRecordOffsetShift);
219  // move the read pointer to the slice start pointer (first move monarch to the correct record)
220  if (! fMonarch->ReadRecord(readPtrRecordOffsetShift))
221  {
222  KTWARN(eggreadlog, "End of egg file reached after reading new records (or something else went wrong)");
223  return KTDataPtr();
224  }
225  }
226  }
227  // Move the read pointer to the slice start pointer within the record
228  fReadState.fReadPtrRecordOffset = 0;
229  fReadState.fReadPtrOffset = fReadState.fSliceStartPtrOffset;
230  }
231 
232  KTDataPtr newData(new KTData());
233 
234  // Fill out slice header information
235  KTSliceHeader& sliceHeader = newData->Of< KTSliceHeader >().SetNComponents(fHeader.GetNChannels());
236  if (fReadState.fStatus == MonarchReadState::kAtStartOfRun)
237  {
238  sliceHeader.SetIsNewAcquisition(true);
239  }
240  else
241  {
242  sliceHeader.SetIsNewAcquisition(false);
243  }
244  sliceHeader.SetSampleRate(fHeader.GetAcquisitionRate());
245  sliceHeader.SetSliceSize(fSliceSize);
246  sliceHeader.CalculateBinWidthAndSliceLength();
247  sliceHeader.SetNonOverlapFrac((double)fStride / (double)fSliceSize);
248  sliceHeader.SetTimeInRun(GetTimeInRun());
249  sliceHeader.SetSliceNumber(fSliceNumber);
250  sliceHeader.SetStartRecordNumber(fReadState.fAbsoluteRecordOffset);
251  sliceHeader.SetStartSampleNumber(fReadState.fReadPtrOffset);
252  sliceHeader.SetRecordSize(fHeader.GetRecordSize());
253  KTDEBUG(eggreadlog, sliceHeader << "\nNote: some fields may not be filled in correctly yet");
254 
255  // Setup pointers to monarch and new katydid records
256  unsigned nChannels = fHeader.GetNChannels();
257  vector< const MonarchRecord* > monarchRecords(nChannels);
258  vector< KTTimeSeries* > newRecords(nChannels);
259  for (unsigned iChannel = 0; iChannel < nChannels; ++iChannel)
260  {
261  monarchRecords[iChannel] = (fMonarch->*fMonarchGetRecord[iChannel])();
262  sliceHeader.SetAcquisitionID(monarchRecords[iChannel]->fAcquisitionId, iChannel);
263  sliceHeader.SetRecordID(monarchRecords[iChannel]->fRecordId, iChannel);
264  sliceHeader.SetTimeStamp(monarchRecords[iChannel]->fTime, iChannel);
265 
266  //tsData->SetTimeSeries(new vector< DataType >(monarchRecord->fDataPtr, monarchRecord->fDataPtr+header->GetSliceSize()), iChannel);
267  KTTimeSeries* newRecord;
268  if (fTimeSeriesType == kRealTimeSeries)
269  {
270  newRecord = new KTTimeSeriesReal(fSliceSize, 0., double(fSliceSize) * sliceHeader.GetBinWidth());
271  }
272  else
273  {
274  newRecord = new KTTimeSeriesFFTW(fSliceSize, 0., double(fSliceSize) * sliceHeader.GetBinWidth());
275  }
276  newRecords[iChannel] = newRecord;
277  }
278 
279  KTDEBUG(eggreadlog, "Time in run: " << GetTimeInRun() << " s\n" <<
280  "\tBin width = " << fBinWidth << '\n' <<
281  "\tMonarch record size = " << fRecordSize << '\n' <<
282  "\tRecord offset = " << fReadState.fAbsoluteRecordOffset << '\n' <<
283  "\tSlice start pointer offset = " << fReadState.fSliceStartPtrOffset << '\n' <<
284  "\tRead pointer record offset = " << fReadState.fReadPtrRecordOffset << '\n' <<
285  "\tRead pointer offset = " << fReadState.fReadPtrOffset);
286 
287  if (fReadState.fStatus == MonarchReadState::kAtStartOfRun)
288  {
289  fReadState.fStatus = MonarchReadState::kContinueReading;
290  }
291 
292  // Loop over bins
293  for (unsigned iBin = 0; iBin < fSliceSize; ++iBin)
294  {
295  if (fReadState.fStatus == MonarchReadState::kReachedNextRecord)
296  {
297  KTDEBUG(eggreadlog, "Reading new record mid-slice");
298  // try reading the next record
299  if (! fMonarch->ReadRecord())
300  {
301  // the end of the file has been reached or there was some other error preventing the reading of the next record
302  KTWARN(eggreadlog, "End of egg file reached after reading new records (or something else went wrong)");
303  for (unsigned iChannel = 0; iChannel < nChannels; ++iChannel)
304  {
305  delete newRecords[iChannel];
306  }
307  return KTDataPtr();
308  }
309  ++(fReadState.fAbsoluteRecordOffset);
310  fReadState.fReadPtrOffset = 0;
311  ++(fReadState.fReadPtrRecordOffset);
312 
313  // check if the acquisition ID has changed on any channel
314  if (fReadState.fAcquisitionID != monarchRecords[0]->fAcquisitionId)
315  {
316  KTDEBUG(eggreadlog, "Acquisition ID change; resetting slice to start with this monarch record.\n"
317  << "\tNumber of unused bins: " << iBin - 1);
318  // this means that a new monarch record has come up, and it has a different acquisition id.
319  // in this situation we need to start the time series over with the new monarch record
320  // first, update the read state
321  fReadState.fAcquisitionID = monarchRecords[0]->fAcquisitionId;
322  fReadState.fReadPtrOffset = 0;
323  fReadState.fReadPtrRecordOffset = 0;
324  fReadState.fSliceStartPtrOffset = 0;
325  // reset slice data
326  sliceHeader.SetIsNewAcquisition(true);
327  sliceHeader.SetStartRecordNumber(fReadState.fAbsoluteRecordOffset);
328  sliceHeader.SetStartSampleNumber(fReadState.fReadPtrOffset);
329  for (unsigned iChannel = 0; iChannel < nChannels; ++iChannel)
330  {
331  sliceHeader.SetAcquisitionID(monarchRecords[iChannel]->fAcquisitionId, iChannel);
332  sliceHeader.SetRecordID(monarchRecords[iChannel]->fRecordId, iChannel);
333  sliceHeader.SetTimeStamp(monarchRecords[iChannel]->fTime, iChannel);
334  }
335  // reset bin count to 0
336  iBin = 0;
337  // change the time in run since we're going back to the beginning of the record
338  sliceHeader.SetTimeInRun(GetTimeInRun());
339  KTDEBUG(eggreadlog, "Correction to time in run: " << GetTimeInRun() << " s\n" <<
340  "\tBin width = " << fBinWidth << '\n' <<
341  "\tMonarch record size = " << fRecordSize << '\n' <<
342  "\tRecord offset = " << fReadState.fAbsoluteRecordOffset << '\n' <<
343  "\tSlice start pointer offset = " << fReadState.fSliceStartPtrOffset << '\n' <<
344  "\tRead pointer record offset = " << fReadState.fReadPtrRecordOffset << '\n' <<
345  "\tRead pointer offset = " << fReadState.fReadPtrOffset);
346  sliceHeader.SetSliceNumber(fSliceNumber);
347  }
348 
349  fReadState.fStatus = MonarchReadState::kContinueReading;
350  }
351 
352  // Read the data from the records
353  for (unsigned iChannel = 0; iChannel < nChannels; ++iChannel)
354  {
355  // set the data
356  newRecords[iChannel]->SetValue(iBin, double(monarchRecords[iChannel]->fData[fReadState.fReadPtrOffset]));
357  }
358 
359  // advance the pointer for the next bin
360  ++(fReadState.fReadPtrOffset);
361 
362  // check if we've reached the end of a monarch record
363  if (fReadState.fReadPtrOffset >= recordSize)
364  {
365  KTDEBUG(eggreadlog, "End of Monarch record reached.");
366  fReadState.fStatus = MonarchReadState::kReachedNextRecord;
367  }
368  } // end loop over bins
369 
370  sliceHeader.SetEndRecordNumber(fReadState.fAbsoluteRecordOffset);
371  sliceHeader.SetEndSampleNumber(fReadState.fReadPtrOffset - 1);
372 
373 
374  // finally, set the records in the new data object
375  KTTimeSeriesData& tsData = newData->Of< KTTimeSeriesData >().SetNComponents(nChannels);
376  for (unsigned iChannel = 0; iChannel < nChannels; ++iChannel)
377  {
378  tsData.SetTimeSeries(newRecords[iChannel], iChannel);
379  }
380 
381  return newData;
382  }
