117  boost::shared_ptr< KTData > KTEggReaderMonarch::HatchNextSlice()
118  {
119  UInt_t recordSize = fHeader.GetRecordSize();
120 
121  if (fMonarch == NULL)
122  {
123  KTERROR(eggreadlog, "Monarch file has not been opened");
124  return shared_ptr< KTData >();
125  }
126  if (fReadState.fStatus == MonarchReadState::kInvalid)
127  {
128  KTERROR(eggreadlog, "Read state status is <invalid>. Did you hatch the egg first?");
129  return shared_ptr< KTData >();
130  }
131 
132  if (fReadState.fStatus == MonarchReadState::kAtStartOfRun)
133  {
134  KTDEBUG(eggreadlog, "Reading first record");
135  // if we're at the beginning of the run, load the first records
136  if (! fMonarch->ReadRecord())
137  {
138  KTERROR(eggreadlog, "File appears to contain no slices.");
139  return shared_ptr< KTData >();
140  }
141  fReadState.fReadPtrOffset = 0;
142  fReadState.fReadPtrRecordOffset = 0;
143  fReadState.fSliceStartPtrOffset = 0;
144  fReadState.fAbsoluteRecordOffset = 0;
145  fSliceNumber = 0;
146  fReadState.fStatus = MonarchReadState::kContinueReading;
147  }
148  else
149  {
150  fSliceNumber++;
151 
152  if (fReadState.fStatus == MonarchReadState::kReachedNextRecord)
153  {
154  KTDEBUG(eggreadlog, "Slice and record boundaries coincided; reading new record before creating new slice");
155  // if the slice boundary and record boundary coincide, we need to load the next record
156  if (! fMonarch->ReadRecord())
157  {
158  KTWARN(eggreadlog, "End of egg file reached after reading new records (or something else went wrong)");
159  return shared_ptr< KTData >();
160  }
161  fReadState.fAbsoluteRecordOffset++;
162  fReadState.fReadPtrOffset = 0;
163  fReadState.fReadPtrRecordOffset++;
164  fReadState.fStatus = MonarchReadState::kContinueReading;
165  }
166 
167  // shift the slice start pointer by the stride
168  fReadState.fSliceStartPtrOffset += fStride;
169  UInt_t sliceStartRecordOffset = 0; // how many records to shift to the start of the slice
170  while (fReadState.fSliceStartPtrOffset >= recordSize)
171  {
172  fReadState.fSliceStartPtrOffset -= recordSize;
173  sliceStartRecordOffset++;
174  }
175 
176  // Calculate whether we need to move the read pointer to a different record
177  // If this is 0, it doesn't need to be moved
178  // If it's > 0, then it needs to be reduced by 1 because of how the offset number in Monarch::ReadRecord is used (offset=0 will advance to the next record)
179  Int_t readPtrRecordOffsetShift = Int_t(sliceStartRecordOffset) - Int_t(fReadState.fReadPtrRecordOffset);
180  if (readPtrRecordOffsetShift != 0)
181  {
182  // change the absolute record offset first because it should be done before the adjustment to Monarch::ReadRecord offset counting is made
183  fReadState.fAbsoluteRecordOffset += readPtrRecordOffsetShift;
184  if (readPtrRecordOffsetShift > 0) readPtrRecordOffsetShift--;
185  KTDEBUG(eggreadlog, "Reading new record with offset " << readPtrRecordOffsetShift);
186  // move the read pointer to the slice start pointer (first move monarch to the correct record)
187  if (! fMonarch->ReadRecord(readPtrRecordOffsetShift))
188  {
189  KTWARN(eggreadlog, "End of egg file reached after reading new records (or something else went wrong)");
190  return shared_ptr< KTData >();
191  }
192  }
193  // Move the read pointer to the slice start pointer within the record
194  fReadState.fReadPtrRecordOffset = 0;
195  fReadState.fReadPtrOffset = fReadState.fSliceStartPtrOffset;
196  }
197 
198  boost::shared_ptr< KTData > newData(new KTData());
199 
200  // Fill out slice header information
201  KTSliceHeader& sliceHeader = newData->Of< KTSliceHeader >().SetNComponents(fHeader.GetNChannels());
202  sliceHeader.SetSampleRate(fHeader.GetAcquisitionRate());
203  sliceHeader.SetSliceSize(fSliceSize);
204  sliceHeader.CalculateBinWidthAndSliceLength();
205  sliceHeader.SetTimeInRun(GetTimeInRun());
206  sliceHeader.SetSliceNumber(fSliceNumber);
207  sliceHeader.SetStartRecordNumber(fReadState.fAbsoluteRecordOffset);
208  sliceHeader.SetStartSampleNumber(fReadState.fReadPtrOffset);
209  KTDEBUG(eggreadlog, "Filled out slice header:\n"
210  << "\tSample rate: " << sliceHeader.GetSampleRate() << " Hz\n"
211  << "\tSlice size: " << sliceHeader.GetSliceSize() << '\n'
212  << "\tBin width: " << sliceHeader.GetBinWidth() << " s\n"
213  << "\tSlice length: " << sliceHeader.GetSliceLength() << " s\n"
214  << "\tTime in run: " << sliceHeader.GetTimeInRun() << " s\n"
215  << "\tSlice number: " << sliceHeader.GetSliceNumber() << '\n'
216  << "\tStart record number: " << sliceHeader.GetStartRecordNumber() << '\n'
217  << "\tStart sample number: " << sliceHeader.GetStartSampleNumber());
218 
219  // Normalization of the record values
220  Double_t normalization = fFullVoltageScale / (Double_t)fNADCLevels;
221 
222  // Setup pointers to monarch and new katydid records
223  UInt_t nChannels = fHeader.GetNChannels();
224  vector< const MonarchRecord* > monarchRecords(nChannels);
225  vector< KTTimeSeries* > newRecords(nChannels);
226  for (UInt_t iChannel = 0; iChannel < nChannels; iChannel++)
227  {
228  monarchRecords[iChannel] = fMonarch->GetRecord(iChannel);
229  sliceHeader.SetAcquisitionID(monarchRecords[iChannel]->fAId, iChannel);
230  sliceHeader.SetRecordID(monarchRecords[iChannel]->fRId, iChannel);
231  sliceHeader.SetTimeStamp(monarchRecords[iChannel]->fTick, iChannel);
232 
233  //tsData->SetTimeSeries(new vector< DataType >(monarchRecord->fDataPtr, monarchRecord->fDataPtr+header->GetSliceSize()), iChannel);
234  KTTimeSeries* newRecord;
235  if (fTimeSeriesType == kRealTimeSeries)
236  {
237  newRecord = new KTTimeSeriesReal(fSliceSize, 0., Double_t(fSliceSize) * sliceHeader.GetBinWidth());
238  }
239  else
240  {
241  newRecord = new KTTimeSeriesFFTW(fSliceSize, 0., Double_t(fSliceSize) * sliceHeader.GetBinWidth());
242  }
243  newRecords[iChannel] = newRecord;
244  }
245 
246  KTDEBUG(eggreadlog, "Time in run: " << GetTimeInRun() << " s\n" <<
247  "\tBin width = " << fBinWidth << '\n' <<
248  "\tMonarch record size = " << fRecordSize << '\n' <<
249  "\tRecord offset = " << fReadState.fAbsoluteRecordOffset << '\n' <<
250  "\tSlice start pointer offset = " << fReadState.fSliceStartPtrOffset << '\n' <<
251  "\tRead pointer record offset = " << fReadState.fReadPtrRecordOffset << '\n' <<
252  "\tRead pointer offset = " << fReadState.fReadPtrOffset);
253 
254  // Loop over bins
255  for (UInt_t iBin = 0; iBin < fSliceSize; iBin++)
256  {
257  if (fReadState.fStatus == MonarchReadState::kReachedNextRecord)
258  {
259  KTDEBUG(eggreadlog, "Reading new record mid-slice");
260  // try reading the next record
261  if (! fMonarch->ReadRecord())
262  {
263  // the end of the file has been reached or there was some other error preventing the reading of the next record
264  KTWARN(eggreadlog, "End of egg file reached after reading new records (or something else went wrong)");
265  for (UInt_t iChannel = 0; iChannel < nChannels; iChannel++)
266  {
267  delete newRecords[iChannel];
268  }
269  return shared_ptr< KTData >();
270  }
271  fReadState.fAbsoluteRecordOffset++;
272  fReadState.fReadPtrOffset = 0;
273  fReadState.fReadPtrRecordOffset++;
274 
275  // check if the acquisition ID has changed on any channel
276  if (fReadState.fAcquisitionID != monarchRecords[0]->fAId)
277  {
278  KTDEBUG(eggreadlog, "Acquisition ID change; resetting slice to start with this monarch record.\n"
279  << "\tNumber of unused bins: " << iBin - 1);
280  // this means that a new monarch record has come up, and it has a different acquisition id.
281  // in this situation we need to start the time series over with the new monarch record
282  // first, update the read state
283  fReadState.fAcquisitionID = monarchRecords[0]->fAId;
284  fReadState.fReadPtrOffset = 0;
285  fReadState.fReadPtrRecordOffset = 0;
286  fReadState.fSliceStartPtrOffset = 0;
287  // reset slice data
288  sliceHeader.SetStartRecordNumber(fReadState.fAbsoluteRecordOffset);
289  sliceHeader.SetStartSampleNumber(fReadState.fReadPtrOffset);
290  for (UInt_t iChannel = 0; iChannel < nChannels; iChannel++)
291  {
292  sliceHeader.SetAcquisitionID(monarchRecords[iChannel]->fAId, iChannel);
293  sliceHeader.SetRecordID(monarchRecords[iChannel]->fRId, iChannel);
294  sliceHeader.SetTimeStamp(monarchRecords[iChannel]->fTick, iChannel);
295  }
296  // reset bin count to 0
297  iBin = 0;
298  // change the time in run since we're going back to the beginning of the record
299  sliceHeader.SetTimeInRun(GetTimeInRun());
300  KTDEBUG(eggreadlog, "Correction to time in run: " << GetTimeInRun() << " s\n" <<
301  "\tBin width = " << fBinWidth << '\n' <<
302  "\tMonarch record size = " << fRecordSize << '\n' <<
303  "\tRecord offset = " << fReadState.fAbsoluteRecordOffset << '\n' <<
304  "\tSlice start pointer offset = " << fReadState.fSliceStartPtrOffset << '\n' <<
305  "\tRead pointer record offset = " << fReadState.fReadPtrRecordOffset << '\n' <<
306  "\tRead pointer offset = " << fReadState.fReadPtrOffset);
307  sliceHeader.SetSliceNumber(fSliceNumber);
308  }
309 
310  fReadState.fStatus = MonarchReadState::kContinueReading;
311  }
312 
313  // Read the data from the records
314  for (UInt_t iChannel = 0; iChannel < nChannels; iChannel++)
315  {
316  // set the data
317  newRecords[iChannel]->SetValue(iBin, Double_t(monarchRecords[iChannel]->fDataPtr[fReadState.fReadPtrOffset]) * normalization);
318  }
319 
320  // advance the pointer for the next bin
321  fReadState.fReadPtrOffset++;
322 
323  // check if we've reached the end of a monarch record
324  if (fReadState.fReadPtrOffset >= recordSize)
325  {
326  KTDEBUG(eggreadlog, "End of Monarch record reached.");
327  fReadState.fStatus = MonarchReadState::kReachedNextRecord;
328  }
329  } // end loop over bins
330 
331  sliceHeader.SetEndRecordNumber(fReadState.fAbsoluteRecordOffset);
332  sliceHeader.SetEndSampleNumber(fReadState.fReadPtrOffset - 1);
333 
334 
335  // finally, set the records in the new data object
336  KTTimeSeriesData& tsData = newData->Of< KTTimeSeriesData >().SetNComponents(nChannels);
337  for (UInt_t iChannel = 0; iChannel < nChannels; iChannel++)
338  {
339  tsData.SetTimeSeries(newRecords[iChannel], iChannel);
340  }
341 
342  return newData;
343  }
