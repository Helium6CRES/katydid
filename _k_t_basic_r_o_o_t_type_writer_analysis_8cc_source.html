56  void KTBasicROOTTypeWriterAnalysis::WriteCorrelationData(const KTCorrelationData* data)
57  {
58  KTBundle* bundle = data->GetBundle();
59  UInt_t bundleNumber = 0;
60  if (bundle != NULL) bundleNumber = bundle->GetBundleNumber();
61  UInt_t nPairs = data->GetNPairs();
62 
63  if (! fWriter->OpenAndVerifyFile()) return;
64 
65  for (unsigned iPair=0; iPair<nPairs; iPair++)
66  {
67  const KTFrequencySpectrumPolar* spectrum = data->GetCorrelation(iPair);
68  if (spectrum != NULL)
69  {
70  stringstream conv;
71  conv << "histCorr_" << bundleNumber << "_" << iPair;
72  string histName;
73  conv >> histName;
74  TH1D* corrHist = spectrum->CreateMagnitudeHistogram(histName);
75  stringstream titleStream;
76  titleStream << "Bundle " << bundleNumber << ", Correlation " << iPair << ", "
77  "Channels (" << data->GetFirstChannel(iPair) << ", " << data->GetSecondChannel(iPair) << ")";
78  corrHist->SetTitle(titleStream.str().c_str());
79  corrHist->SetDirectory(fWriter->GetFile());
80  corrHist->Write();
81  KTDEBUG(publog, "Histogram <" << histName << "> written to ROOT file");
82  }
83  }
