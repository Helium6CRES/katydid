71  void KTBasicROOTTypeWriterAnalysis::WriteNormalizedFSDataPolar(boost::shared_ptr<KTData> data)
72  {
73  if (! data) return;
74 
75  ULong64_t sliceNumber = data->Of<KTSliceHeader>().GetSliceNumber();
76 
77  KTNormalizedFSDataPolar& fsData = data->Of<KTNormalizedFSDataPolar>();
78  UInt_t nComponents = fsData.GetNComponents();
79 
80  if (! fWriter->OpenAndVerifyFile()) return;
81 
82  for (unsigned iChannel=0; iChannel<nComponents; iChannel++)
83  {
84  const KTFrequencySpectrumPolar* spectrum = fsData.GetSpectrumPolar(iChannel);
85  if (spectrum != NULL)
86  {
87  stringstream conv;
88  conv << "histNFSpolar_" << sliceNumber << "_" << iChannel;
89  string histName;
90  conv >> histName;
91  TH1D* powerSpectrum = spectrum->CreateMagnitudeHistogram(histName);
92  powerSpectrum->SetDirectory(fWriter->GetFile());
93  powerSpectrum->Write();
94  KTDEBUG(publog, "Histogram <" << histName << "> written to ROOT file");
95  }
96  }
