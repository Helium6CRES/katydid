87  void KTComplexFFTW::InitializeFFT()
88  {
89  // fTransformFlag is guaranteed to be valid in the Set method.
90  KTDEBUG(fftlog_comp, "Transform flag: " << fTransformFlag);
91  TransformFlagMap::const_iterator iter = fTransformFlagMap.find(fTransformFlag);
92  unsigned transformFlag = iter->second;
93 
94  // allocate the input and output arrays if they're not there already
95  AllocateArrays();
96 
97  if (fUseWisdom)
98  {
99  KTDEBUG(fftlog_comp, "Reading wisdom from file <" << fWisdomFilename << ">");
100  if (fftw_import_wisdom_from_filename(fWisdomFilename.c_str()) == 0)
101  {
102  KTWARN(fftlog_comp, "Unable to read FFTW wisdom from file <" << fWisdomFilename << ">");
103  }
104  }
105 
106  InitializeMultithreaded();
107 
108  KTDEBUG(fftlog_comp, "Creating plan: " << fSize << " bins; forward FFT");
109  fForwardPlan = fftw_plan_dft_1d(fSize, fInputArray, fOutputArray, FFTW_FORWARD, transformFlag | FFTW_PRESERVE_INPUT);
110  KTDEBUG(fftlog_comp, "Creating plan: " << fSize << " bins; backward FFT");
111  fReversePlan = fftw_plan_dft_1d(fSize, fInputArray, fOutputArray, FFTW_BACKWARD, transformFlag | FFTW_PRESERVE_INPUT);
112 
113  if (fForwardPlan != NULL && fReversePlan != NULL)
114  {
115  fIsInitialized = true;
116  // delete the input and output arrays to save memory, since they're not needed for the transform
117  FreeArrays();
118  if (fUseWisdom)
119  {
120  if (fftw_export_wisdom_to_filename(fWisdomFilename.c_str()) == 0)
121  {
122  KTWARN(fftlog_comp, "Unable to write FFTW wisdom to file <" << fWisdomFilename << ">");
123  }
124  }
125  KTDEBUG(fftlog_comp, "FFTW plans created; Initialization complete.");
126  }
127  else
128  {
129  fIsInitialized = false;
130  if (fForwardPlan == NULL)
131  {
132  KTERROR(fftlog_comp, "Unable to create the forward FFT plan! FFT is not initialized.");
133  }
134  if (fReversePlan == NULL)
135  {
136  KTERROR(fftlog_comp, "Unable to create the reverse FFT plan! FFT is not initialized.");
137  }
138  }
139  return;
140  }

149  bool KTComplexFFTW::TransformData(KTTimeSeriesData& tsData)
150  {
151  if (tsData.GetTimeSeries(0)->GetNTimeBins() != GetSize())
152  {
153  SetSize(tsData.GetTimeSeries(0)->GetNTimeBins());
154  InitializeFFT();
155  }
156 
157  if (! fIsInitialized)
158  {
159  KTERROR(fftlog_comp, "FFT must be initialized before the transform is performed\n"
160  << " Please first call InitializeFFT(), then perform the transform.");
161  return false;
162  }
163 
164  unsigned nComponents = tsData.GetNComponents();
165 
166  KTFrequencySpectrumDataFFTW& newData = tsData.Of< KTFrequencySpectrumDataFFTW >().SetNComponents(nComponents);
167 
168  for (unsigned iComponent = 0; iComponent < nComponents; iComponent++)
169  {
170  const KTTimeSeriesFFTW* nextInput = dynamic_cast< const KTTimeSeriesFFTW* >(tsData.GetTimeSeries(iComponent));
171  if (nextInput == NULL)
172  {
173  KTERROR(fftlog_comp, "Incorrect time series type: time series did not cast to KTTimeSeriesFFTW.");
174  return false;
175  }
176 
177  KTFrequencySpectrumFFTW* nextResult = Transform(nextInput);
178 
179  if (nextResult == NULL)
180  {
181  KTERROR(fftlog_comp, "Channel <" << iComponent << "> did not transform correctly.");
182  return false;
183  }
184  KTDEBUG(fftlog_comp, "FFT computed; size: " << nextResult->size() << "; range: " << nextResult->GetRangeMin() << " - " << nextResult->GetRangeMax());
185  newData.SetSpectrum(nextResult, iComponent);
186  }
187 
188  KTINFO(fftlog_comp, "FFT complete; " << nComponents << " channel(s) transformed");
189 
190  return true;
