189  bool KTEggWriter::WriteTSData(KTSliceHeader& slHeader, KTTimeSeriesData& tsData)
190  {
191  if (fFileStatus == kClosed)
192  {
193  KTERROR(eggwritelog, "Cannot write record because the file has not been opened");
194  return false;
195  }
196 
197  if (fFileStatus == kOpened)
198  {
199  KTWARN(eggwritelog, "Writing record on file with no header");
200  }
201  fFileStatus = kWritingRecords;
202 
203  unsigned nComponents = tsData.GetNComponents();
204  if (nComponents != fExpectedNChannels)
205  {
206  KTERROR(eggwritelog, "Received data contains " << nComponents << " channels of data; " << fExpectedNChannels << " were expected");
207  return false;
208  }
209 
210  if (fExpectedNChannels > 2)
211  {
212  KTERROR(eggwritelog, "Interleaved record writing is only supported for 2 or fewer channels, not " << fExpectedNChannels);
213  return false;
214  }
215 
216  fMonarch->SetInterface(sInterfaceSeparate);
217 
218  CopyATimeSeries(0, slHeader, tsData, fMonarch->GetRecordSeparateOne());
219 
220  if (fExpectedNChannels == 2)
221  {
222  CopyATimeSeries(1, slHeader, tsData, fMonarch->GetRecordSeparateTwo());
223  }
224 
225  try
226  {
227  fMonarch->WriteRecord();
228  }
229  catch (MonarchException& e)
230  {
231  KTERROR(eggwritelog, "A problem occurred while writing the record (interleaved): " << e.what());
232  return false;
233  }
234  return true;

237  bool KTEggWriter::CopyATimeSeries(unsigned component, const KTSliceHeader& slHeader, const KTTimeSeriesData& tsData, MonarchRecord* record)
238  {
239  const KTTimeSeries* ts = tsData.GetTimeSeries(component);
240  if (ts->GetNTimeBins() != fExpectedRecordSize)
241  {
242  KTERROR(eggwritelog, "Time series does not have the correct size (should be " << fExpectedRecordSize << "; provided component 0: " << ts->GetNTimeBins());
243  return false;
244  }
245 
246  record->fAcquisitionId = slHeader.GetAcquisitionID(component);
247  record->fRecordId = slHeader.GetRecordID(component);
248  record->fTime = slHeader.GetTimeStamp(component);
249 
250  double value;
251  double scale = 255. / fDigitizerFullscale;
252  for (unsigned iBin = 0; iBin < fExpectedRecordSize; iBin++)
253  {
254  value = ts->GetValue(iBin) * scale;
255  if (value >= 256) value = 255.;
256  if (value < 0.) value = 0.;
257  record->fData[iBin] = DataType(floor(value));
258  }
259  return true;
260  }
