216  void KTWignerVille::CalculateACF(Buffer::iterator data1It, const Buffer::iterator& data2End, unsigned /*iWindow*/)
217  {
218  // data2It will be decremented before it's used
219  Buffer::iterator data2It = data2End + fWindowSize;
220 
221  register double t1_real;
222  register double t1_imag;
223  register double t2_real;
224  register double t2_imag;
225 
226  //KTERROR(wvlog, "iWindow = " << iWindow);
227 
228  for (unsigned fftBin = 0; fftBin < fWindowSize; fftBin++)
229  {
230  // decrement data2It first so that it doesn't run past begin() on the first window
231  --data2It;
232  t1_real = data1It->real();
233  t1_imag = data1It->imag();
234  t2_real = data2It->real();
235  t2_imag = data2It->imag();
236  (*fInputArray)(fftBin)[0] = t1_real * t2_real + t1_imag * t2_imag;
237  (*fInputArray)(fftBin)[1] = t1_imag * t2_real - t1_real * t2_imag;
238  //KTWARN(wvlog, " " << fftBin << " -- " << t1_real << " " << t1_imag << " -- " << t2_real << " " << t2_imag << " -- " << (*fInputArray)(fftBin)[0] << " " << (*fInputArray)(fftBin)[1]);
239  ++data1It;
240  }
241 
242  if (fUseWindowFunction)
243  {
244  fWindower->ApplyWindow(fInputArray);
245  }
246
