50  Bool_t KTSwitchFFTWPolar::SwitchToPolar(KTFrequencySpectrumDataFFTW& fsData)
51  {
52  UInt_t nComponents = fsData.GetNComponents();
53 
54  KTFrequencySpectrumDataPolar& newData = fsData.Of< KTFrequencySpectrumDataPolar >().SetNComponents(nComponents);
55 
56  for (UInt_t iComponent=0; iComponent<nComponents; iComponent++)
57  {
58  KTFrequencySpectrumPolar* newSpectrum = fsData.GetSpectrumFFTW(iComponent)->CreateFrequencySpectrumPolar();
59  if (newSpectrum == NULL)
60  {
61  KTERROR(swlog, "Switch of spectrum " << iComponent << " (fftw->polar) failed for some reason. Continuing processing.");
62  continue;
63  }
64  else
65  {
66  KTFrequencySpectrumFFTW* oldSpectrum = fsData.GetSpectrumFFTW(iComponent);
67  KTDEBUG(swlog, "fftw array: " << oldSpectrum->size() << " bins; range: " << oldSpectrum->GetRangeMin() << " - " << oldSpectrum->GetRangeMax());
68  KTDEBUG(swlog, "polar array: " << newSpectrum->size() << " bins; range: " << newSpectrum->GetRangeMin() << " - " << newSpectrum->GetRangeMax());
69  }
70  newData.SetSpectrum(newSpectrum, iComponent);
71  }
72  KTINFO(swlog, "Completed switch (fftw->polar) of " << nComponents << " frequency spectra (polar)");
73 
74  return true;
75  }
