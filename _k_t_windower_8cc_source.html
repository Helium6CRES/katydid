150  Bool_t KTWindower::ApplyWindow(KTTimeSeriesReal* data) const
151  {
152  UInt_t nBins = data->size();
153  if (nBins != fWindowFunction->GetSize())
154  {
155  KTWARN(windowlog, "Number of bins in the data provided does not match the number of bins set for this window\n"
156  << " Bin expected: " << fWindowFunction->GetSize() << "; Bins in data: " << nBins);
157  return false;
158  }
159 
160  for (UInt_t iBin=0; iBin < nBins; iBin++)
