167  bool KTWindower::ApplyWindow(KTTimeSeriesReal* ts) const
168  {
169  unsigned nBins = ts->size();
170  if (nBins != fWindowFunction->GetSize())
171  {
172  KTWARN(windowlog, "Number of bins in the data provided does not match the number of bins set for this window\n"
173  << " Bin expected: " << fWindowFunction->GetSize() << "; Bins in data: " << nBins);
174  return false;
175  }
176 
177  for (unsigned iBin=0; iBin < nBins; ++iBin)
178  {
179  (*ts)(iBin) = (*ts)(iBin) * fWindowFunction->GetWeight(iBin);
180  }
