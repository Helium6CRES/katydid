65  Bool_t KTDistanceClustering::FindClusters(KTDiscriminatedPoints1DData& data)
66  {
67  typedef KTDiscriminatedPoints1DData::SetOfPoints OriginalPoints;
68  typedef set< UInt_t > Cluster;
69 
70  if (fCalculateMaxBinDistance) SetMaxBinDistance(KTMath::Nint(fMaxFrequencyDistance / data.GetBinWidth()));
71 
72  UInt_t nComponents = data.GetNComponents();
73 
74  KTCluster1DData& newData = data.Of< KTCluster1DData >().SetNComponents(nComponents);
75  newData.SetNBins(data.GetNBins());
76  newData.SetBinWidth(data.GetBinWidth());
77 
78  for (UInt_t iComponent=0; iComponent<nComponents; iComponent++)
79  {
80  newData.SetThreshold(data.GetThreshold(iComponent), iComponent);
81 
82  const OriginalPoints points = data.GetSetOfPoints(iComponent);
83 
84  if (! points.empty())
85  {
86  OriginalPoints::const_iterator pIt = points.begin();
87  UInt_t thisPoint = pIt->first;
88  Cluster activeCluster;
89  activeCluster.insert(thisPoint);
90  UInt_t lastPointInActiveCluster = thisPoint;
91 
92  for (pIt++; pIt != points.end(); pIt++)
93  {
94  thisPoint = pIt->first;
95  if (thisPoint - lastPointInActiveCluster > fMaxBinDistance)
96  {
97  //KTDEBUG(sdlog, "Adding cluster (ch. " << iChannel << "): " << *(activeCluster.begin()) << " " << *(activeCluster.rbegin()));
98  newData.AddCluster(*(activeCluster.begin()), *(activeCluster.rbegin()), iComponent);
99  activeCluster.clear();
100  }
101  activeCluster.insert(thisPoint);
102  lastPointInActiveCluster = thisPoint;
103  }
104  KTDEBUG(sdlog, "Adding cluster: (ch. " << iComponent << "): " << *(activeCluster.begin()) << " " << *(activeCluster.rbegin()));
105  newData.AddCluster(*(activeCluster.begin()), *(activeCluster.rbegin()), iComponent);
106  }
107 
108  KTINFO(sdlog, newData.GetSetOfClusters(iComponent).size() << " clusters added on channel " << iComponent);
109  }
