108  void KTRQProcessor::ProcessNoiseBundle(boost::shared_ptr<KTBundle> bundle)
109  {
110  // Grab time series data from bundle
111  const KTTimeSeriesData* noise = bundle->GetData<KTTimeSeriesData>(fNoiseName);
112  if( noise != NULL ) {
113  if( !(this->fNACMDidConverge) ) {
114  // Grab the first channel of data out to use.
115  const KTTimeSeriesReal* noiseDt = dynamic_cast<const KTTimeSeriesReal*>(noise->GetTimeSeries(0));
116 
117  // We need to iterate over the chunks in the time series and produce our
118  // own time series. First things first, get the pointer to the raw data
119  // held in the time series.
120  unsigned nElem = (noiseDt->GetData()).data().size();
121  const double* rawPtr = (noiseDt->GetData()).data().begin();
122 
123  // Now we point the data map at the first fChunkSize piece of data and process it.
124  // For now we only use the first chunk to form the ACM. Despite the fact that this
125  // looks like allocating memory, it isn't.
126  new (this->fDataMap) DataMapType(rawPtr, this->fChunkSize);
127 
128  // Update noise estimate with current noise sample.
129  KTINFO(nrq_log,"using noise to update NACM estimate...");
130 
131  // Grab a tick to calculate elapsed time.
132  std::clock_t t0 = clock();
133 
134  // Calculate first row. The NACM is a Toeplitz matrix which means we only need to
135  // calculate the first row.
136  for(unsigned colIdx = 0; colIdx < this->fChunkSize; colIdx++) {
137  (*this->fNoiseACM)(0, colIdx) = KTRQProcessor::LaggedACF(this->fDataMap, colIdx);
138  }
139 
140  for(unsigned row = 1; row < this->fChunkSize; row++) {
141  for(unsigned col = 0; col < this->fChunkSize; col++) {
142  // If row = col, we are on the diagonal, which means this element is equal to
143  // row = 0, col = 0.
144  if( row == col ) (*this->fNoiseACM)(row,col) = (*this->fNoiseACM)(0,0);
145  // If row < col, we are below the diagonal, so use the transposed element for
146  // assignment.
147  if( col < row ) (*this->fNoiseACM)(row,col) = (*this->fNoiseACM)(col,row);
148  // If row > col, we are above the diagonal. Look above and to the left to get this
149  // element.
150  if( col > row ) (*this->fNoiseACM)(row,col) = (*this->fNoiseACM)(row-1, col-1);
151  }
152  }
153 
154  // all done, check elapsed time.
155  std::clock_t tf = clock();
156  double elapsed = (tf-t0)/CLOCKS_PER_SEC;
157  KTINFO("Noise bundle processed in " << elapsed << " seconds.");
158 
159  // Check convergence criterion (TEMPORARILY JUST TRUE) and set convergence flag if
160  // appropriate.
161  if(true) {
162  this->SetNACMConverged(true);
163  KTINFO(nrq_log,"NACM has converged.");
164  KTWARN(nrq_log,"DON'T BELIEVE CONVERGE MSG, TEMPORARY CONVERGENCE CRITERION USED.");
165  }
166  }
167  }
168  else {
169  KTWARN(nrq_log,"time series " << fNoiseName << " not found in bundle! skipping...");
170  }
171  }
