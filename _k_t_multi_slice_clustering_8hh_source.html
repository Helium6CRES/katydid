419  template< class XDataType >
420  void KTMultiSliceClustering::ProcessOneSliceData(boost::shared_ptr<KTData> data)
421  {
422  if (! data->Has< KTDiscriminatedPoints1DData >())
423  {
424  KTWARN(sclog, "No discriminated-points data was present");
425  return;
426  }
427  if (! data->Has< XDataType >())
428  {
429  KTWARN(sclog, "No frequency spectrum (polar) data was present");
430  return;
431  }
432  // signal for any continued use of the input data
433  fOneSliceDataSignal(data);
434  DataList* clusteredData = FindClusters(data->Of< KTDiscriminatedPoints1DData >(), data->Of< XDataType >(), data->Of< KTSliceHeader >());
435 
436  if (data->fLastData)
437  {
438  KTINFO(sclog, "Last input data processed; Cleaning up remaining active clusters");
439  KTMultiSliceClustering::DataList* lastData = CompleteAllClusters();
440  clusteredData->splice(clusteredData->end(), *lastData);
441  delete lastData;
442  }
443 
444  if (clusteredData != NULL)
445  {
446  RunDataLoop(clusteredData);
447  }
448  return;
449  }
