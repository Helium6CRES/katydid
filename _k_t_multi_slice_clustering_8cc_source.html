535  newCluster.fEndMinFreqPoint = fbIt->fFirstPoint;
536  newCluster.fEndMaxFreqPoint = fbIt->fLastPoint;
537  //KTDEBUG(sclog, " range is " << newCluster.fEndMinFreqPoint << " - " << newCluster.fEndMaxFreqPoint);
538  KTDEBUG(sclog, " contains " << newCluster.fPoints.size() << " points");
539  fActiveClusters[component].push_back(newCluster);
540  }
541  }
542 
543  // Last step: add this frequency spectrum into the pre-cluster spectra list; remove the oldest item in the list if necessary
544  fPreClusterSpectra[component].push_back(spectrumPtr);
545  while (fPreClusterSpectra[component].size() > fNFramingTimeBins)
546  {
547  fPreClusterSpectra[component].pop_front();
548  }
549 
550  return completeClusters; //CompleteInactiveClusters(component);

563  shared_ptr<KTData> KTMultiSliceClustering::CreateDataFromCluster(const Cluster& cluster)
564  {
565  KTDEBUG(sclog, "Creating data object # " << fDataCount);
566  fDataCount++;
567 
568  shared_ptr< KTData > data(new KTData());
569 
570  KTWaterfallCandidateData& wfcData = data->Of< KTWaterfallCandidateData >();
571  wfcData.SetComponent(cluster.fDataComponent);
572 
573  wfcData.SetStartRecordNumber(cluster.fPoints.begin()->fHeaderPtr->GetStartRecordNumber());
574  wfcData.SetStartSampleNumber(cluster.fPoints.begin()->fHeaderPtr->GetStartSampleNumber());
575  wfcData.SetEndRecordNumber(cluster.fPoints.rbegin()->fHeaderPtr->GetEndRecordNumber());
576  wfcData.SetEndSampleNumber(cluster.fPoints.rbegin()->fHeaderPtr->GetEndSampleNumber());
577 
578  UInt_t firstTimeBin = cluster.FirstTimeBin();
579  UInt_t lastTimeBin = cluster.LastTimeBin();
580 
581  Int_t firstTimeBinWithFrame = (Int_t)firstTimeBin - (Int_t)fNFramingTimeBins;
582  Int_t lastTimeBinWithFrame = (Int_t)lastTimeBin + (Int_t)fNFramingTimeBins;
583 
584  KTDEBUG(sclog, "final time range: " << firstTimeBin << " - " << lastTimeBin << "; with frame: " << firstTimeBinWithFrame << " - " << lastTimeBinWithFrame);
585 
586  Double_t ftbSumOfWeights = 0., ftbWeightedSum = 0.;
587  Double_t ltbSumOfWeights = 0., ltbWeightedSum = 0.;
588 
589  SetOfPoints::const_iterator it = cluster.fPoints.begin();
590  UInt_t firstFreqBin = it->fFreqBin;
591  UInt_t lastFreqBin = firstFreqBin;
592 
593  if (it->fTimeBin == firstTimeBin)
594  {
595  ftbSumOfWeights += it->fAmplitude;
596  ftbWeightedSum += it->fSpectrumPtr->GetBinCenter(it->fFreqBin) * it->fAmplitude;
597  }
598  if (it->fTimeBin == lastTimeBin)
599  {
600  ltbSumOfWeights += it->fAmplitude;
601  ltbWeightedSum += it->fSpectrumPtr->GetBinCenter(it->fFreqBin) * it->fAmplitude;
602  }
603 
604  for (it++; it != cluster.fPoints.end(); it++)
605  {
606  if (it->fFreqBin < firstFreqBin) firstFreqBin = it->fFreqBin;
607  if (it->fFreqBin > lastFreqBin) lastFreqBin = it->fFreqBin;
608 
609  if (it->fTimeBin == firstTimeBin)
610  {
611  ftbSumOfWeights += it->fAmplitude;
612  ftbWeightedSum += it->fSpectrumPtr->GetBinCenter(it->fFreqBin) * it->fAmplitude;
613  }
614  if (it->fTimeBin == lastTimeBin)
615  {
616  ltbSumOfWeights += it->fAmplitude;
617  ltbWeightedSum += it->fSpectrumPtr->GetBinCenter(it->fFreqBin) * it->fAmplitude;
618  }
619  }
620  KTDEBUG(sclog, "final freq range: " << firstFreqBin << " - " << lastFreqBin);
621 
622  Int_t firstFreqBinWithFrame = (Int_t)firstFreqBin - (Int_t)fNFramingFreqBins;
623  Int_t lastFreqBinWithFrame =(Int_t) lastFreqBin + (Int_t)fNFramingFreqBins;
624  UInt_t firstFreqBinToUse = firstFreqBinWithFrame >= 0 ? (UInt_t)firstFreqBinWithFrame : 0;
625  UInt_t lastFreqBinToUse = lastFreqBinWithFrame < cluster.fPoints.begin()->fSpectrumPtr->size() ? (UInt_t)lastFreqBinWithFrame : cluster.fPoints.begin()->fSpectrumPtr->size() - 1;
626  UInt_t freqBinOffset = UInt_t((Int_t)firstFreqBinToUse - firstFreqBinWithFrame);
627 
628  Double_t timeBinWidth = cluster.fPoints.begin()->fHeaderPtr->GetSliceLength();
629  Double_t freqBinWidth = cluster.fPoints.begin()->fSpectrumPtr->GetBinWidth();
630  UInt_t nTimeBins = lastTimeBin - firstTimeBin + 1;
631  UInt_t nFreqBins = lastFreqBin - firstFreqBin + 1;
632 
633  UInt_t nTimeBinsWithFrame = nTimeBins + 2 * fNFramingTimeBins;
634  UInt_t nFreqBinsWithFrame = nFreqBins + 2 * fNFramingFreqBins;
635 
636  //KTWARN(sclog, "loading spectrum pointers; expecting " << nTimeBinsWithFrame << " spectra");
637  // Create a vector of pointers to spectra, where each component is for a single time bin
638  vector< shared_ptr< KTFrequencySpectrumPolar > > spectra(nTimeBinsWithFrame);
639  // Pre-cluster frame bins
640  UInt_t frameBin = fNFramingTimeBins - cluster.fPreClusterSpectra.size();
641  for (ListOfSpectra::const_iterator losIt = cluster.fPreClusterSpectra.begin(); losIt != cluster.fPreClusterSpectra.end(); losIt++)
642  {
643  spectra[frameBin] = *losIt;
644  //KTDEBUG(sclog, "put spectrum in frame bin " << frameBin << ": " << Bool_t(spectra[frameBin]));
645  frameBin++;
646  }
647  // Cluster bins
648  for (SetOfPoints::const_iterator it = cluster.fPoints.begin(); it != cluster.fPoints.end(); it++)
649  {
650  if (! it->fSpectrumPtr)
651  {
652  KTWARN(sclog, "Spectrum pointer is NULL! " << it->fSpectrumPtr.get());
653  }
654  spectra[it->fTimeBin - firstTimeBin + fNFramingTimeBins] = it->fSpectrumPtr;
655  //KTDEBUG(sclog, "put spectrum in frame bin " << it->fTimeBin - firstTimeBin + fNFramingTimeBins << ": " << Bool_t(spectra[it->fTimeBin - firstTimeBin + fNFramingTimeBins]));
656  }
657  // Post-cluster frame bins
658  frameBin = lastTimeBin - firstTimeBin + 1 + fNFramingTimeBins;
659  for (ListOfSpectra::const_iterator losIt = cluster.fPostClusterSpectra.begin(); losIt != cluster.fPostClusterSpectra.end(); losIt++)
660  {
661  spectra[frameBin] = *losIt;
662  //KTDEBUG(sclog, "put spectrum in frame bin " << frameBin << ": " << Bool_t(spectra[frameBin]));
663  frameBin++;
664  }
665 
666  wfcData.SetTimeInRun(cluster.fPoints.begin()->fHeaderPtr->GetTimeInRun());
667  wfcData.SetFirstSliceNumber(cluster.fPoints.begin()->fHeaderPtr->GetSliceNumber());
668  wfcData.SetLastSliceNumber(cluster.fPoints.begin()->fHeaderPtr->GetSliceNumber());
669  wfcData.SetTimeLength(timeBinWidth * Double_t(nTimeBins));
670  wfcData.SetMeanStartFrequency(ftbWeightedSum / ftbSumOfWeights);
671  wfcData.SetMeanEndFrequency(ltbWeightedSum / ltbSumOfWeights);
672  wfcData.SetFrequencyWidth(freqBinWidth * Double_t(nFreqBins));
673 
674  UInt_t i = 0;
675  for (; ! spectra[i]; i++);
676  if (i != spectra.size())
677  {
678  wfcData.SetMinimumFrequency(spectra[i]->GetBinLowEdge(firstFreqBin));
679  wfcData.SetMaximumFrequency(spectra[i]->GetBinLowEdge(lastFreqBin) + freqBinWidth);
680  }
681 
682  KTDEBUG(sclog, "Creating KTTimeFrequency with " << nTimeBinsWithFrame << " time bins and " << nFreqBinsWithFrame << " freq bins; cluster dimensions are " << nTimeBins << " by " << nFreqBins);
683  KTTimeFrequency* tf = new KTTimeFrequencyPolar(nTimeBinsWithFrame, timeBinWidth * Double_t(firstTimeBinWithFrame), timeBinWidth * Double_t(firstTimeBinWithFrame + (Int_t)nTimeBins), nFreqBinsWithFrame, freqBinWidth * Double_t(firstFreqBinWithFrame), freqBinWidth * Double_t(firstFreqBinWithFrame + (Int_t)nFreqBins));
684  for (Int_t iTBin=firstTimeBinWithFrame; iTBin <= lastTimeBinWithFrame; iTBin++)
685  {
686  Int_t spectrumNum = iTBin - firstTimeBinWithFrame;
687  if (spectra[spectrumNum])
688  {
689  //KTDEBUG(sclog, " have spectrum for spectrumNum = " << spectrumNum);
690  for (UInt_t iFBin=firstFreqBinToUse; iFBin <= lastFreqBinToUse; iFBin++)
691  {
692  //KTDEBUG(sclog, " setting point at (" << spectrumNum << ", " << iFBin-firstFreqBinToUse << "), aka (" << iTBin << ", " << iFBin << "); length of spectrum: " << spectra[spectrumNum]->size());
693  tf->SetPolar(spectrumNum, iFBin - firstFreqBinToUse + freqBinOffset, spectra[spectrumNum]->GetAbs(iFBin), spectra[spectrumNum]->GetArg(iFBin));
694  }
695  }
696  /*
697  else
698  {
699  KTDEBUG(sclog, " NO SPECTRUM for spectrumNum = " << spectrumNum);
700  }
701  */
702  }
703  wfcData.SetCandidate(tf);
704 
705  return data;
706  }
