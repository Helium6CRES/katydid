138  bool KTSpectrumDiscriminator::CoreDiscriminate(KTFrequencySpectrumDataFFTWCore& data, KTDiscriminatedPoints1DData& newData)
139  {
140  if (fCalculateMinBin)
141  {
142  SetMinBin(data.GetSpectrumFFTW(0)->FindBin(fMinFrequency));
143  KTDEBUG(sdlog, "Minimum bin set to " << fMinBin);
144  }
145  if (fCalculateMaxBin)
146  {
147  SetMaxBin(data.GetSpectrumFFTW(0)->FindBin(fMaxFrequency));
148  KTDEBUG(sdlog, "Maximum bin set to " << fMaxBin);
149  }
150 
151  unsigned nComponents = data.GetNComponents();
152 
153  newData.SetNBins(data.GetSpectrumFFTW(0)->size());
154  newData.SetBinWidth(data.GetSpectrumFFTW(0)->GetBinWidth());
155 
156  // Interval: [fMinBin, fMaxBin)
157  unsigned nBins = fMaxBin - fMinBin + 1;
158  double sigmaNorm = 1. / double(nBins - 1);
159 
160  // Temporary storage for magnitude values
161  vector< double > magnitude(data.GetSpectrumFFTW(0)->size());
162 
163  for (unsigned iComponent=0; iComponent<nComponents; iComponent++)
164  {
165  const KTFrequencySpectrumFFTW* spectrum = data.GetSpectrumFFTW(iComponent);
166  if (spectrum == NULL)
167  {
168  KTERROR(sdlog, "Frequency spectrum pointer (component " << iComponent << ") is NULL!");
169  return false;
170  }
171  if (spectrum->size() != magnitude.size())
172  {
173  magnitude.resize(spectrum->size());
174  }
175 
176  double mean = 0.;
177 #pragma omp parallel for reduction(+:mean)
178  for (unsigned iBin=fMinBin; iBin<=fMaxBin; iBin++)
179  {
180  magnitude[iBin] = sqrt((*spectrum)(iBin)[0] * (*spectrum)(iBin)[0] + (*spectrum)(iBin)[1] * (*spectrum)(iBin)[1]);
181  mean += magnitude[iBin];
182  }
183  mean /= (double)nBins;
184 
185  double threshold = 0.;
186  if (fThresholdMode == eSNR_Amplitude)
187  {
188  // SNR = P_signal / P_noise = (A_signal / A_noise)^2, A_noise = mean
189  threshold = sqrt(fSNRThreshold) * mean;
190  KTDEBUG(sdlog, "Discriminator threshold for channel " << iComponent << " set at <" << threshold << "> (SNR mode)");
191  }
192  else if (fThresholdMode == eSNR_Power)
193  {
194  // SNR = P_signal / P_noise, P_noise = mean
195  threshold = fSNRThreshold * mean;
196  KTDEBUG(sdlog, "Discriminator threshold for channel " << iComponent << " set at <" << threshold << "> (SNR mode)");
197  }
198  else if (fThresholdMode == eSigma)
199  {
200  double sigma = 0., diff;
201 #pragma omp parallel for private(diff) reduction(+:sigma)
202  for (unsigned iBin=fMinBin; iBin<=fMaxBin; iBin++)
203  {
204  diff = magnitude[iBin] - mean;
205  sigma += diff * diff;
206  }
207  sigma = sqrt(sigma * sigmaNorm);
208 
209  threshold = mean + fSigmaThreshold * sigma;
210  KTDEBUG(sdlog, "Discriminator threshold for channel " << iComponent << " set at <" << threshold << "> (Sigma mode)");
211  }
212 
213  newData.SetThreshold(threshold, iComponent);
214 
215  // loop over bins, checking against the threshold
216  double value;
217 #pragma omp parallel for private(value)
218  for (unsigned iBin=fMinBin; iBin<=fMaxBin; iBin++)
219  {
220  value = magnitude[iBin];
221  if (value >= threshold) newData.AddPoint(iBin, value, iComponent);
222  }
223  KTDEBUG(sdlog, "Component " << iComponent << " has " << newData.GetSetOfPoints(iComponent).size() << " points above threshold");
224 
225  }
226  KTINFO(sdlog, "Completed discrimination on " << nComponents << " components");
227 
228  return true;
229  }
