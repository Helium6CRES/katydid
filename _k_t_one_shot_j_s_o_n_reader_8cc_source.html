111  Bool_t KTOneShotJSONReader::ReadMCTruthEventsFile(KTData& appendToData)
112  {
113  KTINFO(inlog, "Reading mc-truth-events input file: " << fFilename);
114 
115  rapidjson::Document document;
116  if (! OpenAndParseFile(document))
117  {
118  KTERROR(inlog, "A problem occurred while parsing the mc-truth-events file");
119  return false;
120  }
121 
122  if (! document["record_size"].IsUint())
123  {
124  KTERROR(inlog, "\"record_size\" value is missing or is not an unsigned integer");
125  return false;
126  }
127  UInt_t recordSize = document["record_size"].GetUint();
128 
129  if (! document["records_simulated"].IsUint())
130  {
131  KTERROR(inlog, "\"records_simulated\" value is missing or is not an unsigned integer");
132  return false;
133  }
134  UInt_t recordsSimulated = document["records_simulated"].GetUint();
135 
136  const rapidjson::Value& events = document["events"];
137  if (! events.IsArray())
138  {
139  KTERROR(inlog, "\"events\" value in the mc truth file is either missing or not an array");
140  return false;
141  }
142 
143  KTMCTruthEvents& mcTruth = appendToData.Of< KTMCTruthEvents >();
144  mcTruth.SetRecordSize(recordSize);
145  mcTruth.SetNRecords(recordsSimulated);
146 
147  for (rapidjson::Value::ConstValueIterator evIt = events.Begin(); evIt != events.End(); evIt++)
148  {
149  const rapidjson::Value& support = (*evIt)["support"];
150  if (support.IsArray())
151  {
152  UInt_t startRec = support[rapidjson::SizeType(0)].GetUint(); // explicit cast of array index to SizeType used because of abiguous overload
153  UInt_t startSample = support[rapidjson::SizeType(1)].GetUint(); // explicit cast of array index to SizeType used because of abiguous overload
154  UInt_t endRec = support[rapidjson::SizeType(2)].GetUint(); // explicit cast of array index to SizeType used because of abiguous overload
155  UInt_t endSample = support[rapidjson::SizeType(3)].GetUint(); // explicit cast of array index to SizeType used because of abiguous overload
156  KTDEBUG(inlog, "extracted (" << startRec << ", " << startSample << ", " << endRec << ", " << endSample << ")");
157  if (endRec < startRec || (endRec == startRec && endSample < startSample))
158  {
159  KTWARN(inlog, "Invalid event: (" << startRec << ", " << startSample << " --> " << endRec << ", " << endSample << ")");
160  }
161  else
162  {
163  mcTruth.AddEvent(KTMCTruthEvents::Event(startRec, startSample, endRec, endSample));
164  }
165  }
166  else
167  {
168  KTWARN(inlog, "\"support\" value is either missing or not an array");
169  }
170  }
171 
172  KTDEBUG(inlog, "new data object has " << mcTruth.GetEvents().size() << " events");
173 
174  return true;
175  }
