313  inline std::pair< unsigned, unsigned > KTSliceHeader::GetRecordSamplePairAtSample(unsigned sampleInSlice)
314  {
315  // NOTE 1: doing these both at once, we can take advantage of compiler optimization,
316  // which will most likely only perform one division operation
317  // NOTE 2: this function does not depend on the slice size; this feature is useful
318  // to be able to calculate record offsets without worrying about which slice, exactly, this header pertains to.
319  // It's in use (as of this writing) in KTWignerVille.
320  fTemp1 = fStartSampleNumber + sampleInSlice; // sample in the record + some number of record lengths
321  fTemp2 = fStartRecordNumber + fTemp1 / fRecordSize; // record number
322  fTemp3 = fTemp1 % fRecordSize; // sample in the record
323  return std::make_pair (fTemp2, fTemp3);
324  }
