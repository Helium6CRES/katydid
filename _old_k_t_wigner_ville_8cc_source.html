200  KTSlidingWindowFSDataFFTW* KTWignerVille::TransformData(const KTTimeSeriesData* data)
201  {
202  if (! GetIsInitialized())
203  {
204  KTERROR(wvlog, "The FFT for the Wigner-Ville transform is not initialized! Aborting.");
205  return NULL;
206  }
207 
208  // # of time bins in each windowed FFT, and the number of bins in the data
209  UInt_t windowSize = fWindowFunction->GetSize();
210  UInt_t dataSize = data->GetTimeSeries(0)->GetNTimeBins();
211  if (windowSize >= dataSize)
212  {
213  KTERROR(wvlog, "Window size is larger than time data: " << windowSize << " > " << dataSize << "\n" <<
214  "No transform was performed!");
215  return NULL;
216  }
217 
218  if (fPairs.empty())
219  {
220  KTWARN(wvlog, "No Wigner-Ville pairs specified; no transforms performed.");
221  return NULL;
222  }
223 
224  // cast all time series into KTTimeSeriesFFTW
225  vector< const KTTimeSeriesFFTW* > timeSeries(data->GetNTimeSeries());
226  for (UInt_t iTS=0; iTS < timeSeries.size(); iTS++)
227  {
228  timeSeries[iTS] = dynamic_cast< const KTTimeSeriesFFTW* >(data->GetTimeSeries(iTS));
229  if (timeSeries[iTS] == NULL)
230  {
231  KTERROR(wvlog, "Time series " << iTS << " did not cast to a const KTTimeSeriesFFTW*. No transforms performed.");
232  return NULL;
233  }
234  }
235 
236  // Characteristics of the whole windowed FFT
237  UInt_t windowShift = windowSize - GetEffectiveOverlap();
238  UInt_t nWindows = (dataSize - windowSize) / windowShift + 1; // integer arithmetic gets truncated to the nearest integer
239  UInt_t nTimeBinsUsed = windowSize + (nWindows - 1) * windowShift;
240  UInt_t nTimeBinsNotUsed = dataSize - nTimeBinsUsed;
241 
242  // Characteristics of the frequency spectrum
243  Double_t timeBinWidth = data->GetTimeSeries(0)->GetTimeBinWidth();
244  Double_t freqMin = GetMinFrequency(timeBinWidth);
245  Double_t freqMax = GetMaxFrequency(timeBinWidth);
246 
247  Double_t timeMin = 0.;
248  Double_t timeMax = nTimeBinsUsed * timeBinWidth;
249 
250  KTDEBUG(wvlog, "Performing windowed FFT characteristics:\n"
251  << "\tWindow size: " << windowSize << '\n'
252  << "\tWindow shift: " << windowShift << '\n'
253  << "\t# of windows: " << nWindows << '\n'
254  << "\t# of unused bins: " << nTimeBinsNotUsed)
255 
256  //KTSlidingWindowFSData* newData = new KTSlidingWindowFSData(fPairs.size());
257  KTSlidingWindowFSDataFFTW* newData = new KTSlidingWindowFSDataFFTW(fPairs.size());
258 
259  // Do WV transform for each pair
260  UInt_t iPair = 0;
261  for (PairVector::const_iterator pairIt = fPairs.begin(); pairIt != fPairs.end(); pairIt++)
262  {
263  UInt_t firstChannel = (*pairIt).first;
264  UInt_t secondChannel = (*pairIt).second;
265 
266  //KTPhysicalArray< 1, KTFrequencySpectrumPolar* >* newSpectra = new KTPhysicalArray< 1, KTFrequencySpectrumPolar* >(nWindows, timeMin, timeMax);
267  KTPhysicalArray< 1, KTFrequencySpectrumFFTW* >* newSpectra = new KTPhysicalArray< 1, KTFrequencySpectrumFFTW* >(nWindows, timeMin, timeMax);
268 
269  KTDEBUG(wvlog, "Performing windowed FFT for channels " << firstChannel << " and " << secondChannel)
270 
271  UInt_t windowStart = 0;
272  for (UInt_t iWindow = 0; iWindow < nWindows; iWindow++)
273  {
274  //KTDEBUG(wvlog, "Window: " << iWindow << "; first bin: " << windowStart);
275  CrossMultiplyToInputArray(timeSeries[firstChannel], timeSeries[secondChannel], windowStart);
276  fftw_execute(fFTPlan);
277  (*newSpectra)(iWindow) = ExtractTransformResult(freqMin, freqMax);
278  windowStart += windowShift;
279  }
280  KTINFO(wvlog, "Windowed FFT complete (channels " << firstChannel << " and " << secondChannel << "); windows used: " << nWindows << "; time bins not used: " << nTimeBinsNotUsed);
281  newData->SetSpectra(newSpectra, iPair);
282  iPair++;
283  }
284 
285  newData->SetBundle(data->GetBundle());
286  newData->SetName(fOutputDataName);
287 
288  fWVSignal(newData);
289 
290  return newData;
