157  KTFrequencyCandidateData::Candidates KTFrequencyCandidateIdentifier::IdentifyCandidates(const KTCluster1DData::SetOfClusters& clusters, const KTFrequencySpectrumPolar* freqSpec)
158  {
159  KTFrequencyCandidateData::Candidates candidates(clusters.size());
160 
161  KTFrequencyCandidate newCandidate;
162  UInt_t iCandidate = 0;
163  for (KTCluster1DData::SetOfClusters::const_iterator it=clusters.begin(); it != clusters.end(); it++)
164  {
165  UInt_t firstBin = it->first;
166  UInt_t lastBin = it->second;
167 
168  if (firstBin >= freqSpec->size() || lastBin >= freqSpec->size())
169  {
170  KTWARN(fcilog, "First bin and/or last bin it outside the range of the given frequency spectrum:\n" <<
171  "\tFirst bin: " << firstBin << '\n' <<
172  "\tLast bin: " << lastBin << '\n' <<
173  "\tFS size: " << freqSpec->size());
174  continue;
175  }
176 
177  newCandidate.SetFirstBin(firstBin);
178  newCandidate.SetLastBin(lastBin);
179 
180  Double_t weightedMean = 0.;
181  Double_t integral = 0.;
182  Double_t peakValue = 0.;
183  for (UInt_t iBin=firstBin; iBin <= lastBin; iBin++)
184  {
185  weightedMean += freqSpec->GetBinCenter(iBin) * (*freqSpec)(iBin).abs();
186  integral += (*freqSpec)(iBin).abs();
187  if ((*freqSpec)(iBin).abs() > peakValue) peakValue = (*freqSpec)(iBin).abs();
188  }
189  weightedMean /= integral;
190  newCandidate.SetMeanFrequency(weightedMean);
191  newCandidate.SetPeakAmplitude(peakValue);
192  newCandidate.SetAmplitudeSum(integral);
193 
194  candidates[iCandidate] = newCandidate;
195  iCandidate++;
196  }
197 
198  return candidates;
199  }

207  UInt_t iCandidate = 0;
208  for (KTCluster1DData::SetOfClusters::const_iterator it=clusters.begin(); it != clusters.end(); it++)
209  {
210  UInt_t firstBin = it->first;
211  UInt_t lastBin = it->second;
212 
213  if (firstBin >= freqSpec->size() || lastBin >= freqSpec->size())
214  {
215  KTWARN(fcilog, "First bin and/or last bin it outside the range of the given frequency spectrum:\n" <<
216  "\tFirst bin: " << firstBin << '\n' <<
217  "\tLast bin: " << lastBin << '\n' <<
218  "\tFS size: " << freqSpec->size());
219  continue;
220  }
221 
222  newCandidate.SetFirstBin(firstBin);
223  newCandidate.SetLastBin(lastBin);
224 
225  Double_t weightedMean = 0.;
226  Double_t integral = 0.;
227  Double_t value = 0.;
228  Double_t peakValue = 0.;
229  for (UInt_t iBin=firstBin; iBin <= lastBin; iBin++)
230  {
231  value = sqrt((*freqSpec)(iBin)[0] * (*freqSpec)(iBin)[0] + (*freqSpec)(iBin)[1] * (*freqSpec)(iBin)[1]);
232  weightedMean += freqSpec->GetBinCenter(iBin) * value;
233  integral += value;
234  if (value > peakValue) peakValue = value;
235  }
236  weightedMean /= integral;
237  newCandidate.SetMeanFrequency(weightedMean);
238  newCandidate.SetPeakAmplitude(peakValue);
239  newCandidate.SetAmplitudeSum(integral);
240 
241  candidates[iCandidate] = newCandidate;
242  iCandidate++;
243  }

275  if (corrData != NULL)

278  bundle->AddData(newData);
