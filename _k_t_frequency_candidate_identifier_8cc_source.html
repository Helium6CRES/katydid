82  KTFrequencyCandidateData::Candidates KTFrequencyCandidateIdentifier::IdentifyCandidates(const KTCluster1DData::SetOfClusters& clusters, const KTFrequencySpectrumPolar* freqSpec)
83  {
84  KTFrequencyCandidateData::Candidates candidates(clusters.size());
85 
86  KTFrequencyCandidate newCandidate;
87  UInt_t iCandidate = 0;
88  for (KTCluster1DData::SetOfClusters::const_iterator it=clusters.begin(); it != clusters.end(); it++)
89  {
90  UInt_t firstBin = it->first;
91  UInt_t lastBin = it->second;
92 
93  if (firstBin >= freqSpec->size() || lastBin >= freqSpec->size())
94  {
95  KTWARN(fcilog, "First bin and/or last bin it outside the range of the given frequency spectrum:\n" <<
96  "\tFirst bin: " << firstBin << '\n' <<
97  "\tLast bin: " << lastBin << '\n' <<
98  "\tFS size: " << freqSpec->size());
99  continue;
100  }
101 
102  newCandidate.SetFirstBin(firstBin);
103  newCandidate.SetLastBin(lastBin);
104 
105  Double_t weightedMean = 0.;
106  Double_t integral = 0.;
107  Double_t peakValue = 0.;
108  for (UInt_t iBin=firstBin; iBin <= lastBin; iBin++)
109  {
110  weightedMean += freqSpec->GetBinCenter(iBin) * (*freqSpec)(iBin).abs();
111  integral += (*freqSpec)(iBin).abs();
112  if ((*freqSpec)(iBin).abs() > peakValue) peakValue = (*freqSpec)(iBin).abs();
113  }
114  weightedMean /= integral;
115  newCandidate.SetMeanFrequency(weightedMean);
116  newCandidate.SetPeakAmplitude(peakValue);
117  newCandidate.SetAmplitudeSum(integral);
118 
119  candidates[iCandidate] = newCandidate;
120  iCandidate++;
121  }
122 
123  return candidates;
124  }

140  KTWARN(fcilog, "First bin and/or last bin it outside the range of the given frequency spectrum:\n" <<

144  continue;

147  newCandidate.SetFirstBin(firstBin);

149 
150  Double_t weightedMean = 0.;

153  Double_t peakValue = 0.;
154  for (UInt_t iBin=firstBin; iBin <= lastBin; iBin++)
155  {

157  weightedMean += freqSpec->GetBinCenter(iBin) * value;
158  integral += value;
159  if (value > peakValue) peakValue = value;
160  }
161  weightedMean /= integral;
162  newCandidate.SetMeanFrequency(weightedMean);
163  newCandidate.SetPeakAmplitude(peakValue);
164  newCandidate.SetAmplitudeSum(integral);
