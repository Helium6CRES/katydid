134  KTFrequencySpectrumData* KTSimpleFFT::TransformData(const KTTimeSeriesData* tsData)
135  {
136  if (tsData->GetNTimeSeries() < 1)
137  {
138  KTWARN(fftlog_simp, "Data has no channels!");
139  return NULL;
140  }
141  if (tsData->GetTimeSeries(0)->GetNTimeBins() != GetTimeSize())
142  {
143  SetTimeSize(tsData->GetTimeSeries(0)->GetNTimeBins());
144  InitializeFFT();
145  }
146 
147  if (! fIsInitialized)
148  {
149  KTERROR(fftlog_simp, "FFT must be initialized before the transform is performed\n"
150  << " Please first call InitializeFFT(), then use a TakeData method to set the data, and then finally perform the transform.");
151  return NULL;
152  }
153 
154  KTFrequencySpectrumData* newData = new KTFrequencySpectrumData(tsData->GetNTimeSeries());
155 
156  for (UInt_t iChannel = 0; iChannel < tsData->GetNTimeSeries(); iChannel++)
157  {
158  const KTTimeSeriesReal* nextInput = dynamic_cast< const KTTimeSeriesReal* >(tsData->GetTimeSeries(iChannel));
159  if (nextInput == NULL)
160  {
161  KTERROR(fftlog_simp, "Incorrect time series type: time series did not cast to KTTimeSeriesReal.");
162  delete newData;
163  return NULL;
164  }
165  KTFrequencySpectrumPolar* nextResult = Transform(nextInput);
166  if (nextResult == NULL)
167  {
168  KTERROR(fftlog_simp, "One of the channels did not transform correctly.");
169  delete newData;
170  return NULL;
171  }
172  newData->SetSpectrum(nextResult, iChannel);
173  }
174 
175  newData->SetTimeInRun(tsData->GetTimeInRun());
176  newData->SetSliceNumber(tsData->GetSliceNumber());
177 
178  KTDEBUG(fftlog_simp, "FFT complete; " << newData->GetNChannels() << " channel(s) transformed");
179 
180  newData->SetName(fOutputDataName);
181 
182  return newData;
183  }

185  KTFrequencySpectrumPolar* KTSimpleFFT::Transform(const KTTimeSeriesReal* data) const
186  {
187  UInt_t nTimeBins = (UInt_t)data->size();
188  if (nTimeBins != fTimeSize)
189  {
190  KTWARN(fftlog_simp, "Number of bins in the data provided does not match the number of bins set for this transform\n"
191  << " Bin expected: " << fTimeSize << "; Bins in data: " << nTimeBins);
192  return NULL;
193  }
194 
195  Double_t timeBinWidth = data->GetTimeBinWidth();
196 
197  copy(data->begin(), data->end(), fInputArray);
198 
199  fftw_execute(fFTPlan);
200 
201  return ExtractTransformResult(GetMinFrequency(timeBinWidth), GetMaxFrequency(timeBinWidth));
202  }

204  KTFrequencySpectrumPolar* KTSimpleFFT::ExtractTransformResult(Double_t freqMin, Double_t freqMax) const
205  {
206  UInt_t freqSize = GetFrequencySize();
207  Double_t normalization = sqrt(2. / (Double_t)GetTimeSize());
208 
209  Double_t tempReal, tempImag;
210  KTFrequencySpectrumPolar* newSpect = new KTFrequencySpectrumPolar(freqSize, freqMin, freqMax);
211  for (Int_t iPoint = 0; iPoint<freqSize; iPoint++)
212  {
213  (*newSpect)(iPoint).set_rect(fOutputArray[iPoint][0], fOutputArray[iPoint][1]);
214  (*newSpect)(iPoint) *= normalization;
215  }
216 
217  return newSpect;
218  }
