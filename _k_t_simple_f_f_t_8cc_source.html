292  KTTimeSeriesReal* KTSimpleFFT::Transform(const KTFrequencySpectrumPolar* data) const
293  {
294  unsigned nBins = (unsigned)data->size();
295  unsigned freqSize = GetFrequencySize();
296  unsigned timeSize = GetTimeSize();
297  if (nBins != freqSize)
298  {
299  KTWARN(fftlog_simp, "Number of bins in the data provided does not match the number of bins set for this transform\n"
300  << " Bin expected: " << freqSize << "; Bins in data: " << nBins);
301  return NULL;
302  }
303 
304  for (unsigned iPoint = 0; iPoint < freqSize; iPoint++)
305  {
306  fFSArray[iPoint][0] = real((*data)(iPoint));
307  fFSArray[iPoint][1] = imag((*data)(iPoint));
308  }
309 
310  fftw_execute(fReversePlan);
311 
312  KTTimeSeriesReal* newTS = new KTTimeSeriesReal(timeSize, GetMinTime(), GetMaxTime(data->GetBinWidth()));
313  copy(fTSArray, fTSArray + timeSize, newTS->begin());
314 
315  return newTS;
316  }
