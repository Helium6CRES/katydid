223  inline double KTForwardFFTW::GetMinFrequency(double timeBinWidth) const
224  {
225  if (fState == kR2C || (fState == kC2C && fComplexAsIQ))
226  {
227  // DC bin is centered at 0, with half a bin width on either side
228  return -0.5 * GetFrequencyBinWidth(timeBinWidth);
229  }
230  else // frequencies symmetric about DC
231  {
232  // There's one bin at the center, always: the DC bin.
233  // # of bins on the negative side is nFreqBins/2 (rounded down because of integer division).
234  // 0.5 is added to the # of bins because of the half of the DC bin on the negative frequency side.
235  return -GetFrequencyBinWidth(timeBinWidth) * (double(fFrequencySize/2) + 0.5);
236  }
237  }

239  inline double KTForwardFFTW::GetMaxFrequency(double timeBinWidth) const
240  {
241  if (fState == kR2C || (fState == kC2C && fComplexAsIQ))
242  {
243  return GetFrequencyBinWidth(timeBinWidth) * ((double)fFrequencySize - 0.5);
244  }
245  else // frequencies symmetric about DC
246  {
247  // There's one bin at the center, always: the DC bin.
248  // # of bins on the positive side is nFreqBins/2 if the number of bins is odd, and nFreqBins/2-1 if the number of bins is even (division rounded down because of integer division).
249  // 0.5 is added to the # of bins because of the half of the DC bin on the positive frequency side.
250  unsigned nBinsToSide = fFrequencySize / 2;
251  return GetFrequencyBinWidth(timeBinWidth) * (double(nBinsToSide*2 == fFrequencySize ? nBinsToSide - 1 : nBinsToSide) + 0.5);
252  }
253  }
