122  bool KTDataAccumulator::CoreAddTSDataReal(KTTimeSeriesData& data, Accumulator& accDataStruct, KTTimeSeriesData& accData)
123  {
124  double remainingFrac = 1.;
125  if (fAccumulatorSize != 0 && accDataStruct.GetSliceNumber() >= fAccumulatorSize)
126  remainingFrac -= fAveragingFrac;
127  //KTDEBUG(avlog, "averaging frac: " << fAveragingFrac << " remaining frac: " << remainingFrac);
128 
129  unsigned nComponents = data.GetNComponents();
130 
131  if (accDataStruct.GetSliceNumber() == 0)
132  {
133  accData.SetNComponents(nComponents);
134  for (unsigned iComponent = 0; iComponent < nComponents; ++iComponent)
135  {
136  KTTimeSeriesReal* dataTS = static_cast< KTTimeSeriesReal* >(data.GetTimeSeries(iComponent));
137  KTTimeSeriesReal* newTS = new KTTimeSeriesReal(dataTS->GetNTimeBins(), dataTS->GetRangeMin(), dataTS->GetRangeMax());
138  (*newTS) *= 0.;
139  accData.SetTimeSeries(newTS, iComponent);
140  }
141  }
142 
143  accDataStruct.BumpSliceNumber();
144 
145  if (nComponents != accData.GetNComponents())
146  {
147  KTERROR(avlog, "Numbers of components in the average and in the new data do not match");
148  return false;
149  }
150 
151  unsigned arraySize = data.GetTimeSeries(0)->GetNTimeBins();
152  if (arraySize != accData.GetTimeSeries(0)->GetNTimeBins())
153  {
154  KTERROR(avlog, "Sizes of arrays in the average and in the new data do not match");
155  return false;
156  }
157 
158  for (unsigned iComponent = 0; iComponent < nComponents; ++iComponent)
159  {
160  KTTimeSeriesReal* newTS = static_cast< KTTimeSeriesReal* >(data.GetTimeSeries(iComponent));
161  KTTimeSeriesReal* avTS = static_cast< KTTimeSeriesReal* >(accData.GetTimeSeries(iComponent));
162  for (unsigned iBin = 0; iBin < arraySize; ++iBin)
163  {
164  //KTDEBUG(avlog, (*avTS)(iBin) << " " << (*newTS)(iBin) << " " << remainingFrac << " " << fAveragingFrac);
165  (*avTS)(iBin) = (*avTS)(iBin) * remainingFrac + (*newTS)(iBin) * fAveragingFrac;
166  //KTDEBUG(avlog, (*avTS)(iBin));
167  }
168  }
169 
170  return true;
171  }
