109  KTDiscriminatedPoints1DData* KTSpectrumDiscriminator::Discriminate(const KTFrequencySpectrumData* data)
110  {
111  if (fCalculateMinBin)
112  {
113  SetMinBin(data->GetSpectrum(0)->FindBin(fMinFrequency));
114  KTDEBUG(sdlog, "Minimum bin set to " << fMinBin);
115  }
116  if (fCalculateMaxBin)
117  {
118  SetMaxBin(data->GetSpectrum(0)->FindBin(fMaxFrequency));
119  KTDEBUG(sdlog, "Maximum bin set to " << fMaxBin);
120  }
121 
122  UInt_t nChannels = data->GetNChannels();
123 
124  KTDiscriminatedPoints1DData* newData = new KTDiscriminatedPoints1DData(nChannels);
125 
126  newData->SetNBins(data->GetSpectrum(0)->size());
127  newData->SetBinWidth(data->GetSpectrum(0)->GetBinWidth());
128 
129  // Interval: [fMinBin, fMaxBin)
130  UInt_t nBins = fMaxBin - fMinBin + 1;
131  Double_t sigmaNorm = 1. / Double_t(nBins - 1);
132 
133  for (UInt_t iChannel=0; iChannel<nChannels; iChannel++)
134  {
135 
136  const KTFrequencySpectrumPolar* spectrum = data->GetSpectrum(iChannel);
137 
138  Double_t mean = 0.;
139 #pragma omp parallel for reduction(+:mean)
140  for (UInt_t iBin=fMinBin; iBin<fMaxBin; iBin++)
141  {
142  mean += (*spectrum)(iBin).abs();
143  }
144  mean /= (Double_t)nBins;
145 
146  Double_t threshold = 0.;
147  if (fThresholdMode == eSNR)
148  {
149  // SNR = P_signal / P_noise = (A_signal / A_noise)^2
150  // In this case (i.e. KTFrequencySpectrumPolar), A_noise = mean
151  threshold = sqrt(fSNRThreshold) * mean;
152  KTDEBUG(sdlog, "Discriminator threshold for channel " << iChannel << " set at <" << threshold << "> (SNR mode)");
153  }
154  else if (fThresholdMode == eSigma)
155  {
156  Double_t sigma = 0., diff;
157 #pragma omp parallel for private(diff) reduction(+:sigma)
158  for (UInt_t iBin=fMinBin; iBin<fMaxBin; iBin++)
159  {
160  diff = (*spectrum)(iBin).abs() - mean;
161  sigma += diff * diff;
162  }
163  sigma = sqrt(sigma * sigmaNorm);
164 
165  threshold = mean + fSigmaThreshold * sigma;
166  KTDEBUG(sdlog, "Discriminator threshold for channel " << iChannel << " set at <" << threshold << "> (Sigma mode)");
167  }
168 
169  newData->SetThreshold(threshold, iChannel);
170 
171  // loop over bins, checking against the threshold
172  Double_t value;
173 #pragma omp parallel for private(value)
174  for (UInt_t iBin=fMinBin; iBin<fMaxBin; iBin++)
175  {
176  value = (*spectrum)(iBin).abs();
177  if (value >= threshold) newData->AddPoint(iBin, value, iChannel);
178  }
179 
180  }
181 
182  newData->SetTimeInRun(data->GetTimeInRun());
183  newData->SetSliceNumber(data->GetSliceNumber());
184 
185  newData->SetName(fOutputDataName);
186 
187  return newData;
188  }
