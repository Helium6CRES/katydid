135  bool KTSpectrumDiscriminator::CoreDiscriminate(KTFrequencySpectrumDataFFTWCore& data, KTDiscriminatedPoints1DData& newData)
136  {
137  if (fCalculateMinBin)
138  {
139  SetMinBin(data.GetSpectrumFFTW(0)->FindBin(fMinFrequency));
140  KTDEBUG(sdlog, "Minimum bin set to " << fMinBin);
141  }
142  if (fCalculateMaxBin)
143  {
144  SetMaxBin(data.GetSpectrumFFTW(0)->FindBin(fMaxFrequency));
145  KTDEBUG(sdlog, "Maximum bin set to " << fMaxBin);
146  }
147 
148  unsigned nComponents = data.GetNComponents();
149 
150  newData.SetNBins(data.GetSpectrumFFTW(0)->size());
151  double binWidth = data.GetSpectrumFFTW(0)->GetBinWidth();
152  newData.SetBinWidth(binWidth);
153 
154  // Interval: [fMinBin, fMaxBin)
155  unsigned nBins = fMaxBin - fMinBin + 1;
156  double sigmaNorm = 1. / double(nBins - 1);
157 
158  // Temporary storage for magnitude values
159  vector< double > magnitude(data.GetSpectrumFFTW(0)->size());
160 
161  for (unsigned iComponent=0; iComponent<nComponents; ++iComponent)
162  {
163  const KTFrequencySpectrumFFTW* spectrum = data.GetSpectrumFFTW(iComponent);
164  if (spectrum == NULL)
165  {
166  KTERROR(sdlog, "Frequency spectrum pointer (component " << iComponent << ") is NULL!");
167  return false;
168  }
169  if (spectrum->size() != magnitude.size())
170  {
171  magnitude.resize(spectrum->size());
172  }
173 
174  double mean = 0.;
175 #pragma omp parallel for reduction(+:mean)
176  for (unsigned iBin=fMinBin; iBin<=fMaxBin; iBin++)
177  {
178  magnitude[iBin] = sqrt((*spectrum)(iBin)[0] * (*spectrum)(iBin)[0] + (*spectrum)(iBin)[1] * (*spectrum)(iBin)[1]);
179  mean += magnitude[iBin];
180  }
181  mean /= (double)nBins;
182 
183  double threshold = 0.;
184  if (fThresholdMode == eSNR_Amplitude)
185  {
186  // SNR = P_signal / P_noise = (A_signal / A_noise)^2, A_noise = mean
187  threshold = sqrt(fSNRThreshold) * mean;
188  KTDEBUG(sdlog, "Discriminator threshold for channel " << iComponent << " set at <" << threshold << "> (SNR mode)");
189  }
190  else if (fThresholdMode == eSNR_Power)
191  {
192  // SNR = P_signal / P_noise, P_noise = mean
193  threshold = fSNRThreshold * mean;
194  KTDEBUG(sdlog, "Discriminator threshold for channel " << iComponent << " set at <" << threshold << "> (SNR mode)");
195  }
196  else if (fThresholdMode == eSigma)
197  {
198  double sigma = 0., diff;
199 #pragma omp parallel for private(diff) reduction(+:sigma)
200  for (unsigned iBin=fMinBin; iBin<=fMaxBin; iBin++)
201  {
202  diff = magnitude[iBin] - mean;
203  sigma += diff * diff;
204  }
205  sigma = sqrt(sigma * sigmaNorm);
206 
207  threshold = mean + fSigmaThreshold * sigma;
208  KTDEBUG(sdlog, "Discriminator threshold for channel " << iComponent << " set at <" << threshold << "> (Sigma mode)");
209  }
210 
211  // loop over bins, checking against the threshold
212  double value;
213 #pragma omp parallel for private(value)
214  for (unsigned iBin=fMinBin; iBin<=fMaxBin; iBin++)
215  {
216  value = magnitude[iBin];
217  if (value >= threshold) newData.AddPoint(iBin, KTDiscriminatedPoints1DData::Point(binWidth * ((double)iBin + 0.5), value, threshold), iComponent);
218  }
219  KTDEBUG(sdlog, "Component " << iComponent << " has " << newData.GetSetOfPoints(iComponent).size() << " points above threshold");
220 
221  }
222  KTINFO(sdlog, "Completed discrimination on " << nComponents << " components");
223 
224  return true;
225  }
