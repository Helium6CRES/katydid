154  void KTMaskedArray< XArrayType, XArrayObjectType >::Cut(unsigned int toBeginCut, unsigned int nToCut)
155  {
156  // cache the start position for the search to save time
157  PositionStoreIt startSearch = fPositions.begin();
158  // the condition for staying in the loop is that we haven't yet reached nToCut AND that the requested toCut is less than the array size
159  for (unsigned int toCut=toBeginCut; toCut<fArraySize && toCut-toBeginCut<nToCut; toCut++)
160  {
161  // look for the first position in fPositions that is >= toCut
162  PositionStoreIt posInPositions = std::lower_bound(startSearch, fPositions.end(), toCut);
163  // if we've reached the end of fPositions, then we're done
164  if (posInPositions == fPositions.end()) break;
165  // if the position matches the requested cut (this will be false if the requested cut has already been cut), perform the cut
166  // the return value, which is an iterator to the new location of the element that followed the erased element, is set as the new startSearch
167  if (*posInPositions == toCut) startSearch = fPositions.erase(posInPositions);
168  }
169  return;
170  }

172  void KTMaskedArray< XArrayType, XArrayObjectType >::UnCut(unsigned int toBeginUnCut, unsigned int nToUnCut)
173  {
174  // if we might need to reallocate the vector because the number of requested uncut positions would result in
175  // fPositions exceeding its capacity, do the reallocation first so we don't invalidate iterators
176  if (fPositions.size() + nToUnCut > fPositions.capacity()) fPositions.reserve(fPositions.size() + nToUnCut);
177 
178  // cache the start position for the search to save time
179  PositionStoreIt startSearch = fPositions.begin();
180  // the condition for staying in the loop is that we haven't yet reached nToUnCut AND that the requested toUnCut is less than the array size
181  for (unsigned int toUnCut=toBeginUnCut; toUnCut<fArraySize && toUnCut-toBeginUnCut<nToUnCut; toUnCut++)
182  {
183  // look for the first position in fPositions that is >= toUnCut
184  PositionStoreIt posInPositions = std::lower_bound(startSearch, fPositions.end(), toUnCut);
185  // if we've reached the end of fPositions, we can add toUnCut;
186  // startSearch is set to the end of fPositions because this is guaranteed since fPositions is ordered
187  if (posInPositions == fPositions.end())
188  {
189  fPositions.insert(posInPositions, toUnCut);
190  startSearch = fPositions.end();
191  continue;
192  }
193  // if the requested toUnCut is not in fPositions already, add it
194  // if the insertion is made, the position of the inserted element is the new startSearch
195  // otherwise posInPositions is the new startSearch
196  if (*posInPositions != toUnCut) startSearch = fPositions.insert(posInPositions, toUnCut);
197  else startSearch = posInPositions;
198  }
199  return;
200  }
