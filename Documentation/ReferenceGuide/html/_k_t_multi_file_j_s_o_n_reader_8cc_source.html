132  bool KTMultiFileJSONReader::Run()
133  {
134  for (fFileIter = fFilenames.begin(); fFileIter != fFilenames.end(); fFileIter++)
135  {
136  rapidjson::Document document;
137  if (! OpenAndParseFile(*fFileIter, document))
138  {
139  KTERROR(inlog, "A problem occurred while parsing file <" << *fFileIter << ">");
140  return false;
141  }
142 
143  Nymph::KTDataPtr newData(new Nymph::KTData());
144  for (deque< DataType >::const_iterator dtIt = fDataTypes.begin(); dtIt != fDataTypes.end(); dtIt++)
145  {
146  KTDEBUG(inlog, "Appending data of type " << dtIt->fName);
147  if (! (this->*(dtIt->fAppendFcn))(document, *(newData.get())))
148  {
149  KTERROR(inlog, "Something went wrong while appending data of type <" << dtIt->fName << "> from <" << *fFileIter << ">");
150  }
151  (*(dtIt->fSignal))(newData);
152  }
153  }
154 
155  fDoneSignal();
156 
157  return true;
158  }

228  {
229  const rapidjson::Value& support = (*evIt)["support"];

233  unsigned startSample = support[rapidjson::SizeType(1)].GetUint(); // explicit cast of array index to SizeType used because of abiguous overload
234  unsigned endRec = support[rapidjson::SizeType(2)].GetUint(); // explicit cast of array index to SizeType used because of abiguous overload
235  unsigned endSample = support[rapidjson::SizeType(3)].GetUint(); // explicit cast of array index to SizeType used because of abiguous overload
236  KTDEBUG(inlog, "extracted (" << startRec << ", " << startSample << ", " << endRec << ", " << endSample << ")");
237  if (endRec < startRec || (endRec == startRec && endSample < startSample))
238  {
239  KTWARN(inlog, "Invalid event: (" << startRec << ", " << startSample << " --> " << endRec << ", " << endSample << ")");
240  }
241  else
242  {
243  mcTruth.AddEvent(KTMCTruthEvents::Event(startRec, startSample, endRec, endSample));
244  }
245  }
246  else
247  {
248  KTWARN(inlog, "\"support\" value is either missing or not an array");
249  }
250  }
251 
252  KTDEBUG(inlog, "new data object has " << mcTruth.GetEvents().size() << " events");
253 
254  return true;
