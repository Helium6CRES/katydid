91  // JSON file configuration

154  void KTApplication::AddConfigOptionsToCLHandler(const scarab::param* param, const std::string& rootName)
155  {
156  if (param->is_null())
157  {
158  fCLHandler->AddOption("Config File Options", "Configuration flag: " + rootName, rootName, false);
159  }
160  else if (param->is_value())
161  {
162  fCLHandler->AddOption< string >("Config File Options", "Configuration value: " + rootName, rootName, false);
163  }
164  else if (param->is_array())
165  {
166  string nextRootName(rootName);
167  if (! rootName.empty() && rootName.back() != '.') nextRootName += ".";
168 
169  const scarab::param_array* paramArray = &param->as_array();
170  unsigned arraySize = paramArray->size();
171  for (unsigned iParam = 0; iParam < arraySize; ++iParam)
172  {
173  AddConfigOptionsToCLHandler(paramArray->at(iParam), nextRootName + std::to_string(iParam));
174  }
175  }
176  else if (param->is_node())
177  {
178  string nextRootName(rootName);
179  if (! rootName.empty()) nextRootName += ".";
180 
181  const scarab::param_node* paramNode = &param->as_node();
182  for (scarab::param_node::const_iterator nodeIt = paramNode->begin(); nodeIt != paramNode->end(); ++nodeIt)
183  {
184  AddConfigOptionsToCLHandler(nodeIt->second, nextRootName + nodeIt->first);
185  }
186  }
187 
188  return;
