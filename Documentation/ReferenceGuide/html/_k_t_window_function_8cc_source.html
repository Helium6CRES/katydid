92  double KTWindowFunction::SetLength(double length)
93  {
94  fLength = fabs(length);
95  if (fLastSetParameter == kBinWidth)
96  {
97  // Priority is to preserve the bin width, but it might not be an even divisor of the new length
98  double prelimSize = fLength / fBinWidth;
99  fSize = (unsigned)KTMath::Nint(prelimSize);
100  fBinWidth = fLength / (double)fSize;
101  }
102  else if (fLastSetParameter == kSize || fLastSetParameter == kLength)
103  {
104  fBinWidth = fLength / fSize;
105  }
106  RebuildWindowFunction();
107  fLastSetParameter = kLength;
108  return fBinWidth;
109  }
