428  template< class XDataType >
429  void KTMultiSliceClustering::ProcessOneSliceData(Nymph::KTDataPtr data)
430  {
431  if (! data->Has< KTDiscriminatedPoints1DData >())
432  {
433  KTWARN(sclog, "No discriminated-points data was present");
434  return;
435  }
436  if (! data->Has< XDataType >())
437  {
438  KTWARN(sclog, "No frequency spectrum (polar) data was present");
439  return;
440  }
441  // signal for any continued use of the input data
442  fOneSliceDataSignal(data);
443  DataList* clusteredData = FindClusters(data->Of< KTDiscriminatedPoints1DData >(), data->Of< XDataType >(), data->Of< KTSliceHeader >());
444 
445  if (data->GetLastData())
446  {
447  KTINFO(sclog, "Last input data processed; Cleaning up remaining active clusters");
448  KTMultiSliceClustering::DataList* lastData = CompleteAllClusters();
449  clusteredData->splice(clusteredData->end(), *lastData);
450  delete lastData;
451  }
452 
453  if (clusteredData != NULL)
454  {
455  RunDataLoop(clusteredData);
456  }
457  return;
458  }
