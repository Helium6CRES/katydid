185  void KTHDF5TypeWriterEventAnalysis::WriteMTEBuffer() {
186  KTDEBUG("writing MTE buffer.");
187  // Now create the dataspace we need
188  hsize_t* dims_cands = new hsize_t(this->fMTEDataBuffer.size());
189  hsize_t* dims_tracks = new hsize_t(this->fMTETracksDataBuffer.size());
190  H5::DataSpace dspace_cands(1, dims_cands);
191  H5::DataSpace dspace_tracks(1, dims_tracks);
192 
193  if( !fWriter->OpenAndVerifyFile() ) return;
194  // Make a group for the events, and a separate group for the tracks belonging to the event
195  H5::Group* candidatesGroup = fWriter->AddGroup("candidates");
196  H5::Group* candidateTracksGroup = fWriter->AddGroup("candidate_tracks");
197 
198  // OK, create the dataset and write it down.
199  // Write the event information
200  std::stringstream namestream;
201  std::string dsetname;
202  namestream << "candidates_" << this->fFlushMTEIdx;
203  namestream >> dsetname;
204  H5::DataSet* dset_cands = new H5::DataSet(candidatesGroup->createDataSet(dsetname.c_str(),
205  *(this->fMTEType),
206  dspace_cands));
207  dset_cands->write((this->fMTEDataBuffer).data(),*(this->fMTEType));
208 
209  // Write the tracks that belong to the current events
210  //dsetname.clear();
211  //namestream.str(std::string());
212  std::stringstream namestream2;
213  std::string dsetname2;
214  namestream2 << "candidate_tracks_" << this->fFlushMTEIdx;
215  namestream2 >> dsetname2;
216  H5::DataSet* dset_tracks = new H5::DataSet(candidateTracksGroup->createDataSet(dsetname2.c_str(),
217  *(this->fPTType),
218  dspace_tracks));
219  dset_tracks->write((this->fMTETracksDataBuffer).data(),*(this->fPTType));
220 
221  this->fMTETracksDataBuffer.clear();
222  this->fMTEDataBuffer.clear();
223  this->fFlushMTEIdx++;
224  }
