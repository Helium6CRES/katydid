94  bool KTAnalyticAssociator::CreateAssociateData(KTTimeSeriesData& tsData)
95  {
96  if (! CheckAndDoFFTInit())
97  {
98  return false;
99  }
100 
101  unsigned nComponents = tsData.GetNComponents();
102 
103  KTFrequencySpectrumDataFFTW* fsData = NULL;
104  if (fSaveFrequencySpectrum)
105  {
106  fsData = &(tsData.Of< KTFrequencySpectrumDataFFTW >().SetNComponents(nComponents));
107  }
108 
109  // New data to hold the time series of the analytic associate
110  KTAnalyticAssociateData& aaTSData = tsData.Of< KTAnalyticAssociateData >().SetNComponents(nComponents);
111 
112  // Calculate the analytic associates
113  for (unsigned iComponent = 0; iComponent < nComponents; ++iComponent)
114  {
115  const KTTimeSeriesReal* nextInput = dynamic_cast< const KTTimeSeriesReal* >(tsData.GetTimeSeries(iComponent));
116  if (nextInput == NULL)
117  {
118  KTERROR(aalog, "Incorrect time series type: time series did not cast to KTTimeSeriesReal. Other types of time series data are not yet supported.");
119  return false;
120  }
121 
122  KTFrequencySpectrumFFTW* newFS = NULL;
123  KTTimeSeriesFFTW* newTS = NULL;
124  if (fSaveFrequencySpectrum)
125  {
126  newTS = CalculateAnalyticAssociate(nextInput, &newFS);
127  fsData->SetSpectrum(newFS, iComponent);
128  }
129  else
130  {
131  newTS = CalculateAnalyticAssociate(nextInput);
132  }
133 
134  if (newTS == NULL)
135  {
136  KTERROR(aalog, "Component <" << iComponent << "> did not transform correctly.");
137  return false;
138  }
139 
140  aaTSData.SetTimeSeries(newTS, iComponent);
141  }
142  KTINFO(aalog, "Analytic associate calculation (from TS) complete for " << nComponents << " components");
143 
144  return true;
145  }

212  KTTimeSeriesFFTW* KTAnalyticAssociator::CalculateAnalyticAssociate(const KTTimeSeriesReal* inputTS, KTFrequencySpectrumFFTW** outputFS)
213  {
214  // Forward FFT
215  KTFrequencySpectrumFFTW* freqSpec = fForwardFFT.TransformAsComplex(inputTS);
216  if (freqSpec == NULL)
217  {
218  KTERROR(aalog, "Something went wrong with the forward FFT on the time series.");
219  return NULL;
220  }
221 
222  // Calculate the analytic associate in frequency space
223  freqSpec->AnalyticAssociate();
224 
225  // reverse FFT
226  KTTimeSeriesFFTW* outputTS = fReverseFFT.TransformToComplex(freqSpec);
227 
228  // copy the address of the frequency spectrum to outputFS if it's being kept; otherwise delete
229  if (fSaveFrequencySpectrum) outputFS = &freqSpec;
230  else delete freqSpec;
231 
232  if (outputTS == NULL)
233  {
234  KTERROR(aalog, "Something went wrong with the reverse FFT on the frequency spectrum.");
235  }
236 
237  return outputTS;
238  }
