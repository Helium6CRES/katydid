87  KTPowerSpectrum* KTFrequencySpectrumPolar::CreatePowerSpectrum() const
88  {
89  // This function creates a power spectrum that runs from the smallest to the largest absolute frequency.
90  // It can handle frequency ranges that do or don't cross DC, and that are symmetric or asymmetric.
91  // If the frequency range does not cross DC, the power spectrum range will not either.
92  // If the frequency range crosses DC, the power spectrum range will run from DC to the maximum absolute frequency
93  // In this case negative-frequency bins are added to positive-frequency bins.
94 
95  double maxFreq = std::max(fabs(GetRangeMin()), fabs(GetRangeMax()));
96  double minFreq = -0.5 * GetBinWidth();
97  unsigned nBins = (maxFreq - minFreq) / GetBinWidth();
98  if (GetRangeMax() < 0. || GetRangeMin() > 0.)
99  {
100  minFreq = std::min(fabs(GetRangeMin()), fabs(GetRangeMax()));
101  nBins = size();
102  }
103 
104  KTPowerSpectrum* newPS = new KTPowerSpectrum(nBins, minFreq, maxFreq);
105  for (unsigned iBin = 0; iBin < nBins; ++iBin) (*newPS)(iBin) = 0.;
106 
107  int dcBin = FindBin(0.);
108  // default case: dcBin >= 0 && dcBin < size()
109  int firstPosFreqBin = dcBin;
110  int lastPosFreqBin = size();
111  int firstNegFreqBin = 0;
112  int lastNegFreqBin = dcBin;
113  if (dcBin >= (int)size())
114  {
115  firstPosFreqBin = size(); // lastPosFreqBin = size();
116  lastNegFreqBin = size(); // firstNegFreqBin = 0
117  }
118  else if (dcBin < 0)
119  {
120  firstPosFreqBin = 0; // lastPosFreqBin = size();
121  firstNegFreqBin = dcBin; // lastNegFreqBin = dcBin;
122  }
123 
124  double scaling = 1. / KTPowerSpectrum::GetResistance() / (double)GetNTimeBins();
125 
126  double value;
127 #pragma omp parallel for private(value)
128  for (unsigned iBin = firstPosFreqBin; iBin < lastPosFreqBin; ++iBin)
129  {
130  value = (*this)(iBin).abs();
131  (*newPS)(iBin) = value * value * scaling;
132  }
133 #pragma omp parallel for private(value)
134  for (unsigned iBin = firstNegFreqBin; iBin < lastNegFreqBin; ++iBin)
135  {
136  value = (*this)(iBin).abs();
137  (*newPS)(iBin) = value * value * scaling;
138  }
139 
140  return newPS;
141  }
