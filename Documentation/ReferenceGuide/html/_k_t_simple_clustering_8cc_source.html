443  void KTSimpleClustering::ProcessOneSliceBundle(boost::shared_ptr<KTBundle> bundle)
444  {
445  // This function removes the frequency spectrum data from the bundle.
446  // Therefore it should be the last thing done to the one-slice bundle.
447  // The one-slice-bundle signal is called before anything else does, to give other processors an opportunity to use the bundle.
448  // The existence of the extracted data is maintained by smart pointers from each cluster.
449 
450  // signal for any continued use of this bundle
451  fOneSliceBundleSignal(bundle);
452 
453  KTDiscriminatedPoints1DData* discData = bundle->GetData< KTDiscriminatedPoints1DData >(fDPInputDataName);
454  if (discData == NULL)
455  {
456  KTWARN(sclog, "No discriminated-points data was present in the bundle");
457  return;
458  }
459 
460  BundleList* clusteredBundles = NULL;
461 
462  // ExtractData is used instead of the standard GetData so that the data object is removed from the bundle.
463  KTFrequencySpectrumData* fsData = bundle->ExtractData< KTFrequencySpectrumData >(fFSInputDataName);
464  if (fsData != NULL)
465  {
466  shared_ptr<KTData> ptr(fsData);
467  clusteredBundles = FindClusters(discData, ptr);
468  }
469  else
470  {
471  KTFrequencySpectrumDataFFTW* fsDataFFTW = bundle->ExtractData< KTFrequencySpectrumDataFFTW >(fFSInputDataName);
472  if (fsDataFFTW != NULL)
473  {
474  shared_ptr<KTData> ptr(fsDataFFTW);
475  clusteredBundles = FindClusters(discData, ptr);
476  }
477  else
478  {
479  KTCorrelationData* corrData = bundle->ExtractData< KTCorrelationData >(fFSInputDataName);
480  if (fsData != NULL)
481  {
482  shared_ptr<KTData> ptr(corrData);
483  clusteredBundles = FindClusters(discData, ptr);
484  }
485  else
486  {
487  KTWARN(sclog, "No spectrum data was found with name <" << fFSInputDataName << ">");
488  return;
489  }
490  }
491  }
492 
493  if (clusteredBundles != NULL)
494  {
495  RunBundleLoop(clusteredBundles);
496  }
497  return;
498  }
