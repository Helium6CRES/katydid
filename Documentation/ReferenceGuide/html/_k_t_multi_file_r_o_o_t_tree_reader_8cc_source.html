112  bool KTMultiFileROOTTreeReader::Run()
113  {
114  for (fFileIter = fFilenames.begin(); fFileIter != fFilenames.end(); fFileIter++)
115  {
116  TFile* file = OpenFile(*fFileIter);
117  if (file == NULL)
118  {
119  KTERROR(inlog, "A problem occurred while trying to open file <" << *fFileIter << ">");
120  return false;
121  }
122 
123  Nymph::KTDataPtr newData(new Nymph::KTData());
124  for (deque< DataType >::const_iterator dtIt = fDataTypes.begin(); dtIt != fDataTypes.end(); dtIt++)
125  {
126  KTDEBUG(inlog, "Appending data of type " << dtIt->fName);
127 
128  TTree* tree = ExtractTree(file, dtIt->fTreeName);
129  if (tree == NULL)
130  {
131  KTERROR(inlog, "Tree <" << dtIt->fTreeName << "> was not extracted from file <" << *fFileIter << ">");
132  return false;
133  }
134 
135  if (! (this->*(dtIt->fAppendFcn))(tree, *(newData.get())))
136  {
137  KTERROR(inlog, "Something went wrong while appending data of type <" << dtIt->fName << "> from tree <" << dtIt->fTreeName << "> from file <" << *fFileIter << ">");
138  return false;
139  }
140  (*(dtIt->fSignal))(newData);
141  }
142  }
143 
144  fDoneSignal();
145 
146  return true;

187  bool KTMultiFileROOTTreeReader::AppendAmpDistData(TTree* tree, Nymph::KTData& appendToData)
188  {
189  // Determine the number of components and the number of frequency bins
190  unsigned nComponents = 0;
191  unsigned nFreqBins = 0;
192  while (true)
193  {
194  stringstream cut;
195  cut << "Component == " << nComponents;
196  unsigned nEntriesForComponent = (unsigned)tree->GetEntries(cut.str().c_str());
197  if (nEntriesForComponent == 0)
198  {
199  nComponents++; // advance this by 1 to be the number of components, not the last component number
200  break;
201  }
202  if (nEntriesForComponent > nFreqBins)
203  {
204  nFreqBins = nEntriesForComponent;
205  }
206  }
207 
208  // Create data structures
209  TAmplitudeDistributionData ampDistData;
210  tree->SetBranchAddress("Component", &ampDistData.fComponent);
211  tree->SetBranchAddress("FreqBin", &ampDistData.fFreqBin);
212  tree->SetBranchAddress("Distribution", &ampDistData.fDistribution);
213 
214  KTDEBUG(inlog, "Initializing new set of amplitude distributions, with " << nComponents << " components and " << nFreqBins << "frequency bins");
215  KTAmplitudeDistribution& ampDist = appendToData.Of< KTAmplitudeDistribution >();
216  ampDist.InitializeNull(nComponents, nFreqBins);
217 
218  // Read in the data
219  unsigned nEntries = (unsigned)tree->GetEntries();
220  for (unsigned iEntry=0; iEntry < nEntries; iEntry++)
221  {
222  tree->GetEntry(iEntry);
223 
224  unsigned nDistBins = (unsigned)ampDistData.fDistribution->GetNbinsX();
225  ampDist.InitializeADistribution(ampDistData.fComponent, ampDistData.fFreqBin, nDistBins, ampDistData.fDistribution->GetXaxis()->GetXmin(), ampDistData.fDistribution->GetXaxis()->GetXmax());
226  for (unsigned iDistBin = 0; iDistBin < nDistBins; iDistBin++)
227  {
228  ampDist.SetDistValue(ampDistData.fDistribution->GetBinContent(iDistBin+1), ampDistData.fFreqBin, iDistBin, ampDistData.fComponent);
229  }
230  }
231 
232 
233  KTDEBUG(inlog, "some success message!");
234 
235  return true;
236  }
