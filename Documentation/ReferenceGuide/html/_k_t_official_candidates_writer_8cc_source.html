64  bool KTOfficialCandidatesWriter::OpenFile()
65  {
66  ScopedLock lock(fMutex);
67 
68  if (fStatus != kNotOpenedYet)
69  {
70  KTERROR(publog, "Status must be <" << kNotOpenedYet << "> to open a file; current status is <" << fStatus << ">");
71  return false;
72  }
73 
74  if (fFilename == "stdout")
75  {
76  fFileStream = new rapidjson::FileStream(stdout);
77  }
78  else
79  {
80  fFile = fopen(fFilename.c_str(), fFileMode.c_str());
81  if (fFile == NULL)
82  {
83  KTERROR(publog, "File did not open\n" <<
84  "\tFilename: " << fFilename <<
85  "\tMode: " << fFileMode);
86  return false;
87  }
88  fFileStream = new rapidjson::FileStream(fFile);
89  }
90 
91  if (fPrettyJSONFlag)
92  {
93  fJSONMaker = new KTJSONMakerPretty< rapidjson::FileStream >(*fFileStream);
94  }
95  else
96  {
97  fJSONMaker = new KTJSONMakerCompact< rapidjson::FileStream >(*fFileStream);
98  }
99 
100  fJSONMaker->StartObject();
101 
102  fStatus = kPriorToCandidates;
103 
104  return true;

171  bool KTOfficialCandidatesWriter::WriteWaterfallCandidate(KTWaterfallCandidateData& wcData)
172  {
173  using rapidjson::SizeType;
174 
175  ScopedLock lock(fMutex);
176 
177  if (! OpenAndVerifyFile()) return false;
178
