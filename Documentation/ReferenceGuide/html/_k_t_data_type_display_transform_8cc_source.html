73  void KTDataTypeDisplayTransform::DrawFrequencySpectrumDataPolar(Nymph::KTDataPtr data)
74  {
75  if (! data) return;
76 
77  ULong64_t sliceNumber = data->Of<KTSliceHeader>().GetSliceNumber();
78 
79  KTFrequencySpectrumDataPolar& fsData = data->Of<KTFrequencySpectrumDataPolar>();
80  UInt_t nComponents = fsData.GetNComponents();
81 
82  if (! fWriter->OpenWindow()) return;
83 
84  for (unsigned iChannel=0; iChannel<nComponents; iChannel++)
85  {
86  const KTFrequencySpectrumPolar* spectrum = fsData.GetSpectrumPolar(iChannel);
87  if (spectrum != NULL)
88  {
89  stringstream conv;
90  conv << "histFSpolar_" << sliceNumber << "_" << iChannel;
91  string histName;
92  conv >> histName;
93  TH1D* magHist = KT2ROOT::CreateMagnitudeHistogram(spectrum, histName);
94  fWriter->Draw(magHist);
95  }
96  }
97  return;
