338  po::variables_map tGeneralOptsVarMap;
339  po::store(tParsedOpts, tGeneralOptsVarMap);
340  po::notify(tGeneralOptsVarMap);
341 
342  // Use the general options information
343  if (tGeneralOptsVarMap.count("help"))
344  {
345  PrintHelpMessageAndExit();
346  }
347  if (tGeneralOptsVarMap.count("help-config"))
348  {
