233  bool KTCommandLineHandler::DelayedCommandLineProcessing()
234  {
235  if (! fProposedGroups.empty())
236  {
237  if (! this->FinalizeNewOptionGroups())
238  {
239  KTERROR(utillog, "An error occurred while adding the proposed option groups\n" <<
240  "Command-line options were not parsed");
241  return false;
242  }
243  }
244 
245  // Parse the command line options that remain after the initial parsing
246  try
247  {
248  fParsedOptions = po::command_line_parser(fCommandLineParseLater).options(fCommandLineOptions).allow_unregistered().run();
249  }
250  catch (std::exception& e)
251  {
252  KTERROR(utillog, "An error occurred while boost was parsing the command line options:\n" << e.what());
253  return false;
254  }
255 
256  // these will be the unregistered items, which are assumed to be intended to edit the config file
257  vector< string > tRemainingToParse = po::collect_unrecognized(fParsedOptions.options, po::include_positional);
258 
259  // Create the variable map from the parse options
260  po::store(fParsedOptions, fCommandLineVarMap);
261  po::notify(fCommandLineVarMap);
262 
263  // now parse the remaining items into the config override param node
264  for (vector< string >::const_iterator tokenIt = tRemainingToParse.begin(); tokenIt != tRemainingToParse.end(); ++tokenIt)
265  {
266  string argument(*tokenIt);
267  size_t t_name_pos = argument.find_first_not_of( fDash );
268  size_t t_val_pos = argument.find_first_of( fSeparator );
269  // the name should have 2 dashes before it, and there should be a separator
270  if( t_name_pos == 2 && t_val_pos != string::npos )
271  {
272  string t_full_name(argument.substr( t_name_pos, t_val_pos-2 ));
273 
274  size_t t_node_start_pos = 0;
275  size_t t_node_sep_pos = t_full_name.find_first_of( fNodeSeparator );
276  scarab::param_node* parentNode = &fConfigOverrideValues;
277  while (t_node_sep_pos != string::npos)
278  {
279  string nodeName(t_full_name.substr(t_node_start_pos, t_node_sep_pos));
280  if (parentNode->has(nodeName))
281  {
282  parentNode = parentNode->node_at(nodeName);
283  }
284  else
285  {
286  scarab::param_node* newChildNode = new scarab::param_node();
287  parentNode->add(nodeName, newChildNode);
288  parentNode = newChildNode;
289  }
290  t_node_start_pos = t_node_sep_pos + 1;
291  t_node_sep_pos = t_full_name.find_first_of(fNodeSeparator, t_node_start_pos);
292  }
293 
294  string valueName(t_full_name.substr(t_node_start_pos, t_val_pos));
295 
296  scarab::param_value* new_value = new scarab::param_value();
297  new_value->set(argument.substr( t_val_pos + 1 ));
298 
299  //std::cout << "(parser) adding < " << t_name << "<" << t_type << "> > = <" << new_value.value() << ">" << std::endl;
300 
301  parentNode->replace( valueName, new_value );
302 
303  continue;
304  }
305 
306  KTERROR(utillog, "Argument <" << argument << "> does not match --<name>=<value> pattern");
307  return false;
308  }
309 
310  return true;

313  void KTCommandLineHandler::InitialCommandLineProcessing()
314  {
315  if (fNArgs <= 0 || fArgV == NULL)
316  {
317  fNArgs = 0;
318  fArgV = NULL;
319  return;
320  }
321 
322  // Get the executable name
323  if (fNArgs >= 1) fExecutableName = string(fArgV[0]);
324 
325  // If no arguments were given, just return now
326  if (fNArgs == 1) return;
327 
328  // Define general options, and add them to the complete option list
329  po::options_description tGeneralOpts("General options");
330  tGeneralOpts.add_options()("help,h", "Print help message")("help-config", "Print help message after reading config file")("version,v", "Print version information");
331  /* WHEN NOT USING POSITIONAL CONFIG FILE ARGUMENT */
332  tGeneralOpts.add_options()("config,c", po::value< string >(), "Configuration file");
333  tGeneralOpts.add_options()("json,j", po::value< string >(), "Command-Line JSON");
334 
335  // We want to have the general options printed if --help is used
336  fPrintHelpOptions.add(tGeneralOpts);
337 
338  // Fill in the duplication-checking sets
339  fAllGroupKeys.insert("General");
340  fAllOptionsLong.insert("help");
341  fAllOptionsShort.insert('h');
342  fAllOptionsLong.insert("help-config");
343  fAllOptionsLong.insert("version");
344  fAllOptionsShort.insert('v');
345 
346  // Define the option for the user configuration file; this does not get printed in list of options when --help is used
347  po::options_description tHiddenOpts("Hidden options");
348  /* WHEN USING POSITIONAL CONFIG FILE ARGUMENT
349  tHiddenOpts.add_options()("config-file", po::value< string >(), "Configuration file");
350  */
351  // Add together any options that will be parsed here, in the initial command-line processing
352  po::options_description tInitialOptions("Initial options");
353  tInitialOptions.add(tGeneralOpts).add(tHiddenOpts);
354 
355  // Allow the UserConfiguration file to be specified with the only positional option
356  /* WHEN USING POSITIONAL CONFIG FILE ARGUMENT
357  po::positional_options_description tPositionOpt;
358  tPositionOpt.add("config-file", 1);
359  */
360 
361  // Add contributions of other options from elsewhere in Nymph
362  FinalizeNewOptionGroups();
363 
364  // Command line style
365  po::command_line_style::style_t pstyle = po::command_line_style::unix_style;
366  /*
367  po::command_line_style::style_t pstyle = po::command_line_style::style_t(
368  po::command_line_style::allow_long |
369  po::command_line_style::allow_short |
370  po::command_line_style::allow_dash_for_short |
371  po::command_line_style::long_allow_next |
372  po::command_line_style::long_allow_adjacent |
373  po::command_line_style::short_allow_next |
374  po::command_line_style::short_allow_adjacent |
375  po::command_line_style::allow_guessing |
376  po::command_line_style::allow_sticky);
377  */
378 
379  po::parsed_options tParsedOpts(NULL);
380  // Parse the command line looking only for the general options
381  try
382  {
383  tParsedOpts = po::command_line_parser(fNArgs, fArgV).style(pstyle).options(tInitialOptions).allow_unregistered().run();
384  /* WHEN USING POSITIONAL CONFIG FILE ARGUMENT
385  tParsedOpts = po::command_line_parser(fNArgs, fArgV).style(pstyle).options(tInitialOptions).positional(tPositionOpt).allow_unregistered().run();
386  */
387  }
388  catch (std::exception& e)
389  {
390  KTERROR(utillog, "Exception caught while performing initial CL parsing:\n"
391  << '\t' << e.what());
392  throw std::logic_error(e.what());
393  }
394  // Save the remaining command-line options for later parsing (after the full option list has been populated)
395  fCommandLineParseLater = po::collect_unrecognized(tParsedOpts.options, po::include_positional);
396  /* some debugging couts
397  std::cout << "there are " << fCommandLineParseLater.size() << " tokens to parse later." << std::endl;
398  for (unsigned i = 0; i < fCommandLineParseLater.size(); i++)
399  {
400  std::cout << " " << fCommandLineParseLater[i] << std::endl;
401  }
402  */
403 
404  // Create the variable map from the general options
405  po::variables_map tGeneralOptsVarMap;
406  po::store(tParsedOpts, tGeneralOptsVarMap);
407  po::notify(tGeneralOptsVarMap);
408 
409  // Use the general options information
410  if (tGeneralOptsVarMap.count("help"))
411  {
412  fPrintHelpMessage = true;
413  }
414  if (tGeneralOptsVarMap.count("help-config"))
415  {
416  fPrintHelpMessageAfterConfig = true;
417  }
418  if (tGeneralOptsVarMap.count("version"))
419  {
420  fPrintVersionMessage = true;
421  }
422  if (tGeneralOptsVarMap.count("config"))
423  {
424  fConfigFilename = tGeneralOptsVarMap["config"].as< string >();
425  }
426  if (tGeneralOptsVarMap.count("json"))
427  {
428  fCommandLineJSON = tGeneralOptsVarMap["json"].as< string >();
429  }
430 
431  return;
432  }
