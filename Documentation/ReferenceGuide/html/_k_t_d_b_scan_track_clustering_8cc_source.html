149  bool KTDBScanTrackClustering::DoClustering(KTKDTreeData& data)
150  {
151  KTPROG(tclog, "Starting DBSCAN track clustering");
152 
153  typedef KTDBScan< KTKDTreeData::TreeIndex > DBSCAN;
154 
155  DBSCAN dbScan;
156 
157  dbScan.SetRadius(1.);
158  dbScan.SetMinPoints(fMinPoints);
159  KTINFO(tclog, "DBScan configured");
160 
161  for (unsigned iComponent = 0; iComponent < data.GetNComponents(); ++iComponent)
162  {
163  KTDEBUG(tclog, "Clustering component " << iComponent);
164 
165  // do the clustering!
166  KTINFO(tclog, "Starting DBSCAN");
167  DBSCAN::DBSResults results;
168  if (! dbScan.DoClustering(*(data.GetTreeIndex(iComponent)), results))
169  {
170  KTERROR(tclog, "An error occurred while clustering");
171  return false;
172  }
173  KTDEBUG(tclog, "DBSCAN finished");
174 
175  std::vector< KTKDTreeData::Point > points = data.GetSetOfPoints(iComponent);
176 
177  // loop over the clusters found, and create data objects for them
178  KTDEBUG(tclog, "Found " << results.fClusters.size() << " clusters; creating candidates");
179  for (vector< DBSCAN::Cluster >::const_iterator clustIt = results.fClusters.begin(); clustIt != results.fClusters.end(); ++clustIt)
180  {
181  if (clustIt->empty())
182  {
183  KTWARN(tclog, "Empty cluster");
184  continue;
185  }
186 
187  KTDEBUG(tclog, "Creating candidate " << fDataCount << "; includes " << clustIt->size() << " points");
188 
189  ++fDataCount;
190 
191  Nymph::KTDataPtr newData(new Nymph::KTData());
192  KTSparseWaterfallCandidateData& cand = newData->Of< KTSparseWaterfallCandidateData >();
193 
194  DBSCAN::Cluster::const_iterator pointIdIt = clustIt->begin();
195  double time = points[*pointIdIt].fCoords[0] * data.GetXScaling();
196  double freq = points[*pointIdIt].fCoords[1] * data.GetYScaling();
197  double timeInAcq = points[*pointIdIt].fTimeInAcq * data.GetXScaling();
198  double minFreq = freq;
199  double maxFreq = minFreq;
200  double minTime = time;
201  double minTimeInAcq = timeInAcq;
202  double maxTime = minTime;
203  cand.AddPoint(KTSparseWaterfallCandidateData::Point(time, freq, 1., timeInAcq));
204  KTDEBUG(tclog, "Added point #" << *pointIdIt << ": " << time << ", " << freq)
205 
206  for (++pointIdIt; pointIdIt != clustIt->end(); ++pointIdIt)
207  {
208  time = points[*pointIdIt].fCoords[0] * data.GetXScaling();
209  freq = points[*pointIdIt].fCoords[1] * data.GetYScaling();
210  timeInAcq = points[*pointIdIt].fTimeInAcq * data.GetXScaling();;
211  cand.AddPoint(KTSparseWaterfallCandidateData::Point(time, freq, 1., timeInAcq));
212  KTDEBUG(tclog, "Added point #" << *pointIdIt << ": " << time << ", " << freq)
213 
214  if (time > maxTime)
215  {
216  maxTime = time;
217  }
218  else if (time < minTime)
219  {
220  minTime = time;
221  minTimeInAcq = timeInAcq;
222  }
223 
224  if (freq > maxFreq)
225  {
226  maxFreq = freq;
227  }
228  else if (freq < minFreq)
229  {
230  minFreq = freq;
231  }
232  }
233 
234  cand.SetComponent(iComponent);
235  cand.SetAcquisitionID(data.GetAcquisitionID());
236  cand.SetCandidateID(fDataCount);
237 
238  //cand.SetTimeBinWidth(fTimeBinWidth);
239  //cand.SetFreqBinWidth(fFreqBinWidth);
240 
241  cand.SetTimeInRunC(minTime);
242  cand.SetTimeInAcq(minTimeInAcq);
243  cand.SetTimeLength(maxTime - minTime);
244 
245  cand.SetMinimumFrequency(minFreq);
246  cand.SetMaximumFrequency(maxFreq);
247 
248  cand.SetFrequencyWidth(maxFreq - minFreq);
249 
250  fCandidates.insert(newData);
251  fTrackSignal(newData);
252 
253  } // loop over clusters
254
