28 // Method to determine whether a line intersects a bin with finite size
29 // The line extends over a finite segment from (x1,y1) to (x2,y2)
30 // The bin has lower-left corner (xx1,yy1) and upper-right corner (xx2,yy2)
31 // It follows that the lower-right corner is (xx2,yy1) and the upper-left corner is (xx1,yy2)
32 bool lineIntersects( double x1, double y1, double x2, double y2, double xx1, double yy1, double xx2, double yy2 )
33 {
34  // If the x or y coordinates of the line are entirely outside the bin range, we can immediately return false
35  if( x2 <= xx1 || x1 >= xx2 )
36  return false;
37  if( y2 <= yy1 || y1 >= yy2 )
38  return false;
39 
40  double m = (y2 - y1) / (x2 - x1); // slope
41  double yLeft = y1 + m * (xx1 - x1); // y-position of the line at xx1
42  double yRight = y1 + m * (xx2 - x1); // y-position of the line at xx2
43 
44  if( m > 0 )
45  {
46  // Positive slope
47  // We check the upper-left corner and the lower-right corner
48  if( yLeft >= yy2 || yRight <= yy1 )
49  return false;
50  else
51  return true;
52  }
53  else
54  {
55  // Negative slope
56  // We check the lower-left corner and the upper-right corner
57  if( yLeft <= yy1 || yRight >= yy2 )
58  return false;
59  else
60  return true;
61  }
62 }
