145  inline std::pair< unsigned, unsigned > KTSliceHeader::GetRecordSamplePairAtSample(unsigned sampleInSlice)
146  {
147  // NOTE 1: doing these both at once, we can take advantage of compiler optimization,
148  // which will most likely only perform one division operation
149  // NOTE 2: this function does not depend on the slice size; this feature is useful
150  // to be able to calculate record offsets without worrying about which slice, exactly, this header pertains to.
151  // It's in use (as of this writing) in KTWignerVille.
152  fTemp1 = fStartSampleNumber + sampleInSlice; // sample in the record + some number of record lengths
153  fTemp2 = fStartRecordNumber + fTemp1 / fRecordSize; // record number
154  fTemp3 = fTemp1 % fRecordSize; // sample in the record
155  return std::make_pair (fTemp2, fTemp3);
156  }
