50  bool KTNNFilter::FilterByNNDist(KTKDTreeData& kdTreeData)
51  {
52  KTINFO(flog, "Performing filtering by nearest-neighbor distance on k-d tree data");
53  unsigned nComponents = kdTreeData.GetNComponents();
54  for (unsigned iComponent = 0; iComponent < nComponents; ++iComponent)
55  {
56  const KTTreeIndex< double >* kdTree = kdTreeData.GetTreeIndex(iComponent);
57  const std::vector< KTKDTreeData::Point >& setOfPoints = kdTreeData.GetSetOfPoints(iComponent);
58  std::vector< size_t > noisePoints;
59  if (! FilterByNNDist(kdTree, setOfPoints, noisePoints))
60  {
61  KTERROR(flog, "Filtering by nearest-neighbor distance failed");
62  return false;
63  }
64  if (fRemoveNoiseFlag)
65  {
66  KTDEBUG(flog, "Filtering by nearest-neighbor distance is removing " << noisePoints.size() << " points");
67  kdTreeData.RemovePoints(noisePoints); // also rebuilds k-d tree index
68  }
69  else
70  {
71  KTDEBUG(flog, "Filtering by nearest-neighbor distance is flagging " << noisePoints.size() << " points");
72  kdTreeData.FlagPoints(noisePoints); // does NOT rebuild k-d tree index
73  }
74  }
75  return true;

106  bool KTNNFilter::FilterByNNDist(const KTTreeIndex< double >* kdTree, const KTKDTreeData::SetOfPoints& setOfPoints, std::vector< size_t >& noiseIndices)
107  {
108  double maxDistSq = fMaxDist * fMaxDist;
109  unsigned nPoints = kdTree->size();
110  noiseIndices.clear();
111  double timeDelta, frequencyDelta;
112  for (unsigned iPoint = 0; iPoint < nPoints; ++iPoint)
113  {
114  KTTreeIndex< double >::Neighbors ne = kdTree->NearestNeighborsByNumber(iPoint, 2);
115  timeDelta = setOfPoints[ne[1]].fCoords[0] - setOfPoints[iPoint].fCoords[0];
116  frequencyDelta = setOfPoints[ne[1]].fCoords[1] - setOfPoints[iPoint].fCoords[1];
117  if (timeDelta*timeDelta + frequencyDelta*frequencyDelta > maxDistSq)
118  {
119  noiseIndices.push_back(iPoint);
120  }
121  }
122  return true;
123  }

125  bool KTNNFilter::FilterByMinInRadius(const KTTreeIndex< double >* kdTree, const KTKDTreeData::SetOfPoints& setOfPoints, std::vector< size_t >& noiseIndices)
126  {
127  unsigned nPoints = kdTree->size();
128  noiseIndices.clear();
129  double timeDelta, frequencyDelta;
130  unsigned nPtsInRadiusThreshold = fMinInRadius + 1; // since the neighbors returned by the k-d tree search function always includes the point itself
