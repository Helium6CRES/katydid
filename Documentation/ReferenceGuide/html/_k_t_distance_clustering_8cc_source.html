77  KTCluster1DData* KTDistanceClustering::FindClusters(const KTDiscriminatedPoints1DData* data)
78  {
79  typedef KTDiscriminatedPoints1DData::SetOfPoints OriginalPoints;
80  typedef set< UInt_t > Cluster;
81 
82  if (fCalculateMaxBinDistance) SetMaxBinDistance(KTMath::Nint(fMaxFrequencyDistance / data->GetBinWidth()));
83 
84  UInt_t nChannels = data->GetNChannels();
85 
86  KTCluster1DData* newData = new KTCluster1DData(nChannels);
87  newData->SetNBins(data->GetNBins());
88  newData->SetBinWidth(data->GetBinWidth());
89 
90  for (UInt_t iChannel=0; iChannel<nChannels; iChannel++)
91  {
92  newData->SetThreshold(data->GetThreshold(iChannel), iChannel);
93 
94  const OriginalPoints points = data->GetSetOfPoints(iChannel);
95 
96  if (! points.empty())
97  {
98  OriginalPoints::const_iterator pIt = points.begin();
99  UInt_t thisPoint = pIt->first;
100  Cluster activeCluster;
101  activeCluster.insert(thisPoint);
102  UInt_t lastPointInActiveCluster = thisPoint;
103 
104  for (pIt++; pIt != points.end(); pIt++)
105  {
106  thisPoint = pIt->first;
107  if (thisPoint - lastPointInActiveCluster > fMaxBinDistance)
108  {
109  //KTDEBUG(sdlog, "Adding cluster (ch. " << iChannel << "): " << *(activeCluster.begin()) << " " << *(activeCluster.rbegin()));
110  newData->AddCluster(*(activeCluster.begin()), *(activeCluster.rbegin()), iChannel);
111  activeCluster.clear();
112  }
113  activeCluster.insert(thisPoint);
114  lastPointInActiveCluster = thisPoint;
115  }
116  KTDEBUG(sdlog, "Adding cluster: (ch. " << iChannel << "): " << *(activeCluster.begin()) << " " << *(activeCluster.rbegin()));
117  newData->AddCluster(*(activeCluster.begin()), *(activeCluster.rbegin()), iChannel);
118  }
119 
120  KTDEBUG(sdlog, newData->GetSetOfClusters(iChannel).size() << " clusters added on channel " << iChannel);
121  }
122 
123  newData->SetName(fOutputDataName);
124  newData->SetBundle(data->GetBundle());
125 
126  fCluster1DSignal(newData);
127 
128  return newData;
129  }
