97  bool KTRQProcessor::ProcessNoiseData(KTTimeSeriesData& noise)
98  {
99  if( !(this->fNACMDidConverge) ) {
100  // Grab the first channel of data out to use.
101  const KTTimeSeriesReal* noiseDt = dynamic_cast<const KTTimeSeriesReal*>(noise.GetTimeSeries(0));
102 
103  // We need to iterate over the chunks in the time series and produce our
104  // own time series. First things first, get the pointer to the raw data
105  // held in the time series.
106  //(unused) unsigned nElem = (noiseDt->GetData()).data().size();
107  const double* rawPtr = (noiseDt->GetData()).data().begin();
108 
109  // Now we point the data map at the first fChunkSize piece of data and process it.
110  // For now we only use the first chunk to form the ACM. Despite the fact that this
111  // looks like allocating memory, it isn't.
112  new (this->fDataMap) DataMapType(rawPtr, this->fChunkSize);
113 
114  // Update noise estimate with current noise sample.
115  KTINFO(nrq_log,"using noise to update NACM estimate...");
116 
117  // Grab a tick to calculate elapsed time.
118  std::clock_t t0 = clock();
119 
120  // Calculate first row. The NACM is a Toeplitz matrix which means we only need to
121  // calculate the first row.
122  for(unsigned colIdx = 0; colIdx < this->fChunkSize; colIdx++) {
123  (*this->fNoiseACM)(0, colIdx) = KTRQProcessor::LaggedACF(this->fDataMap, colIdx);
124  }
125 
126  for(unsigned row = 1; row < this->fChunkSize; row++) {
127  for(unsigned col = 0; col < this->fChunkSize; col++) {
128  // If row = col, we are on the diagonal, which means this element is equal to
129  // row = 0, col = 0.
130  if( row == col ) (*this->fNoiseACM)(row,col) = (*this->fNoiseACM)(0,0);
131  // If row < col, we are below the diagonal, so use the transposed element for
132  // assignment.
133  if( col < row ) (*this->fNoiseACM)(row,col) = (*this->fNoiseACM)(col,row);
134  // If row > col, we are above the diagonal. Look above and to the left to get this
135  // element.
136  if( col > row ) (*this->fNoiseACM)(row,col) = (*this->fNoiseACM)(row-1, col-1);
137  }
138  }
139 
140  // all done, check elapsed time.
141  std::clock_t tf = clock();
142  double elapsed = (tf-t0)/CLOCKS_PER_SEC;
143  KTINFO(nrq_log, "Noise data processed in " << elapsed << " seconds.");
144 
145  // Check convergence criterion (TEMPORARILY JUST TRUE) and set convergence flag if
146  // appropriate.
147  if(true) {
148  this->SetNACMConverged(true);
149  KTINFO(nrq_log,"NACM has converged.");
150  KTWARN(nrq_log,"DON'T BELIEVE CONVERGE MSG, TEMPORARY CONVERGENCE CRITERION USED.");
151  }
152  }
153 
154  return true;
155  }

158  {
159  double result = 0.0;

172  bool KTRQProcessor::ProcessCandidateData(KTTimeSeriesData& c)
173  {
174  if( this->fNACMDidConverge ) {
175  // grab data from the bundle.
176  // cast data to time series real data.
177  const KTTimeSeriesReal* cDt = dynamic_cast<const KTTimeSeriesReal*>(c.GetTimeSeries(0));
178 
179  // We need to iterate over the chunks in the time series and produce our
180  // own time series. First things first, get the pointer to the raw data
181  // held in the time series.
182  unsigned nElem = (cDt->GetData()).data().size();
183  const double* rawPtr = (cDt->GetData()).data().begin();
184 
185  // Now here is our time series. The number of elements is equal to the floor of the
186  // number of elements in the incoming time series divided by the chunk size.
187  unsigned nOut = nElem/fChunkSize;
188  KTTimeSeriesData& nDt = c.Of<KTTimeSeriesData>().SetNComponents(1);
189  KTTimeSeriesReal* rqOut = new KTTimeSeriesReal(nOut);
190 
191  // Grab clock tick for elapsed time calculation
192  std::clock_t t0 = clock();
193 
194  /*
195  * Iterate over the data in the bundle, pointing the data map at each chunk consecutively.
196  * first we need to know
197  */
198  for(unsigned offset = 0; offset < nOut; offset++) {
199  new (this->fDataMap) DataMapType(rawPtr + (offset*fChunkSize), this->fChunkSize);
200 
201  // Now we should be able to compute the rayleigh quotient for this chunk.
202  double rq = this->RayleighQuotient(this->fDataMap);
203 
204  // Now set the data in the output time series.
205  rqOut->SetValue(offset, rq);
206  }
207 
208  /*
209  * Attach the new data to the KTBundle and fire the signal that indicates we have
210  * compressed this time series.
211  */
212  nDt.SetTimeSeries(rqOut);
213 
214  // all done, check elapsed time.
215  std::clock_t tf = clock();
216  double elapsed = (tf-t0)/CLOCKS_PER_SEC;
217  KTINFO(nrq_log, "Candidate record processed by NRQ processor in " << elapsed << " seconds.");
218  }
219  else {
220  KTWARN(nrq_log,"NACM has not converged - no NRQ calculation performed.");
221  }
222 
223  return true;
224  }
