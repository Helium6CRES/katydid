90  double Significance( vector<double> x, vector<int> omit, int include, std::string metric )
91  {
92  double noiseAmp = 0;
93  double noiseDev = 0;
94 
95  int s = x.size();
96  for( int i = 0; i < s; i++ )
97  {
98  if( find( omit.begin(), omit.end(), i ) == omit.end() )
99  {
100  noiseAmp += x[i] / s;
101  noiseDev += pow( x[i], 2 ) / s;
102  }
103  }
104  noiseDev = sqrt( noiseDev - pow( noiseAmp, 2 ) );
105  if( metric == "Sigma" )
106  return (x[include] - noiseAmp) / noiseDev;
107  else if( metric == "SNR" )
108  return x[include] / noiseAmp;
109  else
110  return -1.;
111  }

113  double KTLinearDensityProbeFit::findIntercept( KTDiscriminatedPoints2DData& pts, double dalpha, double q, double width )
114  {
115  double alpha = fMinFrequency;
116  double bestAlpha = 0, bestError = 0, error = 0;
117  while( alpha <= fMaxFrequency )
118  {
119  error = 0;
120 
121  // Calculate the associated error to the current value of alpha
122  for( KTDiscriminatedPoints2DData::SetOfPoints::const_iterator it = pts.GetSetOfPoints(0).begin(); it != pts.GetSetOfPoints(0).end(); ++it )
123  {
124  error -= Gaus_Eval( it->second.fOrdinate - q * it->second.fAbscissa - alpha, width );
125  }
126 
127  if( error < bestError || bestError == 0 )
128  {
129  bestError = error;
130  bestAlpha = alpha;
131  }
132 
133  // Increment alpha
134  alpha += dalpha;
135  }
136 
137  return bestAlpha;
138  }

146  bool KTLinearDensityProbeFit::Calculate(KTProcessedTrackData& data, KTDiscriminatedPoints2DData& pts)
147  {
148  KTLinearFitResult& newData = data.Of< KTLinearFitResult >();
149  KTPSCollectionData& fullSpectrogram = data.Of< KTPSCollectionData >();
150 
151  newData.SetNComponents( 2 );
152  newData.SetSlope( data.GetSlope(), 0 );
153  newData.SetSlope( data.GetSlope(), 1 );
154  newData.SetTrackDuration( data.GetEndTimeInRunC() - data.GetStartTimeInRunC(), 0 );
155  newData.SetTrackDuration( data.GetEndTimeInRunC() - data.GetStartTimeInRunC(), 1 );
156 
157  double intercept1, intercept2;
158 
159  PerformTest( pts, newData, fProbeWidthBig, fStepSizeBig, 0 );
160  PerformTest( pts, newData, fProbeWidthSmall, fStepSizeSmall, 1 );
161 
162  KTDEBUG(sdlog, "Beginning fourier analysis of sideband" );
163 
164  intercept1 = newData.GetIntercept( 0 );
165  intercept2 = newData.GetIntercept( 1 );
166 
167  newData.SetSidebandSeparation( intercept1 - intercept2, 0 );
168  newData.SetSidebandSeparation( intercept1 - intercept2, 1 );
169 
170  // We will need to calculate the unweighted projection first
171  double delta_f;
172  double alphaBound_upper = intercept1 + 1e6;
173  double alphaBound_lower = intercept1 - 1e6;
174 
175  int xBinStart, xBinEnd, xWindow, yBinStart, yBinEnd, yWindow;
176  double ps_xmin, ps_xmax, ps_ymin, ps_dx, ps_dy;
177  double q_fit = newData.GetSlope( 0 );
178  double x, y;
179 
180  ps_xmin = fullSpectrogram.GetStartTime();
181  ps_xmax = fullSpectrogram.GetEndTime();
182  ps_ymin = fullSpectrogram.GetSpectra().begin()->second->GetRangeMin();
183  ps_dx = fullSpectrogram.GetDeltaT();
184  ps_dy = fullSpectrogram.GetSpectra().begin()->second->GetFrequencyBinWidth();
185 
186  // We add +1 for the underflow bin
187  xBinStart = floor( (data.GetStartTimeInRunC() - ps_xmin) / ps_dx ) + 1;
188  xBinEnd = floor( (data.GetEndTimeInRunC() - ps_xmin) / ps_dx ) + 1;
189  xWindow = xBinEnd - xBinStart + 1;
190  KTINFO(sdlog, "Set xBin range to " << xBinStart << ", " << xBinEnd);
191 
192  newData.SetFit_width( xWindow, 0 );
193  newData.SetFit_width( xWindow, 1 );
194 
195  // The y window this time will be floating, but its size will be consistent
196  // The number of bins between the alpha bounds
197  yWindow = ceil( (alphaBound_upper - alphaBound_lower) / ps_dy );
198 
199  //double *unweighted = new double[xWindow];
200  vector< double > unweighted, weighted, fourier;
201 
202  KTDEBUG(sdlog, "Computing unweighted projection");
203 
204  int i = 0;
205  // First we compute the unweighted projection
206  for( std::map< double, KTPowerSpectrum* >::const_iterator it = fullSpectrogram.GetSpectra().begin(); it != fullSpectrogram.GetSpectra().end(); ++it )
207  {
208  // Set x value and starting y-bin
209  x = ps_xmin + (i - 1) * ps_dx;
210  yBinStart = it->second->FindBin( alphaBound_lower + q_fit * x );
211 
212  // Unweighted power = sum of raw power spectrum
213  unweighted.push_back( 0 );
214  for( int j = yBinStart; j < yBinStart + yWindow; j++ )
215  {
216  y = ps_ymin + ps_dy * (j - 1);
217 
218  // We reevaluate the spline rather than deal with the appropriate index of power_minus_bkgd
219  unweighted[i] += (*it->second)(j) - fGVData.GetSpline()->Evaluate( y );
220  }
221  i++;
222  }
223 
224  KTDEBUG(sdlog, "Computing weighted projection");
225 
226  i = 0;
227 
228  // Weighted projection
229  double cumulative;
230  for( std::map< double, KTPowerSpectrum* >::const_iterator it = fullSpectrogram.GetSpectra().begin(); it != fullSpectrogram.GetSpectra().end(); ++it )
231  {
232  cumulative = 0.;
233 
234  x = ps_xmin + (i - 1) * ps_dx;
235  yBinStart = it->second->FindBin( alphaBound_lower + q_fit * x );
236 
237  for( int j = yBinStart; j < yBinStart + yWindow; j++ )
238  {
239  y = ps_ymin + ps_dy * (j - 1);
240 
241  // Calculate delta-f using the fit values
242  delta_f = y - (q_fit * x + newData.GetIntercept(0));
243  cumulative += delta_f * ((*it->second)(j) - fGVData.GetSpline()->Evaluate( y )) / unweighted[i];
244  }
245 
246  weighted.push_back( cumulative );
247  i++;
248  }
249 
250  // Discrete Cosine Transform (real -> real) of type I
251  // Explicit, not fast (i.e. n^2 operations)
252 
253  double pi = 3.14159265358979;
254  for( int k = 0; k < xWindow; k++ )
255  {
256  cumulative = 0.;
257  for( int n = 1; n <= xWindow - 2; n++ )
258  cumulative += weighted[n] * cos( n * k * pi / (xWindow - 1) );
259 
260  fourier.push_back( pow( 0.5 * (weighted[0] + pow( -1, k ) * weighted[xWindow - 1]) + cumulative, 2 ) );
261  }
262 
263  // Evaluate SNR and std dev significance of the largest fourier peak
264 
265  double avg_fourier = 0.;
266  double max_fourier = 0.;
267  double freq_step = 1/(2 * (xWindow - 1) * ps_dx);
268 
269  for( int k = 0; k < xWindow; k++ )
270  avg_fourier += fourier[k] / xWindow;
271 
272  for( int k = 0; k < xWindow; k++ )
273  {
274  if( fourier[k] > max_fourier )
275  {
276  max_fourier = fourier[k];
277  newData.SetFFT_peak( double(k) * freq_step, 0 );
278  newData.SetFFT_SNR( max_fourier / avg_fourier, 0 );
279  newData.SetFFT_peak( double(k) * freq_step, 1 );
280  newData.SetFFT_SNR( max_fourier / avg_fourier, 1 );
281  }
282  }
283 
284  KTINFO(sdlog, "Successfully obtained power modulation. Maximum fourier peak at frequency " << newData.GetFFT_peak( 0 ) << " with SNR " << newData.GetFFT_SNR( 0 ));
285 
286  return true;
287  }
