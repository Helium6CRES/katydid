204  inline Double_t KTComplexFFTW::GetMinFrequency(Double_t timeBinWidth) const
205  {
206  // There's one bin at the center, always: the DC bin.
207  // # of bins on the negative side is nFreqBins/2 (rounded down because of integer division).
208  // 0.5 is added to the # of bins because of the half of the DC bin on the negative frequency side.
209  return -GetFrequencyBinWidth(timeBinWidth) * (Double_t(fSize/2) + 0.5);
210  }
211 
212  inline Double_t KTComplexFFTW::GetMaxFrequency(Double_t timeBinWidth) const
213  {
214  // There's one bin at the center, always: the DC bin.
215  // # of bins on the positive side is nFreqBins/2 if the number of bins is odd, and nFreqBins/2-1 if the number of bins is even (division rounded down because of integer division).
216  // 0.5 is added to the # of bins because of the half of the DC bin on the positive frequency side.
217  UInt_t nBinsToSide = fSize / 2;
218  return GetFrequencyBinWidth(timeBinWidth) * (Double_t(nBinsToSide*2 == fSize ? nBinsToSide - 1 : nBinsToSide) + 0.5);
219  }
