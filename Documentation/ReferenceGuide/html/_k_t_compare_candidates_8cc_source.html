53 
54  if (events.size() == 0)
55  {
56  KTINFO(cclog, "No events are present");
57  // return true;

72  return false;

74 
75  // eventMatches: each position in the vector represents one of the events; this vector records how many candidates matched each event.
76  vector< unsigned > eventMatches(events.size());
77  // candidateMatches: each position in the vector represents one of the candidates; this vector records how many events matched each candidate.
78  vector< unsigned > candidateMatches(candidates.size());
79 
80  //uint64_t candidateSampleSum = 0;
81  //uint64_t eventSampleSum = 0;
82  //uint64_t candidateCorrectSampleSum = 0;

88  unsigned candidateCounterStart = 0;

92  {
93  candidateCounter = candidateCounterStart;
94  for (KTAnalysisCandidates::CandidateSet::const_iterator candIt = candStartHere; candIt != candidates.end(); candIt++)
95  {
96  comparison = CompareAnEventToACandidate(*truthIt, *candIt, eventRecordSize, candidateRecordSize);
97  if (comparison == 0)
98  {
99  eventMatches[eventCounter] = eventMatches[eventCounter] + 1;
100  candidateMatches[candidateCounter] = candidateMatches[candidateCounter] + 1;
101  //KTDEBUG(cclog, "Match found:\n" <<
102  // "\tEvent " << eventCounter << "): " << truthIt->fStartRecord << ", " << truthIt->fStartSample << ", " << truthIt->fEndRecord << ", " << truthIt->fEndSample << '\n' <<
103  // "\tCandidate " << candidateCounter << "): " << candIt->fStartRecord << ", " << candIt->fStartSample << ", " << candIt->fEndRecord << ", " << candIt->fEndSample << '\n');
104  }
105  else if (comparison < 0)
106  {
107  //KTDEBUG(cclog, "Candidate before event:\n" <<
108  // "\tEvent: " << truthIt->fStartRecord << ", " << truthIt->fStartSample << ", " << truthIt->fEndRecord << ", " << truthIt->fEndSample << '\n' <<
109  // "\tCandidate: " << candIt->fStartRecord << ", " << candIt->fStartSample << ", " << candIt->fEndRecord << ", " << candIt->fEndSample << '\n');
110  if (fAssumeSparseCandidates)
111  {
112  // this candidate is earlier than any remaining events, so it can be skipped for following events
113  candStartHere = candIt;
114  candidateCounterStart = candidateCounter;
115  // move up to the next event as the place to start
116  candStartHere++;
117  candidateCounterStart++;
118  // if this is the last candidate, than there's no use in doing any more comparisons to events
119  if (candStartHere == candidates.end())
120  {
121  continueEventLoop = false;
122  // if we're here, the candidate loop will complete after this cycle
123  }
124  }
125  }
126  else // comparison > 0
127  {
128  //KTDEBUG(cclog, "Candidate after event:\n" <<
129  // "\tEvent: " << truthIt->fStartRecord << ", " << truthIt->fStartSample << ", " << truthIt->fEndRecord << ", " << truthIt->fEndSample << '\n' <<
130  // "\tCandidate: " << candIt->fStartRecord << ", " << candIt->fStartSample << ", " << candIt->fEndRecord << ", " << candIt->fEndSample << '\n');
131  // all remaining candidates are after the current truth event, so they can be skipped
132  break;
133  }
134  candidateCounter++;
135  }
136  eventCounter++;
137  }
138 
139  Nymph::KTDataPtr dataPtr(new Nymph::KTData());
140  dataPtr->SetLastData(true);
141  KTCCResults& ccrData = dataPtr->Of< KTCCResults >();

144  ccrData.Setdfdt(mcEventData.Getdfdt());
145  ccrData.SetSignalPower(mcEventData.GetSignalPower());
146 
147  ccrData.SetNEvents(events.size());
148  ccrData.SetNCandidates(candidates.size());
149

151  unsigned largestNumberOfMatches = 0;
152  unsigned nEventsWithAtLeastOneCandidateMatch = 0;

154  for (unsigned iEvent = 0; iEvent < eventMatches.size(); iEvent++)

162  nEventsWithAtLeastOneCandidateMatch++;

165  //nEventsWithCandidateMatches[eventMatches[iEvent]] = nEventsWithCandidateMatches[eventMatches[iEvent]] + 1;
166  }
167  //nEventsWithCandidateMatches.resize(largestNumberOfMatches + 1);
168  ccrData.ResizeNEventsWithXCandidateMatches(largestNumberOfMatches + 1);
169 
170  KTPROG(cclog, "Number of events: " << ccrData.GetNEvents());
171  KTINFO(cclog, "Largest number of candidates matching an event: " << largestNumberOfMatches);
172  std::stringstream textHist1;
173  const vector< unsigned >& nEventsWithXCandidateMatches = ccrData.GetNEventsWithXCandidateMatches();
174  for (unsigned iNEvents = 0; iNEvents < nEventsWithXCandidateMatches.size(); iNEvents++)
175  {

202  for (unsigned iNCandidates = 0; iNCandidates < nCandidatesWithXEventMatches.size(); iNCandidates++)
203  {

205  }

207 
208  ccrData.SetFalseRate(1.e6 * double(nCandidatesWithXEventMatches[0]) / (double(nRecords) * double(eventRecordSize)));

211  fCCResultsSignal(dataPtr);

216  int KTCompareCandidates::CompareAnEventToACandidate(const KTMCTruthEvents::Event& event, const KTAnalysisCandidates::Candidate& candidate, unsigned eventRecordSize, unsigned candidateRecordSize) const
217  {
218  /* more readable version
219  uint64_t candidateSampleStart = candidate.fStartRecord * candidateRecordSize + candidate.fStartSample;
220  uint64_t eventSampleEnd = event.fEndRecord * eventRecordSize + event.fEndSample;
221  if (candidateSampleStart > eventSampleEnd) return 1;
222 
223  uint64_t candidateSampleEnd = candidate.fEndRecord * candidateRecordSize + candidate.fEndSample;
224  uint64_t eventSampleStart = event.fStartRecord * eventRecordSize + event.fStartSample;
225  if (candidateSampleEnd < eventSampleStart) return -1;
226  */
227 
228  if (candidate.fStartRecord * candidateRecordSize + candidate.fStartSample >
229  event.fEndRecord * eventRecordSize + event.fEndSample)
230  return 1;
231 
232  if (candidate.fEndRecord * candidateRecordSize + candidate.fEndSample <
233  event.fStartRecord * eventRecordSize + event.fStartSample)
234  return -1;
235 
236  return 0;
237  }
