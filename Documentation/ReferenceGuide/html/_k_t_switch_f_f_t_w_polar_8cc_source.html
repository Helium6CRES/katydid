47  bool KTSwitchFFTWPolar::SwitchToPolar(KTFrequencySpectrumDataFFTW& fsData)
48  {
49  unsigned nComponents = fsData.GetNComponents();
50 
51  KTFrequencySpectrumDataPolar& newData = fsData.Of< KTFrequencySpectrumDataPolar >().SetNComponents(nComponents);
52 
53  for (unsigned iComponent=0; iComponent<nComponents; ++iComponent)
54  {
55  KTFrequencySpectrumPolar* newSpectrum = fsData.GetSpectrumFFTW(iComponent)->CreateFrequencySpectrumPolar();
56  if (newSpectrum == NULL)
57  {
58  KTERROR(swlog, "Switch of spectrum " << iComponent << " (fftw->polar) failed for some reason. Continuing processing.");
59  continue;
60  }
61  else
62  {
63  KTFrequencySpectrumFFTW* oldSpectrum = fsData.GetSpectrumFFTW(iComponent);
64  KTDEBUG(swlog, "fftw array: " << oldSpectrum->size() << " bins; range: " << oldSpectrum->GetRangeMin() << " - " << oldSpectrum->GetRangeMax());
65  KTDEBUG(swlog, "polar array: " << newSpectrum->size() << " bins; range: " << newSpectrum->GetRangeMin() << " - " << newSpectrum->GetRangeMax());
66  }
67  newData.SetSpectrum(newSpectrum, iComponent);
68  }
69  KTINFO(swlog, "Completed switch (fftw->polar) of " << nComponents << " frequency spectra (polar)");
70 
71  return true;
72  }
