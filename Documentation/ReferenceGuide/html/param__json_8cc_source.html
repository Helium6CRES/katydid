33  {
34  FILE* t_config_file = fopen( a_filename.c_str(), "r" );
35  if( t_config_file == NULL )
36  {
37  LERROR( dlog, "file <" << a_filename << "> did not open" );
38  return NULL;
39  }
40  rapidjson::FileStream t_file_stream( t_config_file );
41 
42  rapidjson::Document t_config_doc;
43  if( t_config_doc.ParseStream<0>( t_file_stream ).HasParseError() )
44  {
45  unsigned errorPos = t_config_doc.GetErrorOffset();
46  rewind( t_config_file );
47  unsigned iChar, newlineCount = 1, lastNewlinePos = 0;
48  int thisChar;
49  for( iChar = 0; iChar != errorPos; ++iChar )
50  {
51  thisChar = fgetc( t_config_file );
52  if( thisChar == EOF )
53  {
54  break;
55  }
56  if( thisChar == '\n' || thisChar == '\r' )
57  {
58  newlineCount++;
59  lastNewlinePos = iChar + 1;
60  }
61  }
62  if( iChar == errorPos )
63  {
64  LERROR( dlog, "error parsing config file :\n" <<
65  '\t' << t_config_doc.GetParseError() << '\n' <<
66  "\tThe error was reported at line " << newlineCount << ", character " << errorPos - lastNewlinePos );
67  }
68  else
69  {
70  LERROR( dlog, "error parsing config file :\n" <<
71  '\t' << t_config_doc.GetParseError() <<
72  "\tend of file reached before error location was found" );
73  }
74  fclose( t_config_file );
75  return NULL;
76  }
77  fclose( t_config_file );
