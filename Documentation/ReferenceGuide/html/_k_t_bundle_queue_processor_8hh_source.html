171  template< class XProcessorType >
172  Bool_t KTBundleQueueProcessorTemplate< XProcessorType >::ProcessQueue()
173  {
174  KTDEBUG(eqplog, "Beginning to process publication queue");
175  while (fStatus != kStopped)
176  {
177  KTDEBUG(eqplog, "processing . . .");
178  boost::shared_ptr<KTBundle> bundleToPublish;
179  if (fQueue.wait_and_pop(bundleToPublish))
180  {
181  KTDEBUG(eqplog, "Bundle acquired for publishing");
182  (static_cast<XProcessorType*>(this)->*fFuncPtr)(bundleToPublish);
183  if (bundleToPublish->GetIsLastBundle()) fStatus = kStopped;
184  }
185  }
186  return true;
187  }
