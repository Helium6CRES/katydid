59  bool KTTrackProcessing::ProcessTrack(KTSparseWaterfallCandidateData& swfData, KTHoughData& htData)
60  {
61  unsigned component = swfData.GetComponent();
62  unsigned trackID = swfData.GetCandidateID();
63 
64  typedef KTSparseWaterfallCandidateData::Points Points;
65  // not const because points will be removed later
66  Points& points = swfData.GetPoints();
67 
68  // not const because the HT will be smoothed in place
69  KTPhysicalArray< 2, double >* houghTransform = htData.GetTransform(component);
70 
71  // NOTE: smoothes the actual data, not a copy
72  if (! KTSmooth::Smooth(houghTransform))
73  {
74  KTERROR(tlog, "Error while smoothing Hough Transform");
75  return false;
76  }
77 
78  unsigned maxBinX, maxBinY;
79  houghTransform->GetMaximumBin(maxBinX, maxBinY);
80 
81  double htAngle = houghTransform->GetBinCenter(1, maxBinX);
82  double htRadius = houghTransform->GetBinCenter(2, maxBinY);
83 
84  // Hough line: a*x + b*y + c = 0, c = -radius
85  double htCosAngle = cos(htAngle); // ht_a
86  double htSinAngle = sin(htAngle); // ht_b
87 
88  // scaling for Hough Transform, used throughout for line fits and modifications
89  // "scaled" means the axis has been scaled to 0-1, as was the case when the Hough Transform was calculated
90  // "unscaled" means the axis is in the original units
91  // unscaled = scaled * scale + offset
92  double xScale = htData.GetXScale(component);
93  double yScale = htData.GetYScale(component);
94  double xOffset = htData.GetXOffset(component);
95  double yOffset = htData.GetYOffset(component);
96 
97  double htSlope = -1.0 * (htCosAngle * yScale) / (htSinAngle * xScale);
98  double htIntercept = (htRadius / htSinAngle) * yScale + yOffset - htSlope * xOffset; // this is r/sin(angle), rescaled, plus extra from the xOffset
99 
100  KTDEBUG(tlog, "Hough Transform track processing results:\n"
101  << "\tmaxBinX: " << maxBinX << "\t maxBinY: " << maxBinY << "\t angle: " << htAngle << "\t radius: " << htRadius << '\n'
102  << "\ta=cos(angle): " << htCosAngle << "\t b=sin(angle): " << htSinAngle << '\n'
103  << "\txScale: " << xScale << "\t yScale: " << yScale << '\n'
104  << "\tSlope: " << htSlope << " Hz/s" << '\n'
105  << "\tIntercept: " << htIntercept << " Hz");
106 
107  // First loop over points, and first point-line-distance cut
108 
109  unsigned nPoints = points.size();
110  typedef vector< std::pair< double, double > > SimplePoints2D;
111  SimplePoints2D pointsScaled, pointsUnscaled, pointsUnscaledInAcq;
112  pointsScaled.reserve(nPoints);
113  pointsUnscaledInAcq.reserve(nPoints);
114  vector< unsigned > pointsCuts;
115  pointsCuts.reserve(nPoints);
116  unsigned nPointsUsed = 0;
117  unsigned nPointsCut1 = 0;
118  double xScaled, yScaled, distance;
119  double sumX = 0., sumY = 0., sumX2 = 0., sumXY = 0.; // for least-squares calculation
120  // first distance cut, plus calculation of the initial least-squares line calculations
121  for (Points::const_iterator pIt = points.begin(); pIt != points.end(); ++pIt)
122  {
123  //cout << "calculating a distance..." << endl;
124  pointsUnscaled.push_back(SimplePoints2D::value_type(pIt->fTimeInRunC, pIt->fFrequency));
125  pointsUnscaledInAcq.push_back(SimplePoints2D::value_type(pIt->fTimeInAcq, pIt->fFrequency));
126  xScaled = (pIt->fTimeInRunC - xOffset) / xScale;
127  yScaled = (pIt->fFrequency - yOffset) / yScale;
128  pointsScaled.push_back(SimplePoints2D::value_type(xScaled, yScaled));
129  //cout << "i: " << iPoint << "\t y_i: " << ys[iPoint] << "\t x_i: " << xs[iPoint] << endl;
130  //cout << "scaled: y_isc: " << yiscaled << "\t xisc: " << xiscaled << endl;
131  distance = PointLineDistance(xScaled, yScaled, htCosAngle, htSinAngle, -htRadius);
132  //cout << "distance: " << distance << endl;
133 
134  if (distance < fPointLineDistCut1 )
135  {
136  // point is not cut
137  ++nPointsUsed;
138  sumX += xScaled;
139  sumY += yScaled;
140  sumX2 += xScaled * xScaled;
141  sumXY += xScaled * yScaled;
142  pointsCuts.push_back(0);
143  }
144  else
145  {
146  // point is cut
147  ++nPointsCut1;
148  pointsCuts.push_back(1);
149  }
150  } // loop over points
151  KTDEBUG(tlog, "Points removed with cut 1: " << nPointsCut1);
152 
153  // Refine the line with a least-squares line calculation
154  double xMean = sumX / (double)nPointsUsed;
155  double yMean = sumY / (double)nPointsUsed;
156  double lsSlopeScaled = (sumXY - sumX * yMean) / (sumX2 - sumX * xMean);
157  double lsInterceptScaled = yMean - lsSlopeScaled * xMean;
158  double lsSlope = lsSlopeScaled * yScale / xScale;
159  double lsIntercept = lsInterceptScaled * yScale + yOffset - lsSlope * xOffset;
160  KTDEBUG(tlog, "Lease-squares fit result\n"
161  << "\tSlope: " << lsSlope << " Hz/s\n"
162  << "\tIntercept: " << lsIntercept << " Hz");
163 
164  // second distance cut based on LS fit
165  double startTime = std::numeric_limits< double >::max();
166  double stopTime = -1.;
167  double startTimeInAcq = 0;
168  double startFreq, stopFreq;
169  nPointsUsed = 0;
170  unsigned nPointsCut2 = 0;
171  for (unsigned iPoint = 0; iPoint < nPoints; ++iPoint)
172  {
173  distance = PointLineDistance(pointsScaled[iPoint].first, pointsScaled[iPoint].second, lsSlopeScaled, -1., lsInterceptScaled);;
174 
175  if (pointsCuts[iPoint] == 0 && distance < fPointLineDistCut2)
176  {
177  // point is not cut
178  ++nPointsUsed;
179  if (pointsUnscaled[iPoint].first < startTime) //possibly update start time/frequency
180  {
181  startTime = pointsUnscaled[iPoint].first;
182  startFreq = startTime * lsSlope + lsIntercept;
183  startTimeInAcq = pointsUnscaledInAcq[iPoint].first;
184  }
185  if (pointsUnscaled[iPoint].first > stopTime)
186  { //possibly update stop time/frequency
187  stopTime = pointsUnscaled[iPoint].first;
188  stopFreq = stopTime * lsSlope + lsIntercept;
189  }
190  }
191  else if (pointsCuts[iPoint] == 0)
192  {
193  // point is cut
194  ++nPointsCut2;
195  pointsCuts[iPoint] = 2;
196  }
197  }
198  KTDEBUG(tlog, "Points removed with cut 2: " << nPointsCut2);
199 
200  // Remove points and sum amplitudes
201  Points::iterator pItMaster = points.begin();
202  Points::iterator pItCache;
203  double amplitudeSum = 0.;
204  for (unsigned iPoint = 0; iPoint < nPoints; ++iPoint)
205  {
206  pItCache = pItMaster;
207  ++pItMaster;
208  if (pointsCuts[iPoint] == 0)
209  {
210  amplitudeSum += pItCache->fAmplitude;
211  }
212  else
213  {
214  points.erase(pItCache);
215  }
216  }
217  KTDEBUG(tlog, "Points present after cuts: " << points.size());
218 
219  // Add the new data
220  KTProcessedTrackData& procTrack = htData.Of< KTProcessedTrackData >();
221  procTrack.SetComponent(component);
222  procTrack.SetAcquisitionID(swfData.GetAcquisitionID());
223  procTrack.SetTrackID(trackID);
224 
225  if (lsSlope < fSlopeMinimum || points.size() < fProcTrackMinPoints)
226  {
227  procTrack.SetIsCut(true);
228  }
229 
230  procTrack.SetStartTimeInAcq(startTimeInAcq);
231  procTrack.SetStartTimeInRunC(startTime);
232  procTrack.SetEndTimeInRunC(stopTime);
233  procTrack.SetTimeLength(stopTime - startTime);
234  procTrack.SetStartFrequency(startFreq);
235  procTrack.SetEndFrequency(stopFreq);
236  procTrack.SetFrequencyWidth(stopFreq - startFreq);
237  procTrack.SetSlope(lsSlope);
238  procTrack.SetIntercept(lsIntercept);
239  procTrack.SetTotalPower(amplitudeSum);
240  //TODO: Add calculation of uncertainties
241 
242  return true;
243  }
