87  bool KTCreateKDTree::AddPoints(KTSliceHeader& slHeader, KTDiscriminatedPoints1DData& discPoints)
88  {
89 
90  KTDEBUG(kdlog, "Is this a new acquisition? fHaveNewData=" << fHaveNewData << " and GetIsNewAcquisition=" << slHeader.GetIsNewAcquisition());
91  // first check to see if this is a new acquisition; if so, run clustering on the previous acquistion's data
92  if (fHaveNewData && slHeader.GetIsNewAcquisition())
93  {
94  KTDEBUG(kdlog, "New Acquisition - Run clustering on previous acquistion, clear out old points from tree");
95  if (! MakeTree(false))
96  {
97  KTERROR(kdlog, "An error occurred while clustering from the previous acquisition");
98  return false;
99  }
100  if (! ClearTree(false))
101  {
102  KTERROR(kdlog, "An error occurred while clearing the tree");
103  return false;
104  }
105  }
106 
107  // update time and frequency bin widths, though they probably don't actually need to be updated
108  //fTimeBinWidth = slHeader.GetSliceLength();
109  //fFreqBinWidth = discPoints.GetBinWidth();
110 
111  // verify that we have the right number of components
112  unsigned nComponents = slHeader.GetNComponents();
113  if (nComponents > fTreeData.GetNComponents())
114  {
115  fTreeData.SetNComponents(nComponents);
116  }
117  fTreeData.SetAcquisitionID(slHeader.GetAcquisitionID());
118 
119  KTKDTreeData::Point newPoint;
120  newPoint.fSliceNumber = slHeader.GetSliceNumber();
121  if (newPoint.fSliceNumber > fTreeData.GetLastSlice()) fTreeData.SetLastSlice(newPoint.fSliceNumber);
122  newPoint.fCoords[0] = fInvScalingX * (slHeader.GetTimeInRun() + 0.5 * slHeader.GetSliceLength());
123  for (unsigned iComponent = 0; iComponent != nComponents; ++iComponent)
124  {
125  const KTDiscriminatedPoints1DData::SetOfPoints& incomingPts = discPoints.GetSetOfPoints(iComponent);
126  for (KTDiscriminatedPoints1DData::SetOfPoints::const_iterator pIt = incomingPts.begin();
127  pIt != incomingPts.end(); ++pIt)
128  {
129  newPoint.fCoords[1] = fInvScalingY * pIt->second.fAbscissa;
130  newPoint.fAmplitude = pIt->second.fOrdinate;
131  newPoint.fTimeInAcq = fInvScalingX * (slHeader.GetTimeInAcq() + 0.5 * slHeader.GetSliceLength());
132  fTreeData.AddPoint(newPoint, iComponent);
133  }
134  KTDEBUG(kdlog, "Tree data (component " << iComponent << ") now has " << fTreeData.GetSetOfPoints(iComponent).size() << " points (Slice Number: " << newPoint.fSliceNumber << ")");
135  }
136 
137  ++fSliceInWindowCount;
138  KTDEBUG(kdlog, "Slice-count-in-window is now " << fSliceInWindowCount << " (window size = " << fWindowSize << "; window overlap = " << fWindowOverlap << ")");
139  // if fSliceInWindowCount == fWindowSize, then we need to move the window before taking the next slice
140  if (fWindowSize != 0 && fSliceInWindowCount == fWindowSize)
141  {
142  if (! MakeTree(true) || ! ClearTree(true, slHeader.GetSliceNumber() - fWindowOverlap + 1))
143  {
144  KTERROR(kdlog, "An error occurred while clustering or clearing the tree");
145  return false;
146  }
147  fSliceInWindowCount = 0;
148  }
149 
150  fHaveNewData = true;
151  return true;
152  }

163  bool KTCreateKDTree::AddPoints(KTSparseWaterfallCandidateData& swfcData)
164  {
165  // verify that we have the right number of components
166  unsigned component = swfcData.GetComponent();
167  if (component >= fTreeData.GetNComponents())
168  {
169  fTreeData.SetNComponents(component + 1);
170  }
171 
172  KTKDTreeData::Point newPoint;
173  newPoint.fSliceNumber = 0; // slice number isn't available in KTSparseWaterfallCandidateData
174  if (newPoint.fSliceNumber > fTreeData.GetLastSlice()) fTreeData.SetLastSlice(newPoint.fSliceNumber);
175  const KTSparseWaterfallCandidateData::Points& incomingPts = swfcData.GetPoints();
176  for (KTSparseWaterfallCandidateData::Points::const_iterator pIt = incomingPts.begin();
177  pIt != incomingPts.end(); ++pIt)
178  {
179  newPoint.fCoords[0] = fInvScalingX * pIt->fTimeInRunC;
180  newPoint.fCoords[1] = fInvScalingY * pIt->fFrequency;
181  newPoint.fAmplitude = pIt->fAmplitude;
182  newPoint.fTimeInAcq = fInvScalingX * pIt->fTimeInAcq;
183  fTreeData.AddPoint(newPoint, component);
184  }
185  KTDEBUG(kdlog, "Tree data (component " << component << ") now has " << fTreeData.GetSetOfPoints(component).size() << " points");
186 
187  fHaveNewData = true;

219  bool KTCreateKDTree::ClearTree(bool willContinue, uint64_t firstSliceKept)
220  {
221  KTDEBUG(kdlog, "ClearTree");
222  if (! fHaveNewData) return true;
223 
224  // firstSliceKept is only used if willContinue == true
225  if (willContinue)
226  {
227  KTDEBUG(kdlog, "ClearTree(true)");
228 
229  // clear data up to the
230  unsigned nComponents = fTreeData.GetNComponents();
231  for (unsigned iComponent = 0; iComponent != nComponents; ++iComponent)
232  {
233  fTreeData.ClearIndex(iComponent);
234  KTKDTreeData::SetOfPoints& points = fTreeData.GetSetOfPoints(iComponent);
235  KTKDTreeData::SetOfPoints::iterator startErase = points.begin();
236  KTKDTreeData::SetOfPoints::iterator endErase = startErase;
237  while (endErase != points.end() && endErase->fSliceNumber < firstSliceKept)
238  {
239  ++endErase;
240  }
241  points.erase(startErase, endErase);
242  }
243  }
244  else
245  {
246  KTDEBUG(kdlog, "ClearTree(false)");
247  // clear all data from the tree, but leave the memory intact
248  unsigned nComponents = fTreeData.GetNComponents();
249  for (unsigned iComponent = 0; iComponent != nComponents; ++iComponent)
250  {
251  fTreeData.ClearPoints(iComponent);
252  }
253  }
254 
255  fHaveNewData = false;
256  return true;
257  }
