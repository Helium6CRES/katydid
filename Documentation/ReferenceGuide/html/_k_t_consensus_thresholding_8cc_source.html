69  {

71  unsigned nComponents = kdTreeData.GetNComponents();

96  {
97  unsigned nPoints = kdTree->size();

99  double timeDelta, frequencyDelta;
100  for (unsigned iPoint = 0; iPoint < nPoints; ++iPoint)
101  {
102  KTDEBUG(ctlog, "checking point (" << iPoint + 1 << "/" << nPoints << ")");

104  (this->*fFindDeltasPtr)(kdTree, setOfPoints, iPoint, timeDelta, frequencyDelta);
105 
106  unsigned voteCount = 0;
107  if (! (timeDelta == 0))

109  //double slope = frequencyDelta / timeDelta;

111 
112  bool closeEnough = true;

125  }
126  else

129  }
130  }
131  closeEnough = true;

136  test_pt[1] = setOfPoints[iPoint].fCoords[1] + k * frequencyDelta;

165  void KTConsensusThresholding::FindDeltasNeighborsInRadius(const KTTreeIndex< double >* kdTree, const KTKDTreeData::SetOfPoints& setOfPoints, unsigned pid, double& deltaTime, double& deltaFreq)
166  {
167  KTTreeIndex< double >::Neighbors ne = kdTree->NearestNeighborsByRadius(pid, fMembershipRadius);
168  unsigned nNeighbors = ne.size();
169  double sumX, sumY, sumX2, sumXY;
170  for (unsigned iNe = 0; iNe < nNeighbors; ++iNe)
171  {
172  sumX += setOfPoints[ne[iNe]].fCoords[0];
173  sumY += setOfPoints[ne[iNe]].fCoords[1];
174  sumX2 += setOfPoints[ne[iNe]].fCoords[0] * setOfPoints[ne[iNe]].fCoords[0];
175  sumXY += setOfPoints[ne[iNe]].fCoords[0] * setOfPoints[ne[iNe]].fCoords[1];
176  }
177  //double xMean = sumX / (double)nNeighbors;
178  //double yMean = sumY / (double)nNeighbors;
179  double slope = (sumXY - sumX * sumY / (double)nNeighbors) / (sumX2 - sumX * sumX / (double)nNeighbors);
180  //double intercept = yMean - slope * xMean;
181  deltaTime = sqrt(fMembershipRadius * fMembershipRadius / (1. + slope*slope));
182  deltaFreq = slope * deltaTime;
183  //deltaFreq = deltaTime * (sumXY - sumX * sumY / (double)nNeighbors) / (sumX2 - sumX * deltaTime); // a.k.a. deltaTime * slope
184  return;
185  }
