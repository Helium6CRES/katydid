162 void Viewer::DoButton()
163 {
164  TGButton* button = (TGButton*)gTQSender;
165  UInt_t id = button->WidgetId();
166 
167  double xmin, ymin, xmax, ymax;
168  double xdiff, ydiff;
169  fCanvas->GetRange(xmin,ymin,xmax,ymax);
170  xdiff = xmax - xmin; ydiff = ymax - ymin;
171 
172  fCanvas->cd();
173 
174  switch (id) {
175  case 0: { // Ellipse
176  TEllipse* ellipse = new TEllipse(gRandom->Uniform(.8*xdiff)+.1*xdiff,
177  gRandom->Uniform(.8*ydiff)+.1*ydiff,
178  gRandom->Uniform(.1*xdiff),
179  gRandom->Uniform(.1*ydiff));
180  ellipse->SetFillColor(Color_t(gRandom->Uniform(10)));
181  ellipse->Draw();
182  fCanvas->Modified();
183  fCanvas->Update();
184  break;
185  }
186  case 1: { // Box
187  TBox* box = new TBox(gRandom->Uniform(xdiff)+xmin,
188  gRandom->Uniform(ydiff)+ymin,
189  gRandom->Uniform(xdiff)+xmin,
190  gRandom->Uniform(ydiff)+ymin);
191  box->SetFillColor(Color_t(gRandom->Uniform(10)));
192  box->Draw();
193  fCanvas->Modified();
194  fCanvas->Update();
195  break;
196  }
197  case 2: { // Triangle
198  double x[4], y[4];
199  x[0] = x[3] = gRandom->Uniform(xdiff)+xmin;
200  y[0] = y[3] = gRandom->Uniform(ydiff)+ymin;
201  x[1] = gRandom->Uniform(xdiff)+xmin;
202  y[1] = gRandom->Uniform(ydiff)+ymin;
203  x[2] = gRandom->Uniform(xdiff)+xmin;
204  y[2] = gRandom->Uniform(ydiff)+ymin;
205  TPolyLine* pl = new TPolyLine(4, x, y);
206  pl->SetLineColor(Color_t(gRandom->Uniform(10)));
207  pl->Draw();
208  fCanvas->Modified();
209  fCanvas->Update();
210  break;
211  }
212  case 3: { // Arrow
213  TArrow* arrow = new TArrow(gRandom->Uniform(xdiff)+xmin,
214  gRandom->Uniform(ydiff)+ymin,
215  gRandom->Uniform(xdiff)+xmin,
216  gRandom->Uniform(ydiff)+ymin);
217  arrow->SetLineColor(Color_t(gRandom->Uniform(10)));
218  arrow->Draw();
219  fCanvas->Modified();
220  fCanvas->Update();
221  break;
222  }
223  case 4: // Zoom out
224  SetRange(xmin-xdiff*.1, ymin-ydiff*.1, xmax+xdiff*.1, ymax+ydiff*.1);
225  break;
226  case 5: // Zoom in
227  SetRange(xmin+xdiff*.1, ymin+ydiff*.1, xmax-xdiff*.1, ymax-ydiff*.1);
228  break;
229  case 6:
230  gApplication->Terminate(0);
231  break;
232  case 11: // nw
233  SetRange(xmin-xdiff*.1, ymin+ydiff*.1, xmax-xdiff*.1, ymax+ydiff*.1);
234  break;
235  case 12: // w
236  SetRange(xmin-xdiff*.1, ymin, xmax-xdiff*.1, ymax);
237  break;
238  case 13: // sw
239  SetRange(xmin-xdiff*.1, ymin-ydiff*.1, xmax-xdiff*.1, ymax-ydiff*.1);
240  break;
241  case 23: // s
242  SetRange(xmin, ymin-ydiff*.1, xmax, ymax-ydiff*.1);
243  break;
244  case 33: // se
245  SetRange(xmin+xdiff*.1, ymin-ydiff*.1, xmax+xdiff*.1, ymax-ydiff*.1);
246  break;
247  case 32: // e
248  SetRange(xmin+xdiff*.1, ymin, xmax+xdiff*.1, ymax);
249  break;
250  case 31: // ne
251  SetRange(xmin+xdiff*.1, ymin+ydiff*.1, xmax+xdiff*.1, ymax+ydiff*.1);
252  break;
253  case 21: // n
254  SetRange(xmin, ymin+ydiff*.1, xmax, ymax+ydiff*.1);
255  break;
256  case 22: // c
257  SetRange(50.0-xdiff/2,50-ydiff/2,50+xdiff/2,50+ydiff/2);
258  break;
259  } // end switch(id)
