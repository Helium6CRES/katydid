140  void KTHDF5Writer::WriteMetadata(std::string name, H5::DataType type, const void* value) {
141  std::string group_name = "/metadata";
142  std::stringstream group_name_builder;
143  group_name_builder << group_name << "/";
144  H5::Group grp = *(this->AddGroup(group_name));
145  std::string delimiter = "/";
146  size_t pos = 0;
147  std::string token;
148 
149  if (name.at(0) == '/') {
150  name.erase(0, 1);
151  }
152  while ((pos = name.find(delimiter)) != std::string::npos) {
153  token = name.substr(0, pos);
154  group_name_builder << token;
155  group_name_builder >> group_name;
156  grp = *(this->AddGroup(group_name));
157  name.erase(0, pos + delimiter.length());
158  KTINFO(group_name + delimiter + name);
159  }
160  H5::DataSpace dspace(H5S_SCALAR);
161  H5::DSetCreatPropList plist;
162  H5::DataSet* dset = new H5::DataSet(grp.createDataSet(name.c_str(),
163  type,
164  dspace,
165  plist));
166  dset->write(value, type);
167  }
