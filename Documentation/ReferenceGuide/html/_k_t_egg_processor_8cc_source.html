166  void KTEggProcessor::UnlimitedLoop(KTEggReader* reader)
167  {
168  unsigned iSlice = 0, iProgress = 0;
169  Nymph::KTDataPtr data, nextData;
170  bool nextSliceIsValid = true;
171  if (! HatchNextSlice(reader, data))
172  {
173  KTERROR(egglog, "Unable to hatch first slice of data.");
174  return;
175  }
176  while (true)
177  {
178  KTINFO(egglog, "Hatching slice " << iSlice);
179 
180  // Hatch the slice
181  if (! HatchNextSlice(reader, nextData))
182  {
183  data->Of< Nymph::KTData >().SetLastData(true);
184  nextSliceIsValid = false;
185  }
186 
187  if (data->Has< KTRawTimeSeriesData >())
188  {
189  KTDEBUG(egglog, "Raw time series data is present.");
190  fRawDataSignal(data);
191  NormalizeData(data);
192  }
193  if (data->Has< KTTimeSeriesData >())
194  {
195  KTDEBUG(egglog, "Normalized time series data is present.");
196  fDataSignal(data);
197  }
198  else
199  {
200  KTWARN(egglog, "No time-series data present in slice");
201  }
202 
203  ++iSlice;
204  ++iProgress;
205 
206  if (iProgress == fProgressReportInterval)
207  {
208  iProgress = 0;
209  KTPROG(egglog, iSlice << " slices processed");
210  }
211 
212  if (! nextSliceIsValid) break;
213  data = nextData;
214  }
215  return;
