76  {
77  // Create new Nymph::KTDataPtr and PSCollectionData
78  Nymph::KTDataPtr ptr( new Nymph::KTData() );
79  KTProcessedTrackData* newTrack = &ptr->Of< KTProcessedTrackData >();
80  KTPSCollectionData* newWaterfall = &ptr->Of< KTPSCollectionData >();
81 
82  // Configure the track to retain only the slope information
83  newTrack->SetSlope( trackData.GetSlope() );

87

89  newWaterfall->SetStartTime( trackData.GetStartTimeInRunC() - fLeadTime );

91  newWaterfall->SetFilling( false );

95

97  KTINFO(evlog, "Track length: " << trackData.GetEndTimeInRunC() - trackData.GetStartTimeInRunC());
98  KTINFO(evlog, "Track slope: " << trackData.GetSlope());
99 
100  return true;

103  bool KTSpectrogramCollector::ConsiderSpectrum( KTPowerSpectrum& ps, KTSliceHeader& slice, unsigned component, bool forceEmit )
104  {
105  // Iterate through each track which has been added
106  for( std::set< std::pair< Nymph::KTDataPtr, KTPSCollectionData* >, KTTrackCompare >::const_iterator it = fWaterfallSets[component].begin(); it != fWaterfallSets[component].end(); ++it )
107  {
108  // If the slice time coincides with the track time window, add the spectrum
109  // The forceEmit flag overrides this; essentially guarantees the spectrum will be interpreted as outside the track window
110  if( !forceEmit && slice.GetTimeInRun() >= it->second->GetStartTime() && slice.GetTimeInRun() <= it->second->GetEndTime() )
111  {
112  it->second->AddSpectrum( slice.GetTimeInRun(), &ps );
113  it->second->SetDeltaT( slice.GetSliceLength() );
114  it->second->SetFilling( true );
115  }
116  else
117  {
118  // If GetFilling() is true, we've reached the end of the track time window
119  // forceEmit=true sends all tracks to this clause, and those still filling will be closed & signals emitted
120  if( it->second->GetFilling() )
121  {
122  it->second->SetFilling( false );
123 
124  // Emit signal
125  KTINFO(evlog, "Finished a track; emitting signal");
126  FinishSC( it->first );
127  }
128  else
129  it->second->SetFilling( false );
130  }
131  }
132 
133  return true;
134  }

136  bool KTSpectrogramCollector::ReceiveTrack( KTProcessedTrackData& data )
137  {
138  if( data.GetIsCut() )
139  {
140  KTINFO(evlog, "Processed track failed cuts, skipping it");
141  return true;
142  }
143 
144  unsigned iComponent = data.GetComponent();
145 
146  // Increase size of fWaterfallSets if necessary
147  int fWSsize = fWaterfallSets.size();
148  std::set< std::pair< Nymph::KTDataPtr, KTPSCollectionData* >, KTTrackCompare > blankSet;
149  for( int i = fWSsize; i <= iComponent; i++ )
150  fWaterfallSets.push_back( blankSet );
151 
152  // Add track
153  if( !AddTrack( data, iComponent ) )
154  {
155  KTERROR(evlog, "Spectrogram collector could not add track! (component " << iComponent << ")" );
156  }
157 
158  return true;
159  }

161  bool KTSpectrogramCollector::ReceiveSpectrum( KTPowerSpectrumData& data, KTSliceHeader& sliceData, bool forceEmit )
162  {
163  KTDEBUG(evlog, "Receiving Spectrum");
164  if (fCalculateMinBin)
165  {
166  SetMinBin(data.GetSpectrum(0)->FindBin(fMinFrequency));
167  KTDEBUG(evlog, "Minimum bin set to " << fMinBin);
168  }
169  if (fCalculateMaxBin)
170  {
171  SetMaxBin(data.GetSpectrum(0)->FindBin(fMaxFrequency));
172  KTDEBUG(evlog, "Maximum bin set to " << fMaxBin);
173  }
174 
175  unsigned nComponents = data.GetNComponents();
176 
177  for (unsigned iComponent=0; iComponent<nComponents; ++iComponent)
178  {
179  if (! ConsiderSpectrum(*data.GetSpectrum(iComponent), sliceData, iComponent, forceEmit))
180  {
181  KTERROR(evlog, "Spectrogram collector could not receive spectrum! (component " << iComponent << ")");
182  return false;
183  }
184  }
185  KTINFO(evlog, "Spectrum finished processing. Awaiting next spectrum");
186 
187  return true;
188  }
