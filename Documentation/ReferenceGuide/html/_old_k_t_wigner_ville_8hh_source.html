218  inline Double_t KTWignerVille::GetMinFrequency(Double_t timeBinWidth) const
219  {
220  //return -0.5 * GetFrequencyBinWidth(timeBinWidth);
221 
222  // There's one bin at the center, always: the DC bin.
223  // # of bins on the negative side is nFreqBins/2 (rounded down because of integer division).
224  // 0.5 is added to the # of bins because of the half of the DC bin on the negative frequency side.
225  return -GetFrequencyBinWidth(timeBinWidth) * (Double_t(GetTimeSize()/2) + 0.5);
226 
227  }

229  inline Double_t KTWignerVille::GetMaxFrequency(Double_t timeBinWidth) const
230  {
231  //return GetFrequencyBinWidth(timeBinWidth) * ((Double_t)GetFrequencySize() - 0.5);
232 
233  // There's one bin at the center, always: the DC bin.
234  // # of bins on the positive side is nFreqBins/2 if the number of bins is odd, and nFreqBins/2-1 if the number of bins is even (division rounded down because of integer division).
235  // 0.5 is added to the # of bins because of the half of the DC bin on the positive frequency side.
236  UInt_t nBins = GetTimeSize();
237  UInt_t nBinsToSide = nBins / 2;
238  return GetFrequencyBinWidth(timeBinWidth) * (Double_t(nBinsToSide*2 == nBins ? nBinsToSide - 1 : nBinsToSide) + 0.5);
239 
240  }
