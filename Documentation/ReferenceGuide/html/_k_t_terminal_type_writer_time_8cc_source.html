101  void KTTerminalTypeWriterTime::WriteDigitizerTestData(Nymph::KTDataPtr data)

104 
105  uint64_t sliceNumber = data->Of<KTSliceHeader>().GetSliceNumber();
106 
107  KTDigitizerTestData& digData = data->Of< KTDigitizerTestData >();
108  unsigned nComponents = digData.GetNComponents();
109 
110  KTPROG(termlog, "Slice " << sliceNumber << " (" << nComponents << " components)\n\n");
111 
112  unsigned nBits = digData.GetNBits();
113 
114  // Bit Occupancy
115  if (digData.GetBitOccupancyFlag())
116  {
117  string prefixSpc(" | ");
118  string prefixOcc("Occupancy | ");
119  string prefixBit(" Bit | ");
120  unsigned prefixSize = prefixOcc.size();
121  unsigned maxHistRows = 10;
122  string betweenBins(" | ");
123  string filledBin(" # ");
124  string emptyBin(" ");
125  unsigned binWidth = filledBin.size();
126  unsigned binPadding = betweenBins.size();
127  unsigned bufferSize = (binWidth + binPadding) * (nBits + 1) + prefixSize;
128  char* buffer = new char[bufferSize];
129  stringstream toTermBitOcc;
130  toTermBitOcc << "Bit Occupancy Test\n";
131  for (unsigned iComponent=0; iComponent<nComponents; ++iComponent)
132  {
133  toTermBitOcc << "Component " << iComponent << '\n';
134  // histogram
135  unsigned maxValue = 0;
136  for (int bit = (int)nBits-1; bit >= 0; --bit)
137  {
138  maxValue = std::max(maxValue, digData.GetBitHistogram()->operator()(bit));
139  }
140  unsigned histRows = std::min(maxHistRows, maxValue);
141  unsigned occPerRow = (unsigned)KTMath::Nint((double)maxValue / (double)histRows);
142  for (int row = (int)histRows-1; row >= 0; --row)
143  {
144  sprintf(buffer, "%s", prefixSpc.c_str());
145  for (int bit = (int)nBits-1; bit >= 0; --bit)
146  {
147  int printOffset = (int)nBits - 1 - bit;
148  if (digData.GetBitHistogram()->operator()(bit) > (unsigned)row * occPerRow)
149  {
150  sprintf(buffer + prefixSize + printOffset*binWidth + printOffset*binPadding, "%s", filledBin.c_str());
151  }
152  else
153  {
154  sprintf(buffer + prefixSize + printOffset*binWidth + printOffset*binPadding, "%s", emptyBin.c_str());
155  }
156  if (bit != 0)
157  {
158  sprintf(buffer + prefixSize + (printOffset+1)*binWidth + printOffset*binPadding, "%s", betweenBins.c_str());
159  }
160  }
161  toTermBitOcc << buffer << '\n';
162  }
163  // bit occupancy line
164  sprintf(buffer, "%s", prefixOcc.c_str());
165  for (int bit = (int)nBits-1; bit >= 0; --bit)
166  {
167  int printOffset = (int)nBits - 1 - bit;
168  sprintf(buffer + prefixSize + printOffset*binWidth + printOffset*binPadding, "%*u", binWidth, digData.GetBitHistogram()->operator()(bit));
169  if (bit != 0)
170  {
171  sprintf(buffer + prefixSize + (printOffset+1)*binWidth + printOffset*binPadding, "%s", betweenBins.c_str());
172  }
173  }
174  toTermBitOcc << buffer << '\n';
175 
176  // bit number line
177  sprintf(buffer, "%s", prefixBit.c_str());
178  for (int bit = (int)nBits-1; bit >= 0; --bit)
179  {
180  int printOffset = (int)nBits - 1 - bit;
181  sprintf(buffer + prefixSize + printOffset*binWidth + printOffset*binPadding, "%*u", binWidth, bit);
182  if (bit != 0)
183  {
184  sprintf(buffer + prefixSize + (printOffset+1)*binWidth + printOffset*binPadding, "%s", betweenBins.c_str());
185  }
186  }
187  toTermBitOcc << buffer;
188  }
189  delete [] buffer;
190  KTPROG(termlog, toTermBitOcc.str() << '\n');
191  }
192 
193  // Clipping
194  if (digData.GetClippingFlag())
195  {
196  stringstream toTermClipping;
197  toTermClipping << "Clipping Test\n";
198  for (unsigned iComponent=0; iComponent<nComponents; ++iComponent)
199  {
200  toTermClipping << "Component " << iComponent << '\n';
201  toTermClipping << "\t\tNumber\tFraction\n";
202  toTermClipping << "\tTop \t" << digData.GetNClipTop(iComponent) << '\t' << digData.GetTopClipFrac(iComponent) << '\n';
203  toTermClipping << "\tBottom\t" << digData.GetNClipBottom(iComponent) << '\t' << digData.GetBottomClipFrac(iComponent) << '\n';
204  toTermClipping << "\tSequential Top \t" << digData.GetNMultClipTop(iComponent) << '\t' << digData.GetMultTopClipFrac(iComponent) << '\n';
205  toTermClipping << "\tSequential Bottom\t" << digData.GetNMultClipBottom(iComponent) << '\t' << digData.GetMultBottomClipFrac(iComponent) << '\n';
206  }
207  KTPROG(termlog, toTermClipping.str());
208  }
209  //Linearity
210  if (digData.GetLinearityFlag())
211  {
212  stringstream toTermLinearity;
213  toTermLinearity << "Linearity Test\n";
214  for (unsigned iComponent=0; iComponent<nComponents; ++iComponent)
215  {
216  toTermLinearity << "Component " << iComponent << '\n';
217  toTermLinearity << "Upslope: "<<'\n';

219  toTermLinearity << '\t' << "Average LinReg Slope: "<<digData.GetAvgLinRegSlope(iComponent)<<'\n'<<'\n';
220  toTermLinearity << "Downslope: "<<'\n';

222  toTermLinearity << '\t' << "Average LinReg Slope: "<<digData.GetAvgLinRegSlopeD(iComponent)<<'\n';
