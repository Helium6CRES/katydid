75  KTPhysicalArray< 2, double >* KTHoughTransform::TransformPoints(const SWFPoints& points, double minTime, double timeLength, double minFreq, double freqWidth)
76  {
77  KTINFO(htlog, "Number of time/frequency points: " << points.size());
78 
79  double maxR = KTMath::Sqrt2();
80  double timeScaling = 1. / timeLength;
81  double freqScaling = 1. / freqWidth;
82 
83  KTPhysicalArray< 2, double >* newTransform = new KTPhysicalArray< 2, double >(fNThetaPoints, 0., KTMath::Pi(), fNRPoints, -maxR, maxR);
84  for (unsigned iTheta = 0; iTheta < fNThetaPoints; ++iTheta)
85  {
86  for (unsigned iRadius = 0; iRadius < fNRPoints; ++iRadius)
87  {
88  (*newTransform)(iTheta, iRadius) = 0.;
89  }
90  }
91 
92  double deltaTheta = KTMath::Pi() / (double)fNThetaPoints;
93 
94  // initial loop over theta bins to create the KTPhysicalArrays and to cache cosTheta and sinTheta values
95  if (fCosTheta.size() != fNThetaPoints || fSinTheta.size() != fNThetaPoints)
96  {
97  fCosTheta.resize(fNThetaPoints);
98  fSinTheta.resize(fNThetaPoints);
99  double theta = newTransform->GetBinCenter(0, 0);
100  for (unsigned iTheta = 0; iTheta < fNThetaPoints; ++iTheta)
101  {
102  fCosTheta[iTheta] = cos(theta);
103  fSinTheta[iTheta] = sin(theta);
104  theta += deltaTheta;
105  }
106  }
107 
108  double timeVal, freqVal, value, radius;
109  unsigned iRadius;
110  for (SWFPoints::const_iterator pIt = points.begin(); pIt != points.end(); ++pIt)
111  {
112  timeVal = (pIt->fTimeInRunC - minTime) * timeScaling;
113  freqVal = (pIt->fFrequency - minFreq) * freqScaling;
114  value = pIt->fAmplitude;
115 
116  for (unsigned iTheta = 0; iTheta < fNThetaPoints; ++iTheta)
117  {
118  radius = timeVal * fCosTheta[iTheta] + freqVal * fSinTheta[iTheta];
119 
120  iRadius = newTransform->FindBin(2, radius);
121 
122  (*newTransform)(iTheta, iRadius) = (*newTransform)(iTheta, iRadius) + value;
123  }
124  }
125 
126  return newTransform;
127  }
