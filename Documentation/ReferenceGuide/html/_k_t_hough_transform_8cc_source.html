106  KTPhysicalArray< 1, KTPhysicalArray< 1, Double_t >* >* KTHoughTransform::TransformSpectrum(const KTPhysicalArray< 1, KTFrequencySpectrumPolar* >* powerSpectrum)
107  {
108  UInt_t nTimeBins = powerSpectrum->size();
109  UInt_t nFreqBins = (*powerSpectrum)(0)->size();
110 
111  KTINFO(htlog, "time info: " << nTimeBins << " " << powerSpectrum->GetRangeMin() << " " << powerSpectrum->GetRangeMax() << " " << powerSpectrum->GetBinWidth());
112  KTINFO(htlog, "freq info: " << nFreqBins << " " << (*powerSpectrum)(0)->GetRangeMin() << " " << (*powerSpectrum)(0)->GetRangeMax() << " " << (*powerSpectrum)(0)->GetBinWidth());
113 
114  Double_t maxR = sqrt(Double_t(nTimeBins*nTimeBins + nFreqBins*nFreqBins));
115 
116  KTPhysicalArray< 1, KTPhysicalArray< 1, Double_t >* >* newTransform = new KTPhysicalArray< 1, KTPhysicalArray< 1, Double_t >* >(fNThetaPoints, 0., KTMath::Pi());
117 
118  Double_t deltaTheta = KTMath::Pi() / (Double_t)fNThetaPoints;
119 
120  // initial loop over theta bins to create the KTPhysicalArrays and to cache cosTheta and sinTheta values
121  vector< Double_t > cosTheta(fNThetaPoints);
122  vector< Double_t > sinTheta(fNThetaPoints);
123  Double_t theta = newTransform->GetBinCenter(0);
124  for (UInt_t iTheta = 0; iTheta < fNThetaPoints; iTheta++)
125  {
126  (*newTransform)(iTheta) = new KTPhysicalArray< 1, Double_t >(fNRPoints, -maxR, maxR);
127  cosTheta[iTheta] = cos(theta);
128  sinTheta[iTheta] = sin(theta);
129  theta += deltaTheta;
130  }
131 
132  Double_t timeVal, freqVal, value, radius;
133  UInt_t iRadius;
134  for (UInt_t iTime = 0; iTime < nTimeBins; iTime++)
135  {
136  timeVal = Double_t(iTime);
137 
138  for (UInt_t iFreq = 0; iFreq < nFreqBins; iFreq++)
139  {
140  value = (*(*powerSpectrum)(iTime))(iFreq).abs();
141  if (value < 1.e-4) continue; // HARD CODED THRESHOLD
142 
143  freqVal = Double_t(iFreq);
144 
145  for (UInt_t iTheta = 0; iTheta < fNThetaPoints; iTheta++)
146  {
147  radius = timeVal * cosTheta[iTheta] + freqVal * sinTheta[iTheta];
148 
149  iRadius = (*newTransform)(iTheta)->FindBin(radius);
150 
151  (*(*newTransform)(iTheta))(iRadius) = (*(*newTransform)(iTheta))(iRadius) + value;
152  }
153  }
154  }
155 
156  return newTransform;
157  }
