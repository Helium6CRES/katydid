94  void KTHDF5TypeWriterTransform::PrepareHDF5File() {
95 
96  if( !fWriter->OpenAndVerifyFile() ) return;
97 
98  /*
99  A Buffer is the memory array where the data is copied to right before being written to file
100  A Dataspace is what gets written to the HDF5 file.
101  Buffers and Dataspaces will have 4 dimensions:
102  1 - Number of channels (sometimes called fNComponents in other modules)
103  2 - Number of slices
104  3 - Number of Parts (real-only: 1; real and imaginary: 2; polar amplitude and angle: 2)
105  4 - Number of samples in slice
106  */
107 
108  /*
109  * Defining Buffer and Dataset Size
110  *
111  * There are 2 choices to make: Polar vs Complex ; FFT Amplitude vs Power
112  *
113  * The choices are made in the functions called by the slot
114  * i.e.: WriteFrequencySpectrumDataFFTW, WriteFrequencySpectrumDataPolar, etc...
115  * There are two variables that need to be defined in theses functions before calling PrepareHDF5File:
116  * (1) fFFTSize and (2) fNParts
117  *
118  * (1) FFTSize: Number of Samples in the FFT in each Slice
119  * Polar FFT -> Each row is fSliceSize/2 + 1 long.
120  * Complex FFT -> Each row is fSliceSize long.
121  * PS and PSD spectrum -> Power is only calculated for positive frequencies and DC. The buffer is therefore the same shape as the polar power spectrum.
122  *
123  * (2) fNParts: number of parts in each sample: Complex FFT has 2 parts (real and imag), Polar FFT has 2 parts (magnitude and angle), power has 1 part (real)
124  *
125  */
126 
127 
129  // Create Buffers
130  this->fFFTBuffer = new fft_buffer(boost::extents[fNChannels][1][this->fNParts][this->fFFTSize]);
131  // Create arrays to store the frequency bins used in the FFT. The arrays are the same size as the Polar and FFTW spectra
132  this->fFFTFreqArrayBuffer = new freq_buffer(boost::extents[this->fFFTSize]);
133 
134 
136  // Create Data Space
137 
138  KTDEBUG(publog, "Creating HDF5 Dataspace...");
139 
140  if(this->fFFTDataSpace == NULL) {
141  this->ds_dims[0] = this->fNChannels;
142  this->ds_dims[1] = this->fNumberOfSlices;
143  this->ds_dims[2] = this->fNParts;
144  this->ds_dims[3] = this->fFFTSize;
145  this->ds_maxdims[0] = this->fNChannels;
146  this->ds_maxdims[1] = H5S_UNLIMITED;
147  this->ds_maxdims[2] = this->fNParts;
148  this->ds_maxdims[3] = this->fFFTSize;
149  this->fFFTDataSpace = new H5::DataSpace(4, this->ds_dims,this->ds_maxdims);
150  }
151  if(this->fFFTFreqArrayDataSpace == NULL) {
152  hsize_t freqarray_dims[] = {this->fFFTSize};
153  this->fFFTFreqArrayDataSpace = new H5::DataSpace(1, freqarray_dims);
154  }
155 
156 
158  // Create Dataset
159 
160  KTDEBUG(publog, "Creating Spectrum Dataset in HDF5 file...");
161  std::string FreqArray_name ("FreqArray");
162  if(this->fFFTGroup == NULL) this->fFFTGroup = new H5::Group(this->fSpectraGroup->createGroup("/spectra/spectrum"));
163  if(this->fDSet_FreqArray == NULL) this->fDSet_FreqArray = this->CreateDSet(FreqArray_name, this->fFFTGroup, *(this->fFFTFreqArrayDataSpace));
164  if(this->fDSet == NULL) this->fDSet = this->CreateDSet(this->fSpectrumName, this->fFFTGroup, *(this->fFFTDataSpace));
165 
166 
168  // Resize Dataset
169 
170  // If sliceN is greater than the number of Slices per Record, then extend the dataset
171  // This probably happens because there is more than 1 record in the Egg file being processed
172  if ( this->fSliceNumber >= this->ds_dims[1] ) {
173  this->ds_dims[1] = this->ds_dims[1] + this->fNumberOfSlices;
174  this->fDSet->extend( this->ds_dims );
175  }
176 
177  }

179  H5::DataSet* KTHDF5TypeWriterTransform::CreateDSet(const std::string& name,
180  const H5::Group* grp,
181  const H5::DataSpace& ds) {
182  herr_t hstatus;
183  H5::DSetCreatPropList plist;
184  unsigned default_value = 0;
185  plist.setFillValue(H5::PredType::NATIVE_DOUBLE, &default_value);
186  int rank = ds.getSimpleExtentNdims();
187  if (rank==4) {
188  // Uses 10 slices per chunk; we want a high number to improve compression, but not too high so that it's still fast
189  hsize_t ChunkSize[4] = {this->ds_dims[0],10,this->ds_dims[2],this->ds_dims[3]};
190  plist.setChunk(4, ChunkSize);
191  }
192  if (rank==1) {
193  hsize_t ChunkSize[1] = {this->fFFTSize};
194  plist.setChunk(1, ChunkSize);
195  }
196  if ( fUseCompressionFlag ) {
197  plist.setDeflate(6);
198  KTDEBUG(publog, "Creating compressed HDF5 dataset.");
199  }
200  KTDEBUG(publog, "Creating HDF5 dataset " << name);
201  KTDEBUG(publog, grp);
202  H5::DataSet* dset = new H5::DataSet(grp->createDataSet(name.c_str(),
203  H5::PredType::NATIVE_DOUBLE,
204  ds,
205  plist));
206  KTDEBUG("Done.");
207  return dset;
208  }

211  void KTHDF5TypeWriterTransform::WriteFrequencySpectrumDataFFTW(Nymph::KTDataPtr data) {
212  if (!data) return;
213 
214  // Get Data and Slice Number
215  KTFrequencySpectrumDataFFTW& fsData = data->Of<KTFrequencySpectrumDataFFTW>();
216  this->fSliceNumber = data->Of<KTSliceHeader>().GetSliceNumber();
217  KTDEBUG(publog, "Writing Slice " << this->fSliceNumber);
218 
219  // Prepare HDF5 for writing
220  if ( fWriter->DidParseHeader() ) {
221  this->ProcessEggHeader();
222  this->fFFTSize = this->fSliceSize;
223  this->fNParts = 2;
224  this->fSpectrumName = "complexFS";
225  this->PrepareHDF5File();
226  }
227  else return;
228 
229  // Select hyperslabs to save the spectrum data
230  hsize_t offset[4] = {0,this->fSliceNumber,0,0};
231  hsize_t dims1[4] = { this->ds_dims[0],1,this->ds_dims[2],this->ds_dims[3]}; /* data slice dimensions */
232  H5::DataSpace memspace1 ( 4, dims1 );
233  H5::DataSpace filespace1 = this->fDSet->getSpace();
234  filespace1.selectHyperslab( H5S_SELECT_SET, dims1, offset );
235 
236  // Copy data to Buffer
237  // For now we need to do this because of the way we want to save the data.
238  // The spec object has the data organized in [Samples][Parts].
239  // The dataset is [...][Parts][Samples].
240  KTDEBUG(publog, "Copying Spectrum Dataset (Complex FFT) to Buffer.");
241  unsigned nChannels = fsData.GetNComponents();
242  for (unsigned iC = 0; iC < nChannels; iC++) {
243  const KTFrequencySpectrumFFTW* spec = fsData.GetSpectrumFFTW(iC);
244  if (spec != NULL) {
245  for(int f=0; f < spec[0].size(); f++) {
246  (*this->fFFTBuffer)[iC][0][0][f] = spec[0].GetReal(f);
247  (*this->fFFTBuffer)[iC][0][1][f] = spec[0].GetImag(f);
248  if ((!fFirstSliceHasBeenWritten) & (iC==0)) (*this->fFFTFreqArrayBuffer)[f] = spec[0].GetBinCenter(f);
249  }
250  }
251  }
252 
253  // Write Buffer to Dataset
254  KTDEBUG(publog, "Writing Spectrum Dataset (Complex FFT) to HDF5 file.");
255  if (!fFirstSliceHasBeenWritten) {
256  this->fDSet_FreqArray->write(this->fFFTFreqArrayBuffer->data(), H5::PredType::NATIVE_DOUBLE);
257  SetFirstSliceHasBeenWritten(true);
258  }
259  this->fDSet->write(this->fFFTBuffer->data(), H5::PredType::NATIVE_DOUBLE, memspace1,filespace1);
260  KTDEBUG(publog, "Done Writing Slice to File.");
261  }

492  void KTHDF5TypeWriterTransform::WritePowerSpectralDensity(Nymph::KTDataPtr data) {
493  if (!data) return;
494 
495  // Get Data and Slice Number
496  KTPowerSpectrumData& fsData = data->Of<KTPowerSpectrumData>();
497  this->fSliceNumber = data->Of<KTSliceHeader>().GetSliceNumber();
498  KTDEBUG(publog, "Writing Slice " << this->fSliceNumber);
499 
500  // Prepare HDF5 for writing
501  if ( fWriter->DidParseHeader() ) {
502  this->ProcessEggHeader();
503  this->fFFTSize = (this->fSliceSize >> 1) + 1;
504  this->fNParts = 1;
505  this->fSpectrumName = "PSD";
506  this->PrepareHDF5File();
507  }
508  else return;
509 
510  // Select hyperslabs to save the spectrum data
511  hsize_t offset[4] = {0,this->fSliceNumber,0,0};
512  hsize_t dims1[4] = { this->ds_dims[0],1,this->ds_dims[2],this->ds_dims[3]}; /* data slice dimensions */
513  H5::DataSpace memspace1 ( 4, dims1 );
514  H5::DataSpace filespace1 = this->fDSet->getSpace();
515  filespace1.selectHyperslab( H5S_SELECT_SET, dims1, offset );
516 
517  // Copy data to Buffer
518  // For now we need to do this because of the way we want to save the data.
519  // The spec object has the data organized in [Samples][Parts].
520  // The dataset is [...][Parts][Samples].
521  KTDEBUG(publog, "Copying Spectrum Dataset (PSD) to Buffer.");
522  unsigned nChannels = fsData.GetNComponents();
523  for (unsigned iC = 0; iC < nChannels; iC++) {
524  KTPowerSpectrum* spec = fsData.GetSpectrum(iC);
525  if (spec != NULL) {
526  spec->ConvertToPowerSpectralDensity();
527  for(int f=0; f < spec[0].size(); f++) {
528  (*this->fFFTBuffer)[iC][0][0][f] = (*spec)(f);
529  if ((!fFirstSliceHasBeenWritten) & (iC==0)) (*this->fFFTFreqArrayBuffer)[f] = spec[0].GetBinCenter(f);
530  }
531  }
532  }
533 
534  // Write Buffer to Dataset
535  KTDEBUG(publog, "Writing Spectrum Dataset (PSD) to HDF5 file.");
536  if (!fFirstSliceHasBeenWritten) {
537  this->fDSet_FreqArray->write(this->fFFTFreqArrayBuffer->data(), H5::PredType::NATIVE_DOUBLE);
538  SetFirstSliceHasBeenWritten(true);
539  }
540  this->fDSet->write(this->fFFTBuffer->data(), H5::PredType::NATIVE_DOUBLE, memspace1,filespace1);
541  KTDEBUG(publog, "Done Writing Slice to File.");
542  }
