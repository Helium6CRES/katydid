83  void KTHDF5TypeWriterTime::CreateDataspaces() {
84  /*
85  If the dataspaces have already been created, this is a no-op.
86  Otherwise, we want to create two dataspaces - 1XM and 1XN, where
87  M is the size of a raw time slice, and N is the size of a time slice.
88  */
89  if(this->fRawTSliceDSpace == NULL
90  || this->fRealTSliceDSpace == NULL) {
91  hsize_t raw_dims[] = {this->fNComponents, this->fRawSliceSize};
92  hsize_t dims[] = {this->fNComponents, this->fSliceSize};
93 
94  this->fRawTSliceDSpace = new H5::DataSpace(2, raw_dims);
95  this->fRealTSliceDSpace = new H5::DataSpace(2, dims);
96  }
97  }

147
