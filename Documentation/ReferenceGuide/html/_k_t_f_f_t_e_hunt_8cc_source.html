129  void KTFFTEHunt::ProcessHeader(const KTEggHeader* header)
130  {
131  // Process the header information
132  fSimpleFFT.ProcessHeader(header);
133  fWindowFFT.ProcessHeader(header);
134 
135  // Set up the bin cuts
136  // get the number of frequency bins and the min and max frequencies
137  Int_t nFreqBins = fWindowFFT.GetFrequencySize();
138  //Double_t freqBinWidth = fWindowFFT.GetFreqBinWidth();
139  Double_t freqMin = fWindowFFT.GetFreqMin();
140  Double_t freqMax = fWindowFFT.GetFreqMax();
141  // create KTFrequencySpectrum w/ number of bins; set bin width
142  // binFinder will go out of scope at the end of this function.
143  // therefore, from that point until fClusteringProc sets a new array, binCuts should NOT be used to access array values!
144  KTFrequencySpectrum binFinder(nFreqBins, freqMin, freqMax);
145  // create KTMaskedArray based on power spectrum magnitude array
146  KTMaskedArray< KTFrequencySpectrum::array_type, complexpolar<Double_t> >* binCuts = new KTMaskedArray< KTFrequencySpectrum::array_type, complexpolar<Double_t> >(binFinder.data(), nFreqBins);
147  // convert cut frequency ranges to bins, and cut bins from the masked array
148  for (vector< CutRange >::const_iterator itCutRange = fCutRanges.begin(); itCutRange != fCutRanges.end(); itCutRange++)
149  {
150  Int_t firstBinCut = binFinder.FindBin((*itCutRange).first / fFrequencyMultiplier);
151  Int_t nCutBins = binFinder.FindBin((*itCutRange).second / fFrequencyMultiplier) - firstBinCut + 1;
152  binCuts->Cut(UInt_t(firstBinCut), UInt_t(nCutBins));
153  }
154  // give the masked array to fClusteringProc
155  fClustering.SetBinCuts(binCuts);
156 
157  EmptyEventPeakBins();
158  fClustering.SetEventPeakBinsList(&fEventPeakBins);
159 
160  if (fWriteTextFileFlag)
161  {
162  fTextFile.open(fTextFilename.c_str(), std::ios::out | std::ios::app);
163  if (! fTextFile.is_open()) fWriteTextFileFlag = kFALSE;
164  }
165 #ifdef ROOT_FOUND
166  if (fWriteROOTFileFlag)
167  {
168  fROOTFile.Open(fROOTFilename.c_str(), "UPDATE");
169  if (! fROOTFile.IsOpen()) fWriteROOTFileFlag = kFALSE;
170  }
171 #endif
172 
173  fTotalCandidates = 0;
174 
175  return;
176  }
