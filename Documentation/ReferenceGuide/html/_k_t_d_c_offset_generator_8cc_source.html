34  bool KTDCOffsetGenerator::ConfigureDerivedGenerator(const scarab::param_node* node)
35  {
36  if (node == NULL) return false;
37 
38  const KTParamArray* offsetPairs = node->ArrayAt("offsets");
39  if (offsetPairs != NULL)

41  for (KTParamArray::const_iterator pairIt = offsetPairs->Begin(); pairIt != offsetPairs->End(); ++pairIt)

43  if (! ((*pairIt)->IsArray() && (*pairIt)->AsArray().Size() == 2))

49  if (fOffsets.size() <= pair.first) fOffsets.resize(pair.first + 1);

51  }

57  bool KTDCOffsetGenerator::GenerateTS(KTTimeSeriesData& data)
58  {
59  const unsigned sliceSize = data.GetTimeSeries(0)->GetNTimeBins();
60 
61  unsigned nComponents = data.GetNComponents();
62  if (fOffsets.size() <= nComponents) fOffsets.resize(nComponents + 1);
63 
64  for (unsigned iComponent = 0; iComponent < nComponents; ++iComponent)
65  {
66  KTTimeSeries* timeSeries = data.GetTimeSeries(iComponent);
