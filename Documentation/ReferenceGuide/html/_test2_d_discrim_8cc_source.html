38 // Method to determine whether a line intersects a bin with finite size
39 // The line extends over a finite segment from (x1,y1) to (x2,y2)
40 // The bin has lower-left corner (xx1,yy1) and upper-right corner (xx2,yy2)
41 // It follows that the lower-right corner is (xx2,yy1) and the upper-left corner is (xx1,yy2)
42 bool lineIntersects( double x1, double y1, double x2, double y2, double xx1, double yy1, double xx2, double yy2 )
43 {
44  // If the x or y coordinates of the line are entirely outside the bin range, we can immediately return false
45  if( x2 <= xx1 || x1 >= xx2 )
46  return false;
47  if( y2 <= yy1 || y1 >= yy2 )
48  return false;
49 
50  double m = (y2 - y1) / (x2 - x1); // slope
51  double yLeft = y1 + m * (xx1 - x1); // y-position of the line at xx1
52  double yRight = y1 + m * (xx2 - x1); // y-position of the line at xx2
53 
54  if( m > 0 )
55  {
56  // Positive slope
57  // We check the upper-left corner and the lower-right corner
58  if( yLeft >= yy2 || yRight <= yy1 )
59  return false;
60  else
61  return true;
62  }
63  else
64  {
65  // Negative slope
66  // We check the lower-left corner and the upper-right corner
67  if( yLeft <= yy1 || yRight >= yy2 )
68  return false;
69  else
70  return true;
71  }
72 }
