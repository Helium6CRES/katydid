108  bool KTGainVarChi2Test::CalculateSpectrum(const KTPowerSpectrum* spectrum, const KTPowerSpectrum* sigma, const KTSpline* spline, KTGainVarChi2Data& newData, unsigned component)
109  {
110  if (spectrum == NULL)
111  {
112  KTERROR(sdlog, "Frequency spectrum pointer (component " << component << ") is NULL!");
113  return false;
114  }
115 
116  unsigned nBins = fMaxBin - fMinBin + 1;
117  double binWidth = spectrum->GetBinWidth();
118  double freqMin = spectrum->GetBinLowEdge(fMinBin);
119  double freqMax = spectrum->GetBinLowEdge(fMaxBin) + spectrum->GetBinWidth();
120  KTSpline::Implementation* splineImp = spline->Implement(nBins, freqMin, freqMax);
121 
122  double chi2 = 0., dev = 0.;
123  uint64_t ndf = 0;
124  for (unsigned iBin=fMinBin; iBin<=fMaxBin; ++iBin)
125  {
126  dev = ((*spectrum)(iBin) - (*splineImp)(iBin - fMinBin))/(*sigma)(iBin);
127  chi2 += dev * dev;
128  ndf += 1;
129  }
130 
131  newData.SetChi2( chi2 );
132  newData.SetNDF( ndf );
133 
134  spline->AddToCache(splineImp);
