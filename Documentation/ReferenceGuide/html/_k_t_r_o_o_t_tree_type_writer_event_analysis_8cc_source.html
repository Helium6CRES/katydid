79  {
80  KTFrequencyCandidateData& fcData = data->Of< KTFrequencyCandidateData >();
81  KTSliceHeader& header = data->Of< KTSliceHeader >();
82 
83  if (! fWriter->OpenAndVerifyFile()) return;
84 
85  if (fFreqCandidateTree == NULL)
86  {
87  if (! SetupFrequencyCandidateTree())
88  {
89  KTERROR(publog, "Something went wrong while setting up the frequency candidate tree! Nothing was written.");

91  }
92  }
93 
94  // Load() also clears any existing data
95  //fFreqCandidateData->Load(*data);
96  fFreqCandidateData.fSlice = header.GetSliceNumber();
97  fFreqCandidateData.fTimeInRun = header.GetTimeInRun();
98  for (fFreqCandidateData.fComponent = 0; fFreqCandidateData.fComponent < fcData.GetNComponents(); fFreqCandidateData.fComponent++)
99  {
100  fFreqCandidateData.fThreshold = fcData.GetThreshold(fFreqCandidateData.fComponent);
101  const KTFrequencyCandidateData::Candidates& candidates = fcData.GetCandidates(fFreqCandidateData.fComponent);
102  for (KTFrequencyCandidateData::Candidates::const_iterator it = candidates.begin(); it != candidates.end(); it++)
103  {
104  fFreqCandidateData.fFirstBin = it->GetFirstBin();
105  fFreqCandidateData.fLastBin = it->GetLastBin();
106  fFreqCandidateData.fMeanFrequency = it->GetMeanFrequency();
107  fFreqCandidateData.fPeakAmplitude = it->GetPeakAmplitude();
108  fFreqCandidateData.fAmplitudeSum = it->GetAmplitudeSum();
109 
110  fFreqCandidateTree->Fill();
111  }
112  }
