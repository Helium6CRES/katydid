375  bool param_array::has_subset( const param& a_subset ) const
376  {
377  if( ! a_subset.is_array() ) return false;

379  if( t_subset_array.size() > f_contents.size() ) return false;
380  const_iterator t_this_it = f_contents.begin();
381  const_iterator t_that_it = t_subset_array.begin();
382  while( t_that_it != t_subset_array.end() ) // loop condition is on a_subset because it's smaller or equal to this

469  void param_node::merge( const param_node& a_object )
470  {
471  //LDEBUG( dlog, "merging object with " << a_object.size() << " items:\n" << a_object );
472  for( const_iterator it = a_object.f_contents.begin(); it != a_object.f_contents.end(); ++it )
473  {
474  if( ! has( it->first ) )
475  {
476  //LDEBUG( dlog, "do not have object <" << it->first << "> = <" << *it->second << ">" );
477  add( it->first, *it->second );
478  continue;
479  }
480  param& t_param = (*this)[ it->first ];
481  if( t_param.is_value() )
482  {
483  //LDEBUG( dlog, "replacing the value of \"" << it->first << "\" <" << get_value( it->first ) << "> with <" << *it->second << ">" );
484  replace( it->first, *it->second );
485  continue;
486  }
487  if( t_param.is_node() && it->second->is_node() )
488  {
489  //LDEBUG( dlog, "merging nodes")
490  t_param.as_node().merge( it->second->as_node() );
491  continue;
492  }
493  if( t_param.is_array() && it->second->is_array() )
494  {
495  //LDEBUG( dlog, "appending array" );
496  t_param.as_array().append( it->second->as_array() );
497  continue;
498  }
499  //LDEBUG( dlog, "generic replace" );
500  this->replace( it->first, *it->second );
501  }
502  }
