174  void KTMultiTrackEventData::ProcessTracks()
175  {
176  KTDEBUG(evlog, "Processing tracks");
177 
178  TrackSetCIt trackIt = fTracks.begin();
179 
180  fStartTimeInAcq = trackIt->GetStartTimeInAcq();
181  fStartTimeInRunC = trackIt->GetStartTimeInRunC();
182  fEndTimeInRunC = trackIt->GetEndTimeInRunC();
183  fTimeLength = trackIt->GetTimeLength();
184  fStartFrequency = trackIt->GetStartFrequency();
185  fEndFrequency = trackIt->GetEndFrequency();
186  fMinimumFrequency = std::min(fStartFrequency, fEndFrequency);
187  fMaximumFrequency = std::max(fStartFrequency, fEndFrequency);
188  fFrequencyWidth = trackIt->GetFrequencyWidth();
189 
190  fStartTimeInRunCSigma = trackIt->GetStartTimeInRunCSigma();
191  fEndTimeInRunCSigma = trackIt->GetEndTimeInRunCSigma();
192  fTimeLengthSigma = trackIt->GetTimeLengthSigma();
193  fStartFrequencySigma = trackIt->GetStartFrequencySigma();
194  fEndFrequencySigma = trackIt->GetEndFrequencySigma();
195  fFrequencyWidthSigma = trackIt->GetFrequencyWidthSigma();
196 
197  fFirstTrackID = trackIt->GetTrackID();
198  fFirstTrackTimeLength = trackIt->GetTimeLength();
199  fFirstTrackFrequencyWidth = trackIt->GetFrequencyWidth();
200  fFirstTrackSlope = trackIt->GetSlope();
201  fFirstTrackIntercept = trackIt->GetIntercept();
202  fFirstTrackTotalPower = trackIt->GetTotalPower();
203 
204  for (++trackIt; trackIt != fTracks.end(); ++trackIt)
205  {
206  KTDEBUG(evlog, "Track " << trackIt->GetTrackID());
207 
208  if (trackIt->GetStartTimeInRunC() < fStartTimeInRunC)
209  {
210  fStartTimeInAcq = trackIt->GetStartTimeInAcq();
211  fStartTimeInRunC = trackIt->GetStartTimeInRunC();
212  fStartTimeInRunCSigma = trackIt->GetStartTimeInRunCSigma();
213  fStartFrequency = trackIt->GetStartFrequency();
214  fStartFrequencySigma = trackIt->GetStartFrequencySigma();
215  KTDEBUG(evlog, "Start time (freq) is now " << fStartTimeInRunC << "(" << fStartFrequency << ")");
216  fFirstTrackID = trackIt->GetTrackID();
217  fFirstTrackTimeLength = trackIt->GetTimeLength();
218  fFirstTrackFrequencyWidth = trackIt->GetFrequencyWidth();
219  fFirstTrackSlope = trackIt->GetSlope();
220  fFirstTrackIntercept = trackIt->GetIntercept();
221  fFirstTrackTotalPower = trackIt->GetTotalPower();
222  }
223 
224  if (trackIt->GetEndTimeInRunC() > fEndTimeInRunC)
225  {
226  fEndTimeInRunC = trackIt->GetEndTimeInRunC();
227  fEndTimeInRunCSigma = trackIt->GetEndTimeInRunCSigma();
228  fEndFrequency = trackIt->GetEndFrequency();
229  fEndFrequencySigma = trackIt->GetEndFrequencySigma();
230  KTDEBUG(evlog, "End time (freq) is now " << fEndTimeInRunC << "(" << fEndFrequency << ")");
231  }
232 
233  double minFreq = std::min(trackIt->GetStartFrequency(), trackIt->GetEndFrequency());
234  if (minFreq < fMinimumFrequency)
235  {
236  fMinimumFrequency = minFreq;
237  KTDEBUG(evlog, "Minimum frequency is now " << fMinimumFrequency);
238  }
239 
240  double maxFreq = std::max(trackIt->GetStartFrequency(), trackIt->GetEndFrequency());
241  if (maxFreq > fMaximumFrequency)
242  {
243  fMaximumFrequency = maxFreq;
244  KTDEBUG(evlog, "Maximum frequency is now " << fMaximumFrequency);
245  }
246  }
247 
248  fTimeLength = fEndTimeInRunC - fStartTimeInRunC;
249  fTimeLengthSigma = sqrt(fEndTimeInRunCSigma * fEndTimeInRunCSigma + fStartTimeInRunCSigma * fStartTimeInRunCSigma);
250 
251  fFrequencyWidth = fEndFrequency - fStartFrequency;
252  fFrequencyWidthSigma = sqrt(fEndFrequencySigma * fEndFrequencySigma + fStartFrequencySigma * fStartFrequencySigma);
253 
254  return;
255  }

295  bool MultiPeakTrackRef::InsertTrack(const TrackSetCIt& trackRef)
296  {
297  if (fTrackRefs.find(trackRef) != fTrackRefs.end()) return false;
298 
299  fTrackRefs.insert(trackRef);
300  fSumStartTimeInRunC += trackRef->GetStartTimeInRunC();
301  fSumEndTimeInRunC += trackRef->GetEndTimeInRunC();
302  double currentSize = (double)fTrackRefs.size();
303  fMeanStartTimeInRunC = fSumStartTimeInRunC / currentSize;
304  fMeanEndTimeInRunC = fSumEndTimeInRunC / currentSize;
305  return true;
