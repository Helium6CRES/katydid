75  return NULL;

78  fPrelude = newPrelude;

115  try

128  rapidxml::xml_node<char>* nodeHeader = headerDOM.first_node("header");
129  if (nodeHeader == NULL)

132  return NULL;
133  }
134

136  if (nodeDataFormat == NULL)

168  if (nodeDigitizer == NULL)

171  return NULL;

174  attr = nodeDigitizer->first_attribute("rate");
