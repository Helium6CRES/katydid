165  Nymph::KTDataPtr KTEgg2Reader::HatchNextSlice()
166  {
167  unsigned recordSize = fHeader.GetChannelHeader(0)->GetRecordSize();
168 
169  if (fMonarch == NULL)
170  {
171  KTERROR(eggreadlog, "Monarch file has not been opened");
172  return Nymph::KTDataPtr();
173  }
174  if (fReadState.fStatus == MonarchReadState::kInvalid)
175  {
176  KTERROR(eggreadlog, "Read state status is <invalid>. Did you hatch the egg first?");
177  return Nymph::KTDataPtr();
178  }
179 
180  if (fReadState.fStatus == MonarchReadState::kAtStartOfRun)
181  {
182  KTDEBUG(eggreadlog, "Reading first record");
183  // if we're at the beginning of the run, load the first records
184  if (! fMonarch->ReadRecord(fReadState.fAbsoluteRecordOffset))
185  {
186  KTERROR(eggreadlog, "File appears to contain no slices.");
187  return Nymph::KTDataPtr();
188  }
189  fSliceNumber = 0;
190  }
191  else
192  {
193  ++fSliceNumber;
194 
195  if (fReadState.fStatus == MonarchReadState::kReachedNextRecord)
196  {
197  KTDEBUG(eggreadlog, "Slice and record boundaries coincided; reading new record before creating new slice");
198  // if the slice boundary and record boundary coincide, we need to load the next record because the final position
199  // of the read pointer from the previous slice is at the sample after the end of the previous slice,
200  // which in this case would be the start of the next record.
201  if (! fMonarch->ReadRecord())
202  {
203  KTWARN(eggreadlog, "End of egg file reached after reading new records (or something else went wrong)");
204  return Nymph::KTDataPtr();
205  }
206  ++(fReadState.fAbsoluteRecordOffset);
207  fReadState.fReadPtrOffset = 0;
208  ++(fReadState.fReadPtrRecordOffset);
209  fReadState.fStatus = MonarchReadState::kContinueReading;
210  }
211 
212  // shift the slice start pointer by the stride
213  // note that this pointer refers to the record in which the previous slice started
214  fReadState.fSliceStartPtrOffset += fStride;
215  unsigned sliceStartRecordOffset = 0; // how many records to shift to the start of the slice
216  while (fReadState.fSliceStartPtrOffset >= recordSize)
217  {
218  fReadState.fSliceStartPtrOffset -= recordSize;
219  ++sliceStartRecordOffset;
220  }
221 
222  // Calculate whether we need to move the read pointer to a different record by subtracing the number
223  // of records read in the last slice (fReadPtrRecordOffset)
224  // If this is 0, it doesn't need to be moved
225  // If it's != 0, then it needs to be reduced by 1 because of how the offset number in Monarch::ReadRecord is used (offset=0 will advance to the next record; offset=-1 will read the same record)
226  int readPtrRecordOffsetShift = int(sliceStartRecordOffset) - int(fReadState.fReadPtrRecordOffset);
227  if (readPtrRecordOffsetShift != 0)
228  {
229  // change the absolute record offset first because it should be done before the adjustment to Monarch::ReadRecord offset counting is made
230  fReadState.fAbsoluteRecordOffset += readPtrRecordOffsetShift;
231  --readPtrRecordOffsetShift;
232  KTDEBUG(eggreadlog, "Reading new record with offset " << readPtrRecordOffsetShift);
233  // move the read pointer to the slice start pointer (first move monarch to the correct record)
234  if (! fMonarch->ReadRecord(readPtrRecordOffsetShift))
235  {
236  KTWARN(eggreadlog, "End of egg file reached after reading new records (or something else went wrong)");
237  return Nymph::KTDataPtr();
238  }
239  }
240  // Move the read pointer to the slice start pointer within the record
241  fReadState.fReadPtrRecordOffset = 0;
242  fReadState.fReadPtrOffset = fReadState.fSliceStartPtrOffset;
243  }
244 
245  Nymph::KTDataPtr newData(new Nymph::KTData());
246 
247  // Fill out slice header information
248  KTSliceHeader& sliceHeader = newData->Of< KTSliceHeader >().SetNComponents(fHeader.GetNChannels());
249  if (fReadState.fStatus == MonarchReadState::kAtStartOfRun)
250  {
251  sliceHeader.SetIsNewAcquisition(true);
252  }
253  else
254  {
255  sliceHeader.SetIsNewAcquisition(false);
256  }
257  sliceHeader.SetSampleRate(fHeader.GetAcquisitionRate());
258  sliceHeader.SetRawSliceSize(fSliceSize);
259  sliceHeader.SetSliceSize(fSliceSize);
260  sliceHeader.CalculateBinWidthAndSliceLength();
261  sliceHeader.SetNonOverlapFrac((double)fStride / (double)fSliceSize);
262  sliceHeader.SetTimeInRun(GetTimeInRun());
263  sliceHeader.SetTimeInAcq(GetTimeInAcq());
264  sliceHeader.SetSliceNumber(fSliceNumber);
265  sliceHeader.SetStartRecordNumber(fReadState.fAbsoluteRecordOffset);
266  sliceHeader.SetStartSampleNumber(fReadState.fReadPtrOffset);
267  sliceHeader.SetRecordSize(fHeader.GetChannelHeader(0)->GetRecordSize());
268  KTDEBUG(eggreadlog, sliceHeader << "\nNote: some fields may not be filled in correctly yet");
269 
270  // Setup pointers to monarch and new katydid records
271  unsigned nChannels = fHeader.GetNChannels();
272  vector< const M2RecordBytes* > monarchRecords(nChannels);
273  // the elements of monarchRecordData will need to be deleted
274  vector< const M2RecordDataInterface< uint64_t >* > monarchRecordData(nChannels);
275  vector< KTRawTimeSeries* > newRecords(nChannels);
276  for (unsigned iChannel = 0; iChannel < nChannels; ++iChannel)
277  {
278  monarchRecords[iChannel] = (fMonarch->*fMonarchGetRecord[iChannel])();
279  sliceHeader.SetAcquisitionID(monarchRecords[iChannel]->fAcquisitionId, iChannel);
280  sliceHeader.SetRecordID(monarchRecords[iChannel]->fRecordId, iChannel);
281  sliceHeader.SetTimeStamp(monarchRecords[iChannel]->fTime, iChannel);
282  monarchRecordData[iChannel] = new M2RecordDataInterface< uint64_t >(monarchRecords[iChannel]->fData, fHeader.GetChannelHeader(iChannel)->GetDataTypeSize());
283 
284  //tsData->SetTimeSeries(new vector< DataType >(monarchRecord->fDataPtr, monarchRecord->fDataPtr+header->GetSliceSize()), iChannel);
285  newRecords[iChannel] = new KTRawTimeSeries(fHeader.GetChannelHeader(iChannel)->GetDataTypeSize(), sDigitizedUS, fSliceSize, 0., double(fSliceSize) * sliceHeader.GetBinWidth());
286  }
287 
288  KTDEBUG(eggreadlog, "Time in run: " << GetTimeInRun() << " s\n" <<
289  "\tBin width = " << fBinWidth << '\n' <<
290  "\tMonarch record size = " << fRecordSize << '\n' <<
291  "\tRecord offset = " << fReadState.fAbsoluteRecordOffset << '\n' <<
292  "\tSlice start pointer offset = " << fReadState.fSliceStartPtrOffset << '\n' <<
293  "\tRead pointer record offset = " << fReadState.fReadPtrRecordOffset << '\n' <<
294  "\tRead pointer offset = " << fReadState.fReadPtrOffset);
295 
296  if (fReadState.fStatus == MonarchReadState::kAtStartOfRun)
297  {
298  fReadState.fStatus = MonarchReadState::kContinueReading;
299  }
300 
301  // Loop over bins
302  for (unsigned iBin = 0; iBin < fSliceSize; ++iBin)
303  {
304  if (fReadState.fStatus == MonarchReadState::kReachedNextRecord)
305  {
306  KTDEBUG(eggreadlog, "Reading new record mid-slice");
307  // try reading the next record
308  if (! fMonarch->ReadRecord())
309  {
310  // the end of the file has been reached or there was some other error preventing the reading of the next record
311  KTWARN(eggreadlog, "End of egg file reached after reading new records (or something else went wrong)");
312  for (unsigned iChannel = 0; iChannel < nChannels; ++iChannel)
313  {
314  delete newRecords[iChannel];
315  }
316  return Nymph::KTDataPtr();
317  }
318  ++(fReadState.fAbsoluteRecordOffset);
319  fReadState.fReadPtrOffset = 0;
320  ++(fReadState.fReadPtrRecordOffset);
321 
322  // check if the acquisition ID has changed on any channel
323  if (fReadState.fAcquisitionID != monarchRecords[0]->fAcquisitionId)
324  {
325  KTDEBUG(eggreadlog, "Acquisition ID change; resetting slice to start with this monarch record.\n"
326  << "\tNumber of unused bins: " << iBin - 1);
327  // this means that a new monarch record has come up, and it has a different acquisition id.
328  // in this situation we need to start the time series over with the new monarch record
329  // first, update the read state
330  fReadState.fAcquisitionID = monarchRecords[0]->fAcquisitionId;
331  fReadState.fReadPtrOffset = 0;
332  fReadState.fReadPtrRecordOffset = 0;
333  fReadState.fSliceStartPtrOffset = 0;
334  // reset slice data
335  sliceHeader.SetIsNewAcquisition(true);
336  sliceHeader.SetStartRecordNumber(fReadState.fAbsoluteRecordOffset);
337  sliceHeader.SetStartSampleNumber(fReadState.fReadPtrOffset);
338  for (unsigned iChannel = 0; iChannel < nChannels; ++iChannel)
339  {
340  sliceHeader.SetAcquisitionID(monarchRecords[iChannel]->fAcquisitionId, iChannel);
341  sliceHeader.SetRecordID(monarchRecords[iChannel]->fRecordId, iChannel);
342  sliceHeader.SetTimeStamp(monarchRecords[iChannel]->fTime, iChannel);
343  sliceHeader.SetRawDataFormatType(fHeader.GetChannelHeader(iChannel)->GetDataFormat(), iChannel);
344  }
345  // reset bin count to 0
346  iBin = 0;
347  // change the time in run since we're going back to the beginning of the record
348  sliceHeader.SetTimeInRun(GetTimeInRun());
349  KTDEBUG(eggreadlog, "Correction to time in run: " << GetTimeInRun() << " s\n" <<
350  "\tBin width = " << fBinWidth << '\n' <<
351  "\tMonarch record size = " << fRecordSize << '\n' <<
352  "\tRecord offset = " << fReadState.fAbsoluteRecordOffset << '\n' <<
353  "\tSlice start pointer offset = " << fReadState.fSliceStartPtrOffset << '\n' <<
354  "\tRead pointer record offset = " << fReadState.fReadPtrRecordOffset << '\n' <<
355  "\tRead pointer offset = " << fReadState.fReadPtrOffset);
356  sliceHeader.SetSliceNumber(fSliceNumber);
357  }
358 
359  fReadState.fStatus = MonarchReadState::kContinueReading;
360  }
361 
362  // Read the data from the records
363  for (unsigned iChannel = 0; iChannel < nChannels; ++iChannel)
364  {
365  // set the data
366  newRecords[iChannel]->SetAt(monarchRecordData[iChannel]->at(fReadState.fReadPtrOffset), iBin);
367  }
368 
369  // advance the pointer for the next bin
370  ++(fReadState.fReadPtrOffset);
371 
372  // check if we've reached the end of a monarch record
373  if (fReadState.fReadPtrOffset >= recordSize)
374  {
375  KTDEBUG(eggreadlog, "End of Monarch record reached.");
376  fReadState.fStatus = MonarchReadState::kReachedNextRecord;
377  }
378  } // end loop over bins
379 
380  sliceHeader.SetEndRecordNumber(fReadState.fAbsoluteRecordOffset);
381  sliceHeader.SetEndSampleNumber(fReadState.fReadPtrOffset - 1);
382 
383  // delete the monarchRecordData objects
384  while(! monarchRecordData.empty())
385  {
386  delete monarchRecordData.back();
387  monarchRecordData.pop_back();
388  }
389 
390  // finally, set the records in the new data object
391  KTRawTimeSeriesData& tsData = newData->Of< KTRawTimeSeriesData >().SetNComponents(nChannels);
392  for (unsigned iChannel = 0; iChannel < nChannels; ++iChannel)
393  {
394  tsData.SetTimeSeries(newRecords[iChannel], iChannel);
395  }
396 
397  return newData;
398  }
