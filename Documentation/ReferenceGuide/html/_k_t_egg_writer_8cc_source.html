191  bool KTEggWriter::WriteTSData(KTSliceHeader& slHeader, KTTimeSeriesData& tsData)
192  {
193  if (fFileStatus == kClosed)
194  {
195  KTERROR(eggwritelog, "Cannot write record because the file has not been opened");
196  return false;
197  }
198 
199  if (fFileStatus == kOpened)
200  {
201  KTWARN(eggwritelog, "Writing record on file with no header");
202  }
203  fFileStatus = kWritingRecords;
204 
205  unsigned nComponents = tsData.GetNComponents();
206  if (nComponents != fExpectedNChannels)
207  {
208  KTERROR(eggwritelog, "Received data contains " << nComponents << " channels of data; " << fExpectedNChannels << " were expected");
209  return false;
210  }
211 
212  if (fExpectedNChannels > 2)
213  {
214  KTERROR(eggwritelog, "Interleaved record writing is only supported for 2 or fewer channels, not " << fExpectedNChannels);
215  return false;
216  }
217 
218  fMonarch->SetInterface(monarch::sInterfaceSeparate);
219 
220  CopyATimeSeries(0, slHeader, tsData, fMonarch->GetRecordSeparateOne());
221 
222  if (fExpectedNChannels == 2)
223  {
224  CopyATimeSeries(1, slHeader, tsData, fMonarch->GetRecordSeparateTwo());
225  }
226 
227  try
228  {
229  fMonarch->WriteRecord();
230  }
231  catch (MonarchException& e)
232  {
233  KTERROR(eggwritelog, "A problem occurred while writing the record (interleaved): " << e.what());
234  return false;
235  }
236  return true;

239  bool KTEggWriter::CopyATimeSeries(unsigned component, const KTSliceHeader& slHeader, const KTTimeSeriesData& tsData, MonarchRecord* record)
240  {
241  const KTTimeSeries* ts = tsData.GetTimeSeries(component);
242  if (ts->GetNTimeBins() != fExpectedRecordSize)
243  {
244  KTERROR(eggwritelog, "Time series does not have the correct size (should be " << fExpectedRecordSize << "; provided component 0: " << ts->GetNTimeBins());
245  return false;
246  }
247 
248  record->fAcquisitionId = slHeader.GetAcquisitionID(component);
249  record->fRecordId = slHeader.GetRecordID(component);
250  record->fTime = slHeader.GetTimeStamp(component);
251 
252  double value;
253  double scale = 255. / fDigitizerFullscale;
254  for (unsigned iBin = 0; iBin < fExpectedRecordSize; iBin++)
255  {
256  value = ts->GetValue(iBin) * scale;
257  if (value >= 256) value = 255.;
258  if (value < 0.) value = 0.;
259  record->fData[iBin] = DataType(floor(value));
260  }
261  return true;
262  }
