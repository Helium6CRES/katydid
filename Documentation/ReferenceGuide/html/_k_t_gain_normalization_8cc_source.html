92  bool KTGainNormalization::Normalize(KTFrequencySpectrumDataPolar& fsData, KTGainVariationData& gvData)
93  {
94  if (fCalculateMinBin) SetMinBin(fsData.GetSpectrumPolar(0)->FindBin(fMinFrequency));
95  if (fCalculateMaxBin) SetMaxBin(fsData.GetSpectrumPolar(0)->FindBin(fMaxFrequency));
96 
97  unsigned nComponents = fsData.GetNComponents();
98  if (nComponents != gvData.GetNComponents())
99  {
100  KTERROR(gnlog, "Mismatch in the number of channels between the frequency spectrum data and the gain variation data! Aborting.");
101  return false;
102  }
103 
104  KTNormalizedFSDataPolar& newData = fsData.Of< KTNormalizedFSDataPolar >().SetNComponents(nComponents);
105 
106  for (unsigned iComponent=0; iComponent<nComponents; ++iComponent)
107  {
108  KTFrequencySpectrumPolar* newSpectrum = Normalize(fsData.GetSpectrumPolar(iComponent), gvData.GetSpline(iComponent));
109  if (newSpectrum == NULL)
110  {
111  KTERROR(gnlog, "Normalization of spectrum " << iComponent << " failed for some reason. Continuing processing.");
112  continue;
113  }
114  KTDEBUG(gnlog, "Computed normalization; size: " << newSpectrum->size() << "; range: " << newSpectrum->GetRangeMin() << " - " << newSpectrum->GetRangeMax());
115  newData.SetSpectrum(newSpectrum, iComponent);
116  }
117  KTINFO(gnlog, "Completed gain normalization of " << nComponents << " frequency spectra (polar)");
118 
119  return true;

182  KTFrequencySpectrumPolar* KTGainNormalization::Normalize(const KTFrequencySpectrumPolar* frequencySpectrum, const KTSpline* spline)
183  {
184  unsigned nBins = fMaxBin - fMinBin + 1;
185  double freqMin = frequencySpectrum->GetBinLowEdge(fMinBin);
186  double freqMax = frequencySpectrum->GetBinLowEdge(fMaxBin) + frequencySpectrum->GetBinWidth();
187 
188  KTSpline::Implementation* splineImp = spline->Implement(nBins, freqMin, freqMax);
189 
190  unsigned nSpectrumBins = frequencySpectrum->size();
191  double freqSpectrumMin = frequencySpectrum->GetRangeMin();
192  double freqSpectrumMax = frequencySpectrum->GetRangeMax();
193 
194  KTDEBUG(gnlog, "Creating new FS for normalized data: " << nSpectrumBins << ", " << freqSpectrumMin << ", " << freqSpectrumMax);
195  KTFrequencySpectrumPolar* newSpectrum = new KTFrequencySpectrumPolar(nSpectrumBins, freqSpectrumMin, freqSpectrumMax);
196  newSpectrum->SetNTimeBins(frequencySpectrum->GetNTimeBins());
197 
198  // First directly copy data that's outside the scaling range
199  unsigned iBin;
200 #pragma omp parallel default(shared)
201  {
202 #pragma omp for private(iBin)
203  for (iBin=0; iBin < fMinBin; ++iBin)
204  {
205  (*newSpectrum)(iBin).set_polar((*frequencySpectrum)(iBin).abs(), (*frequencySpectrum)(iBin).arg());
206  }
207 #pragma omp for private(iBin)
208  for (iBin=fMaxBin+1; iBin < nSpectrumBins; ++iBin)
209  {
210  (*newSpectrum)(iBin).set_polar((*frequencySpectrum)(iBin).abs(), (*frequencySpectrum)(iBin).arg());
211  }
212 
213  // Then scale the bins within the scaling range
214 #pragma omp for private(iBin)
215  for (iBin=fMinBin; iBin < fMaxBin+1; ++iBin)
216  {
217  (*newSpectrum)(iBin).set_polar((*frequencySpectrum)(iBin).abs() / (*splineImp)(iBin - fMinBin), (*frequencySpectrum)(iBin).arg());
218  }
219  }
220 
221  spline->AddToCache(splineImp);
222 
223  return newSpectrum;
224  }
