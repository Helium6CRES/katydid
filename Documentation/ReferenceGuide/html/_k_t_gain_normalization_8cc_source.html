93  {

160  KTFrequencySpectrumPolar* KTGainNormalization::Normalize(const KTFrequencySpectrumPolar* frequencySpectrum, const KTSpline* spline)
161  {
162  UInt_t nBins = fMaxBin - fMinBin + 1;
163  Double_t freqMin = frequencySpectrum->GetBinLowEdge(fMinBin);
164  Double_t freqMax = frequencySpectrum->GetBinLowEdge(fMaxBin) + frequencySpectrum->GetBinWidth();
165 
166  KTSpline::Implementation* splineImp = spline->Implement(nBins, freqMin, freqMax);
167 
168  UInt_t nSpectrumBins = frequencySpectrum->size();
169  Double_t freqSpectrumMin = frequencySpectrum->GetRangeMin();
170  Double_t freqSpectrumMax = frequencySpectrum->GetRangeMax();
171 
172  KTFrequencySpectrumPolar* newSpectrum = new KTFrequencySpectrumPolar(nSpectrumBins, freqSpectrumMin, freqSpectrumMax);
173 
174  // First directly copy data that's outside the scaling range
175  UInt_t iBin;
176 #pragma omp parallel default(shared)
177  {
178 #pragma omp for private(iBin)
179  for (iBin=0; iBin < fMinBin; iBin++)
180  {
181  (*newSpectrum)(iBin).set_polar((*frequencySpectrum)(iBin).abs(), (*frequencySpectrum)(iBin).arg());
182  }
183 #pragma omp for private(iBin)
184  for (iBin=fMaxBin+1; iBin < nSpectrumBins; iBin++)
185  {
186  (*newSpectrum)(iBin).set_polar((*frequencySpectrum)(iBin).abs(), (*frequencySpectrum)(iBin).arg());
187  }
188 
189  // Then scale the bins within the scaling range
190 #pragma omp for private(iBin)
191  for (iBin=fMinBin; iBin < fMaxBin+1; iBin++)
192  {
193  (*newSpectrum)(iBin).set_polar((*frequencySpectrum)(iBin).abs() / (*splineImp)(iBin), (*frequencySpectrum)(iBin).arg());
194  }
195  }
196 
197  spline->AddToCache(splineImp);
198 
199  return newSpectrum;
200  }
