267  template< class XDataType >
268  void KTDataAccumulator::SlotFunction(Nymph::KTDataPtr data)
269  {
270  // Standard data slot pattern:
271  // Check to ensure that the required data type is present
272  if (! data->Has< XDataType >())
273  {
274  KTERROR(avlog_hh, "Data not found with type <" << typeid(XDataType).name() << ">");
275  return;
276  }
277  // Call the function
278  if (! AddData(data->Of< XDataType >()))
279  {
280  KTERROR(avlog_hh, "Something went wrong while analyzing data with type <" << typeid(XDataType).name() << ">");
281  return;
282  }
283  // If there's a signal pointer, emit the signal
284  SignalMapIt sigIt = fSignalMap.find(&typeid(XDataType));
285  if (sigIt != fSignalMap.end())
286  {
287  sigIt->second.fSignalCount++;
288  unsigned sigCount = sigIt->second.fSignalCount;
289  if (sigCount == fSignalInterval)
290  {
291  (*sigIt->second.fAccumulatingSignal)(fLastAccumulatorPtr->fData);
292  sigIt->second.fSignalCount = 0;
293  }
294  if (data->GetLastData())
295  {
296  KTDEBUG(avlog_hh, "Emitting last-data signal");
297  if (! Scale(fLastAccumulatorPtr->fData->Of< XDataType >(), fLastAccumulatorPtr->fSliceHeader))
298  {
299  KTERROR(avlog_hh, "Something went wrong while scaling data with type <" << typeid(XDataType).name() << ">");
300  }
301  (*sigIt->second.fFinishedSignal)(fLastAccumulatorPtr->fData);
302  }
303  }
304  return;
305  }
