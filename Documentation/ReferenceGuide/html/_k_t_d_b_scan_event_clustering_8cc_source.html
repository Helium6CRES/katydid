129  bool KTDBScanEventClustering::DoClustering()
130  {
131  KTPROG(tclog, "Starting DBSCAN event clustering");
132 
133  KTDBScan< DistanceMatrix > dbScan;
134 
135  dbScan.SetRadius(1.);
136  dbScan.SetMinPoints(fMinPoints);
137  KTINFO(tclog, "DBScan configured");
138 
139  for (unsigned iComponent = 0; iComponent < fCompTracks.size(); ++iComponent)
140  {
141  KTDEBUG(tclog, "Clustering component " << iComponent);
142 
143  if (fCompTracks[iComponent].empty() )
144  continue;
145 
146  // calculate the scaling
147  Point scale = fRadii;
148  for (Point::iterator dIt = scale.begin(); dIt != scale.end(); ++dIt)
149  {
150  *dIt = 1. / (*dIt * KTMath::Sqrt2());
151  }
152 
153  // new array for normalized points
154  Points normPoints(fCompTracks[iComponent].size());
155  Point newPoint(fNDimensions);
156  // normalize points
157  KTDEBUG(tclog, "Scale: " << scale);
158  unsigned iPoint = 0;
159  for (vector< KTProcessedTrackData >::const_iterator pIt = fCompTracks[iComponent].begin(); pIt != fCompTracks[iComponent].end(); ++pIt)
160  {
161  //std::cerr << "1" << std::endl;
162  newPoint(0) = pIt->GetStartTimeInRunC() * scale(0); // start time
163  //std::cerr << "2" << std::endl;
164  newPoint(1) = pIt->GetStartFrequency() * scale(1); // start freq
165  //std::cerr << "3" << std::endl;
166  newPoint(2) = pIt->GetEndTimeInRunC() * scale(0); // end time
167  //std::cerr << "4" << std::endl;
168  newPoint(3) = pIt->GetEndFrequency() * scale(1); // end freq
169  //std::cerr << "5" << std::endl;
170 
171 #ifndef NDEBUG
172  std::stringstream ptStr;
173  ptStr << "Point -- before: (" << pIt->GetStartTimeInRunC() << ", " << pIt->GetStartFrequency() << ", " << pIt->GetEndTimeInRunC() << ", " << pIt->GetEndFrequency() << ")";
174 #endif
175  KTDEBUG(tclog, ptStr.str() << " -- after: " << newPoint);
176  normPoints[iPoint++] = newPoint;
177  }
178 
179  DistanceMatrix distMat;
180  distMat.ComputeDistances< TrackDistance< Point > >(normPoints);
181 
182  // do the clustering!
183  KTINFO(tclog, "Starting DBSCAN");
184  KTDBScan< DistanceMatrix >::DBSResults results;
185  if (! dbScan.DoClustering(distMat, results))
186  {
187  KTERROR(tclog, "An error occurred while clustering");
188  return false;
189  }
190  KTDEBUG(tclog, "DBSCAN finished");
191 
192  // loop over the clusters found, and create data objects for them
193  KTDEBUG(tclog, "Found " << results.fClusters.size() << " clusters; creating candidate events");
194  for (vector< KTDBScan< DistanceMatrix >::Cluster >::const_iterator clustIt = results.fClusters.begin(); clustIt != results.fClusters.end(); ++clustIt)
195  {
196  if (clustIt->empty())
197  {
198  KTWARN(tclog, "Empty cluster");
199  continue;
200  }
201 
202  KTDEBUG(tclog, "Creating event " << fDataCount << "; includes " << clustIt->size() << " tracks");
203 
204  ++fDataCount;
205 
206  Nymph::KTDataPtr data(new Nymph::KTData());
207 
208  KTMultiTrackEventData& eventData = data->Of< KTMultiTrackEventData >();
209  eventData.SetComponent(iComponent);
210  eventData.SetAcquisitionID(fCompTracks[0][0].GetAcquisitionID());
211  eventData.SetEventID(fDataCount);
212 
213  for (KTDBScan< DistanceMatrix >::Cluster::const_iterator pointIdIt = clustIt->begin(); pointIdIt != clustIt->end(); ++pointIdIt)
214  {
215  eventData.AddTrack(fCompTracks[iComponent][*pointIdIt]);
216  }
217 
218  eventData.ProcessTracks();
219 
220  fCandidates.insert(data);
221  fEventSignal(data);
222  } // loop over clusters
223 
224  fCompTracks[iComponent].clear();
225 
226  } // loop over components
227 
228  KTDEBUG(tclog, "Clustering complete");
229  fClusterDoneSignal();
230 
231  return true;
232  }
