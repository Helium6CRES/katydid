130 bool KTImpulseAnalysis::Analyze(KTFrequencySpectrumDataPolar& fsData)
131 {
132  KTFrequencySpectrumPolar* fs = fsData.GetSpectrumPolar(0);
133  unsigned size = fs->size();
134  double binWidth = fs->GetBinWidth();
135 
136  // Loop over all the bins in the FS
137  // Calculate the sum and keep the peak bin information
138  unsigned peakBin = 0, previousPeakBin = 0;
139  double peakBinValue = -1.;
140  double previousPeakValue = -1.;
141  double sum = 0.; // sum of squares, since we want to calculate the power
142  double value;
143  for (unsigned iBin=0; iBin < size; iBin++)
144  {
145  value = (*fs)(iBin).abs();
146  sum += value * value;
147  if (value > peakBinValue)
148  {
149  previousPeakValue = peakBinValue;
150  previousPeakBin = peakBin;
151  peakBinValue = value;
152  peakBin = iBin;
153  }
154  }
155 
156  double peakFraction = peakBinValue * peakBinValue / sum;
157  double leakagePeakBin = 1. - peakFraction;
158  double peakThreeBinFraction = (peakBinValue*peakBinValue + (*fs)(peakBin-1).abs()*(*fs)(peakBin-1).abs() + (*fs)(peakBin+1).abs()*(*fs)(peakBin+1).abs()) / sum;
159  double leakagePeakThreeBin = 1. - peakThreeBinFraction;
160  double secondHighestBinRatio = previousPeakValue / peakBinValue;
161  //KTDEBUG(irlog, peakFraction << " " << leakagePeakBin << " " << peakThreeBinFraction << " " << leakagePeakThreeBin << " " << peakBinValue << " " << peakThreeBinValue << " " << sum);
162 
163  // Examine fractional peak width
164  double fraction = 0.1;
165  unsigned leftSideBin = peakBin, rightSideBin = peakBin;
166  if (peakBinValue > 0.)
167  {
168  double fractionalPowerValue = sqrt(fraction) * peakBinValue;
169  while (leftSideBin > 0 && (*fs)(leftSideBin).abs() >= fractionalPowerValue)
170  {
171  leftSideBin--;
172  }
173  while (rightSideBin < size-1 && (*fs)(rightSideBin).abs() >= fractionalPowerValue)
174  {
175  rightSideBin++;
176  }
177  if (leftSideBin != 0 && (*fs)(leftSideBin).abs() < fractionalPowerValue) leftSideBin++; // if we didn't hit the left edge, we went one beyond the FWHM peak
178  if (rightSideBin != size-1 && (*fs)(rightSideBin).abs() < fractionalPowerValue) rightSideBin--; // if we didn't heit the right edge, we went one beyond the FWHM peak
179  }
180 
181  unsigned fracWidthBins = rightSideBin - leftSideBin + 1;
182  double fracWidth = double(fracWidthBins) * binWidth;
183 
184  KTPROG(irlog, "Frequency of peak: " << fs->GetBinCenter(peakBin) << " Hz (bin # " << peakBin << ")");
185  KTPROG(irlog, "Leakage fraction (1 bin): " << leakagePeakBin);
186  KTPROG(irlog, "Leakage fraction (3 bin): " << leakagePeakThreeBin);
187  KTPROG(irlog, "Second-highest-bin ratio: " << secondHighestBinRatio << " (bin # " << previousPeakBin << ")");
188  KTPROG(irlog, fraction * 100. << "% width: " << fracWidth << " Hz (" << fracWidthBins << " bins)");
189 
190  return true;
191 }
