34  Bool_t KTProcessorToolbox::Configure(const KTPStoreNode* node)
35  {
36  // Deal with "processor" blocks first
37  KTPStoreNode::csi_pair itPair = node->EqualRange("processor");
38  for (KTPStoreNode::const_sorted_iterator it = itPair.first; it != itPair.second; it++)
39  {
40  KTPStoreNode subNode = KTPStoreNode(&(it->second));
41  if (! subNode.HasData("type"))
42  {
43  KTERROR(proclog, "Unable to create processor: no processor type given");
44  return false;
45  }
46  string procType = subNode.GetData("type");
47 
48  string procName;
49  if (! subNode.HasData("name"))
50  {
51  KTINFO(proclog, "No name given for processor of type <" << procType << ">; using type as name.");
52  procName = procType;
53  }
54  else
55  {
56  procName = subNode.GetData("name");
57  }
58  KTProcessor* newProc = fProcFactory->Create(procType);
59  if (newProc == NULL)
60  {
61  KTERROR(proclog, "Unable to create processor of type <" << procType << ">");
62  return false;
63  }
64 
65  Bool_t isTopLevel = false;
66  if (subNode.HasData("is-top-level"))
67  {
68  isTopLevel = subNode.GetData< Bool_t >("is-top-level");
69  }
70 
71  if (! AddProcessor(procName, newProc, isTopLevel))
72  {
73  KTERROR(proclog, "Unable to add processor <" << procName << ">");
74  delete newProc;
75  return false;
76  }
77  }
78 
79  // Then deal with "connection blocks"
80  itPair = node->EqualRange("connection");
81  for (KTPStoreNode::const_sorted_iterator it = itPair.first; it != itPair.second; it++)
82  {
83  KTPStoreNode subNode = KTPStoreNode(&(it->second));
84  if (! subNode.HasData("signal-processor") || ! subNode.HasData("signal-name") ||
85  ! subNode.HasData("slot-processor") || ! subNode.HasData("slot-name"))
86  {
87  KTERROR(proclog, "Signal/Slot connection information is incomplete!");
88  return false;
89  }
90 
91  KTProcessor* signalProc = GetProcessor(subNode.GetData("signal-processor"));
92  KTProcessor* slotProc = GetProcessor(subNode.GetData("slot-processor"));
93 
94  if (signalProc == NULL)
95  {
96  KTERROR(proclog, "Processor named <" << subNode.GetData("signal-processor") << "> was not found!");
97  return false;
98  }
99  if (slotProc == NULL)
100  {
101  KTERROR(proclog, "Processor named <" << subNode.GetData("slot-processor") << "> was not found!");
102  return false;
103  }
104 
105  string signalName = subNode.GetData("signal-name");
106  string slotName = subNode.GetData("slot-name");
107 
108  Bool_t useGroupOrdering = false;
109  Int_t groupOrder = 0;
110  if (subNode.HasData("group-order"))
111  {
112  useGroupOrdering = true;
113  groupOrder = subNode.GetData< Int_t >("group-order");
114  }
115 
116  try
117  {
118  if (useGroupOrdering)
119  {
120  signalProc->ConnectASlot(signalName, slotProc, slotName, groupOrder);
121  }
122  else
123  {
124  signalProc->ConnectASlot(signalName, slotProc, slotName);
125  }
126  }
127  catch (std::exception& e)
128  {
129  KTERROR(proclog, "An error occurred while connecting signals and slots:\n"
130  << "\tSignal " << signalName << " from processor " << subNode.GetData("signal_processor") << '\n'
131  << "\tSlot " << slotName << " from processor " << subNode.GetData("slot_processor") << '\n'
132  << '\t' << e.what());
133  return false;
134  }
135  }
136 
137  // Finally, deal with processor-run specifications
138  const KTPStoreNode* subNodePtr = node->GetChild("run-queue");
139  if (subNodePtr != NULL)
140  {
141  for (KTPStoreNode::const_iterator iter = subNodePtr->Begin(); iter != subNodePtr->End(); iter++)
142  {
143  string procName = iter->second.get_value< string >("processor");
144  KTProcessor* procForRunQueue = GetProcessor(procName);
145  if (procForRunQueue == NULL)
146  {
147  KTERROR(proclog, "Unable to find processor <" << procName << "> requested for the run queue");
148  delete subNodePtr;
149  return false;
150  }
151  fRunQueue.push_back(procForRunQueue);
152  }
153  }
154  else
155  {
156  KTWARN(proclog, "No run queue was specified during configuration.");
157  }
158  delete subNodePtr;
159 
160  return true;
161  }

163  Bool_t KTProcessorToolbox::ConfigureProcessors(const KTPStoreNode* node)
164  {
165  for (ProcMapIt iter = fProcMap.begin(); iter != fProcMap.end(); iter++)
166  {
167  if (iter->second.fIsTopLevel)
168  {
169  KTDEBUG(proclog, "Attempting to configure top-level processor <" << iter->first << ">");
170  string procName = iter->first;
171  string nameUsed;
172  const KTPStoreNode* subNode = node->GetChild(procName);
173  if (subNode == NULL)
174  {
175  string configName = iter->second.fProc->GetConfigName();
176  KTWARN(proclog, "Did not find a PSToreNode <" << procName << ">\n"
177  "\tWill check using the generic name of the processor, <" << configName << ">.");
178  subNode = node->GetChild(configName);
179  if (subNode == NULL)
180  {
181  KTWARN(proclog, "Did not find a PStoreNode <" << configName << ">\n"
182  "\tProcessor <" << iter->first << "> was not configured.");
183  continue;
184  }
185  nameUsed = configName;
186  }
187  else
188  {
189  nameUsed = procName;
190  }
191  if (! iter->second.fProc->Configure(subNode))
192  {
193  KTERROR(proclog, "An error occurred while configuring processor <" << iter->first << "> with PStoreNode <" << nameUsed << ">");
194  return false;
195  }
196  }
197  }
198  return true;
199  }
