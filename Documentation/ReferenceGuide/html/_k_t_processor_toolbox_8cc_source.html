43  bool KTProcessorToolbox::Configure(const scarab::param_node* node)
44  {
45  KTPROG(proclog, "Configuring . . .");
46  // Deal with "processor" blocks first
47  const scarab::param_array* procArray = node->array_at( "processors" );
48  if (procArray == NULL)
49  {
50  KTWARN(proclog, "No processors were specified");
51  }
52  else
53  {
54  for( scarab::param_array::const_iterator procIt = procArray->begin(); procIt != procArray->end(); ++procIt )
55  {
56  if( ! (*procIt)->is_node() )
57  {
58  KTERROR( proclog, "Invalid processor entry: not a node" );
59  return false;
60  }
61  const scarab::param_node* procNode = &( (*procIt)->as_node() );
62 
63  if (! procNode->has("type"))
64  {
65  KTERROR(proclog, "Unable to create processor: no processor type given");
66  return false;
67  }
68  string procType = procNode->get_value("type");
69 
70  string procName;
71  if (! procNode->has("name"))
72  {
73  KTINFO(proclog, "No name given for processor of type <" << procType << ">; using type as name.");
74  procName = procType;
75  }
76  else
77  {
78  procName = procNode->get_value("name");
79  }
80  KTProcessor* newProc = fProcFactory->create(procType, procType);
81  if (newProc == NULL)
82  {
83  KTERROR(proclog, "Unable to create processor of type <" << procType << ">");
84  return false;
85  }
86 
87  if (! AddProcessor(procName, newProc))
88  {
89  KTERROR(proclog, "Unable to add processor <" << procName << ">");
90  delete newProc;
91  return false;
92  }
93  }
94  }
95 
96 
97  // Then deal with connections"
98  const scarab::param_array* connArray = node->array_at( "connections" );
99  if (connArray == NULL)
100  {
101  KTWARN(proclog, "No connections were specified");
102  }
103  else
104  {
105  for( scarab::param_array::const_iterator connIt = connArray->begin(); connIt != connArray->end(); ++connIt )
106  {
107  if( ! (*connIt)->is_node() )
108  {
109  KTERROR( proclog, "Invalid connection entry: not a node" );
110  return false;
111  }
112  const scarab::param_node* connNode = &( (*connIt)->as_node() );
113 
114  if ( ! connNode->has("signal") || ! connNode->has("slot") )
115  {
116  KTERROR(proclog, "Signal/Slot connection information is incomplete!");
117  if (connNode->has("signal"))
118  {
119  KTWARN(proclog, "signal = " << connNode->get_value("signal"));
120  }
121  else
122  {
123  KTERROR(proclog, "signal = MISSING");
124  }
125 
126  if (connNode->has("slot"))
127  {
128  KTWARN(proclog, "slot = " << connNode->get_value("slot"));
129  }
130  else
131  {
132  KTERROR(proclog, "slot = MISSING");
133  }
134  return false;
135  }
136 
137  string signalProcName, signalName;
138  if (! ParseSignalSlotName(connNode->get_value("signal"), signalProcName, signalName))
139  {
140  KTERROR(proclog, "Unable to parse signal name: <" << connNode->get_value("signal") << ">");
141  return false;
142  }
143 
144  string slotProcName, slotName;
145  if (! ParseSignalSlotName(connNode->get_value("slot"), slotProcName, slotName))
146  {
147  KTERROR(proclog, "Unable to parse slot name: <" << connNode->get_value("slot") << ">");
148  return false;
149  }
150 
151 
152  KTProcessor* signalProc = GetProcessor(signalProcName);
153  if (signalProc == NULL)
154  {
155  KTERROR(proclog, "Processor named <" << signalProcName << "> was not found!");
156  return false;
157  }
158 
159  KTProcessor* slotProc = GetProcessor(slotProcName);
160  if (slotProc == NULL)
161  {
162  KTERROR(proclog, "Processor named <" << slotProcName << "> was not found!");
163  return false;
164  }
165 
166  try
167  {
168  if (connNode->has("order"))
169  {
170  signalProc->ConnectASlot(signalName, slotProc, slotName, connNode->get_value< int >("order"));
171  }
172  else
173  {
174  signalProc->ConnectASlot(signalName, slotProc, slotName);
175  }
176  }
177  catch (std::exception& e)
178  {
179  KTERROR(proclog, "An error occurred while connecting signals and slots:\n"
180  << "\tSignal " << signalName << " from processor " << signalProcName << " (a.k.a. " << signalProc->GetConfigName() << ")" << '\n'
181  << "\tSlot " << slotName << " from processor " << slotProcName << " (a.k.a. " << slotProc->GetConfigName() << ")" << '\n'
182  << '\t' << e.what());
183  return false;
184  }
185  KTINFO(proclog, "Signal <" << signalProcName << ":" << signalName << "> connected to slot <" << slotProcName << ":" << signalName << ">");
186 
187  }
188  }
189 
190 
191  // Finally, deal with processor-run specifications
192  // The run queue is an array of processor names, or groups of names, which will be run sequentially.
193  // If names are grouped (in another array), those in that group will be run in parallel.
194  // In single threaded mode all threads will be run sequentially in the order they were specified.
195  const scarab::param_array* rqArray = node->array_at( "run-queue" );
196  if (rqArray == NULL)
197  {
198  KTWARN(proclog, "Run queue was not specified");
199  }
200  else
201  {
202  for (scarab::param_array::const_iterator rqIt = rqArray->begin(); rqIt != rqArray->end(); ++rqIt)
203  {
204  ThreadGroup threadGroup;
205  if ((*rqIt)->is_value())
206  {
207  const scarab::param_value* rqValue = &( (*rqIt)->as_value() );
208 
209  if (! AddProcessorToThreadGroup(rqValue, threadGroup)) return false;
210  }
211  else if ((*rqIt)->is_node())
212  {
213  const scarab::param_array* rqNode = &( (*rqIt)->as_array() );
214 
215  for (scarab::param_array::const_iterator rqArrayIt = rqNode->begin(); rqArrayIt != rqNode->end(); ++rqArrayIt)
216  {
217  if (! (*rqArrayIt)->is_value())
218  {
219  KTERROR(proclog, "Invalid run-queue array entry: not a value");
220  return false;
221  }
222  const scarab::param_value* rqValue = &( (*rqArrayIt)->as_value() );
223  if (! AddProcessorToThreadGroup(rqValue, threadGroup)) return false;
224  }
225  }
226  else
227  {
228  KTERROR(proclog, "Invalid run-queue entry: not a value or array");
229  return false;
230  }
231  fRunQueue.push_back(threadGroup);
232  }
233  }
234 
235  return true;

309  bool KTProcessorToolbox::Run()
310  {
311  KTPROG(proclog, "Beginning processing . . .");
312 #ifndef SINGLETHREADED
313  unsigned iGroup = 0;
314 #endif
315  for (RunQueue::const_iterator rqIter = fRunQueue.begin(); rqIter != fRunQueue.end(); ++rqIter)
316  {
317 #ifdef SINGLETHREADED
318  for (ThreadGroup::const_iterator tgIter = rqIter->begin(); tgIter != rqIter->end(); ++tgIter)
319  {
320  if (! tgIter->fProc->Run())
321  {
322  return false;
323  }
324  }
325 #else
326  KTDEBUG(proclog, "Starting thread group " << iGroup);
327  boost::thread_group parallelThreads;
328  unsigned iThread = 0;
329  for (ThreadGroup::const_iterator tgIter = rqIter->begin(); tgIter != rqIter->end(); ++tgIter)
330  {
331  // create a boost::thread object to launch the thread
332  // use boost::ref to avoid copying the processor
333  KTDEBUG(proclog, "Starting thread " << iThread << ": " << tgIter->fName);
334  parallelThreads.create_thread(boost::ref(*(tgIter->fProc)));
335  //parallelThreads.create_thread(boost::ref(**tgIter));
336  iThread++;
337  }
338  // wait for execution to complete
339  parallelThreads.join_all();
340  iGroup++;
341 #endif
342  }
343  KTPROG(proclog, ". . . processing complete.");
344  return true;
345  }
