43  Bool_t KTProcessorToolbox::Configure(const KTPStoreNode* node)
44  {
45  // Deal with "processor" blocks first
46  KTPStoreNode::csi_pair itPair = node->EqualRange("processor");
47  for (KTPStoreNode::const_sorted_iterator it = itPair.first; it != itPair.second; it++)
48  {
49  KTPStoreNode subNode = KTPStoreNode(&(it->second));
50  if (! subNode.HasData("type"))
51  {
52  KTERROR(proclog, "Unable to create processor: no processor type given");
53  return false;
54  }
55  string procType = subNode.GetData("type");
56 
57  string procName;
58  if (! subNode.HasData("name"))
59  {
60  KTINFO(proclog, "No name given for processor of type <" << procType << ">; using type as name.");
61  procName = procType;
62  }
63  else
64  {
65  procName = subNode.GetData("name");
66  }
67  KTProcessor* newProc = fProcFactory->Create(procType);
68  if (newProc == NULL)
69  {
70  KTERROR(proclog, "Unable to create processor of type <" << procType << ">");
71  return false;
72  }
73 
74  Bool_t isTopLevel = false;
75  if (subNode.HasData("is-top-level"))
76  {
77  isTopLevel = subNode.GetData< Bool_t >("is-top-level");
78  }
79 
80  if (! AddProcessor(procName, newProc))
81  {
82  KTERROR(proclog, "Unable to add processor <" << procName << ">");
83  delete newProc;
84  return false;
85  }
86  }
87 
88  // Then deal with "connection blocks"
89  itPair = node->EqualRange("connection");
90  for (KTPStoreNode::const_sorted_iterator it = itPair.first; it != itPair.second; it++)
91  {
92  KTPStoreNode subNode = KTPStoreNode(&(it->second));
93  if (! subNode.HasData("signal-processor") || ! subNode.HasData("signal-name") ||
94  ! subNode.HasData("slot-processor") || ! subNode.HasData("slot-name"))
95  {
96  KTERROR(proclog, "Signal/Slot connection information is incomplete!");
97  if (subNode.HasData("signal-processor"))
98  {
99  KTWARN(proclog, "signal-processor = " << subNode.GetData<string>("signal-processor"));
100  }
101  else
102  {
103  KTERROR(proclog, "signal-processor = MISSING");
104  }
105  if (subNode.HasData("signal-name"))
106  {
107  KTWARN(proclog, "signal-name = " << subNode.GetData<string>("signal-name"));
108  }
109  else
110  {
111  KTERROR(proclog, "signal-name = MISSING");
112  }
113  if (subNode.HasData("slot-processor"))
114  {
115  KTWARN(proclog, "slot-processor = " << subNode.GetData<string>("slot-processor"));
116  }
117  else
118  {
119  KTERROR(proclog, "slot-processor = MISSING");
120  }
121  if (subNode.HasData("slot-name"))
122  {
123  KTWARN(proclog, "slot-name = " << subNode.GetData<string>("slot-name"));
124  }
125  else
126  {
127  KTERROR(proclog, "slot-namer = MISSING");
128  }
129  return false;
130  }
131 
132  KTProcessor* signalProc = GetProcessor(subNode.GetData("signal-processor"));
133  KTProcessor* slotProc = GetProcessor(subNode.GetData("slot-processor"));
134 
135  if (signalProc == NULL)
136  {
137  KTERROR(proclog, "Processor named <" << subNode.GetData("signal-processor") << "> was not found!");
138  return false;
139  }
140  if (slotProc == NULL)
141  {
142  KTERROR(proclog, "Processor named <" << subNode.GetData("slot-processor") << "> was not found!");
143  return false;
144  }
145 
146  string signalName = subNode.GetData("signal-name");
147  string slotName = subNode.GetData("slot-name");
148 
149  Bool_t useGroupOrdering = false;
150  Int_t groupOrder = 0;
151  if (subNode.HasData("group-order"))
152  {
153  useGroupOrdering = true;
154  groupOrder = subNode.GetData< Int_t >("group-order");
155  }
156 
157  try
158  {
159  if (useGroupOrdering)
160  {
161  signalProc->ConnectASlot(signalName, slotProc, slotName, groupOrder);
162  }
163  else
164  {
165  signalProc->ConnectASlot(signalName, slotProc, slotName);
166  }
167  }
168  catch (std::exception& e)
169  {
170  KTERROR(proclog, "An error occurred while connecting signals and slots:\n"
171  << "\tSignal " << signalName << " from processor " << subNode.GetData("signal-processor") << " (a.k.a. " << signalProc->GetConfigName() << ")" << '\n'
172  << "\tSlot " << slotName << " from processor " << subNode.GetData("slot-processor") << " (a.k.a. " << slotProc->GetConfigName() << ")" << '\n'
173  << '\t' << e.what());
174  return false;
175  }
176  }

178  // Finally, deal with processor-run specifications
179  // In the current implementation there is only one ThreadGroup, and all processors specified will be run in separate threads in parallel.
180  // In single threaded mode all threads will be run sequentially in the order they were specified.
181  const KTPStoreNode* subNodePtr = node->GetChild("run-queue");
182  if (subNodePtr != NULL)
183  {
184  ThreadGroup threadGroup;
185  for (KTPStoreNode::const_iterator iter = subNodePtr->Begin(); iter != subNodePtr->End(); iter++)
186  {
187  KTPStoreNode subSubNode = KTPStoreNode(&(iter->second));
188  string procName = subSubNode.GetValue< string >();
189  KTProcessor* procForRunQueue = GetProcessor(procName);
190  KTDEBUG(proclog, "Adding processor of type " << procName << " to the run queue");
191  if (procForRunQueue == NULL)
192  {
193  KTERROR(proclog, "Unable to find processor <" << procName << "> requested for the run queue");
194  delete subNodePtr;
195  return false;
196  }
197 
198  KTPrimaryProcessor* primaryProc = dynamic_cast< KTPrimaryProcessor* >(procForRunQueue);
199  if (primaryProc == NULL)
200  {
201  KTERROR(proclog, "Processor <" << procName << "> is not a primary processor.");
202  delete subNodePtr;
203  return false;
204  }
205 
206  threadGroup.insert(primaryProc);
207  }
208  fRunQueue.push_back(threadGroup);
209  }
210  else
211  {
212  KTWARN(proclog, "No run queue was specified during configuration.");
213  }
214  delete subNodePtr;
215 
216  return true;
217  }

248  return false;

254  Bool_t KTProcessorToolbox::Run()
255  {
256  UInt_t iGroup = 0;
257  for (RunQueue::const_iterator rqIter = fRunQueue.begin(); rqIter != fRunQueue.end(); rqIter++)
258  {
259 #ifdef SINGLETHREADED
260  for (ThreadGroup::const_iterator tgIter = rqIter->begin(); tgIter != rqIter->end(); tgIter++)
261  {
262  if (! (*tgIter)->Run())
263  {
264  return false;
265  }
266  }
267 #else
268  KTDEBUG(proclog, "Starting thread group " << iGroup);
269  boost::thread_group parallelThreads;
270  UInt_t iThread = 0;
271  for (ThreadGroup::const_iterator tgIter = rqIter->begin(); tgIter != rqIter->end(); tgIter++)
272  {
273  // create a boost::thread object to launch the thread
274  // use boost::ref to avoid copying the processor
275  KTDEBUG(proclog, "Starting thread " << iThread);
276  parallelThreads.create_thread(boost::ref(**tgIter));
277  iThread++;
278  }
279  // wait for execution to complete
280  parallelThreads.join_all();
281  iGroup++;
282 #endif
283  }
284  return true;
285  }
