60  void KTBasicROOTTypeWriterTime::WriteRawTimeSeriesData(Nymph::KTDataPtr data)
61  {
62  if (! data) return;
63 
64  KTSliceHeader& slHeader = data->Of<KTSliceHeader>();
65  uint64_t sliceNumber = slHeader.GetSliceNumber();
66 
67  KTRawTimeSeriesData& tsData = data->Of<KTRawTimeSeriesData>();
68  unsigned nComponents = tsData.GetNComponents();
69 
70  if (! fWriter->OpenAndVerifyFile()) return;
71 
72  for (unsigned iComponent=0; iComponent<nComponents; ++iComponent)
73  {
74  const KTRawTimeSeries* ts = tsData.GetTimeSeries(iComponent);
75  if (ts != NULL)
76  {
77  stringstream conv;
78  conv << "histRawTS_" << sliceNumber << "_" << iComponent;
79  string histName;
80  conv >> histName;
81  TH1I* tsHist = NULL;
82  if (slHeader.GetRawDataFormatType(iComponent) == sDigitizedUS)
83  {
84  tsHist = KT2ROOT::CreateHistogram(ts, histName);
85  }
86  else if(slHeader.GetRawDataFormatType(iComponent) == sDigitizedS)
87  {
88  KTVarTypePhysicalArray< int64_t > array(*ts, false);
89  tsHist = KT2ROOT::CreateHistogram(&array, histName);
