109  bool KTGainVariationProcessor::CoreGainVarCalc(KTFrequencySpectrumDataPolarCore& data, KTGainVariationData& newData)
110  {
111  if (fCalculateMinBin)
112  {
113  SetMinBin(data.GetSpectrumPolar(0)->FindBin(fMinFrequency));
114  }
115  else
116  {
117  fMinFrequency = data.GetSpectrumPolar(0)->GetBinCenter(fMinBin);
118  }
119  if (fCalculateMaxBin)
120  {
121  SetMaxBin(data.GetSpectrumPolar(0)->FindBin(fMaxFrequency));
122  }
123  else
124  {
125  fMaxFrequency = data.GetSpectrumPolar(0)->GetBinCenter(fMaxBin);
126  }
127  KTDEBUG(gvlog, "min frequency: " << fMinFrequency << "; max frequency: " << fMaxFrequency << "; min bin: " << fMinBin << "; max bin " << fMaxBin << "; input range max " << data.GetSpectrumPolar(0)->GetRangeMin() << "; input range min: " << data.GetSpectrumPolar(0)->GetRangeMax());
128 
129  unsigned nTotalBins = fMaxBin - fMinBin + 1;
130  unsigned nBinsPerFitPoint = nTotalBins / fNFitPoints; // integer division rounds down; there may be bins leftover unused
131 
132  KTDEBUG(gvlog, "Performing gain variation fits with " << fNFitPoints << " points, and " << nBinsPerFitPoint << " bins averaged per fit point.");
133 
134  unsigned nComponents = data.GetNComponents();
135 
136  //double sigmaNorm = 1. / double(nBinsPerFitPoint - 1);
137  for (unsigned iComponent=0; iComponent<nComponents; ++iComponent)
138  {
139  const KTFrequencySpectrumPolar* spectrum = data.GetSpectrumPolar(iComponent);
140 
141  double* xVals = new double[fNFitPoints];
142  double* yVals = new double[fNFitPoints];
143 
144  // Calculate fit points
145 #pragma omp parallel for default(shared)
146  for (unsigned iFitPoint=0; iFitPoint < fNFitPoints; iFitPoint++)
147  {
148  unsigned fitPointStartBin = iFitPoint * nBinsPerFitPoint + fMinBin;
149  unsigned fitPointEndBin = fitPointStartBin + nBinsPerFitPoint;
150 
151  double leftEdge = spectrum->GetBinLowEdge(fitPointStartBin);
152  double rightEdge = spectrum->GetBinLowEdge(fitPointEndBin);
153  xVals[iFitPoint] = leftEdge + 0.5 * (rightEdge - leftEdge);
154 
155  double mean = 0.;
156  for (unsigned iBin=fitPointStartBin; iBin<fitPointEndBin; iBin++)
157  {
158  mean += (*spectrum)(iBin).abs();
159  }
160  mean /= (double)nBinsPerFitPoint;
161  yVals[iFitPoint] = mean;
162 
163  KTDEBUG(gvlog, "Fit point " << iFitPoint << " " << xVals[iFitPoint] << " " << yVals[iFitPoint]);
164  }
165 
166  if (fNormalize)
167  {
168  // Normalize the fit points to 1
169  double minYVal = yVals[0];
170  for (unsigned iFitPoint=1; iFitPoint < fNFitPoints; iFitPoint++)
171  {
172  if (yVals[iFitPoint] < minYVal) minYVal = yVals[iFitPoint];
173  }
174  for (unsigned iFitPoint=0; iFitPoint < fNFitPoints; iFitPoint++)
175  {
176  yVals[iFitPoint] = yVals[iFitPoint] / minYVal;
177  }
178  }
179 
180  KTSpline* spline = new KTSpline(xVals, yVals, fNFitPoints);
181  spline->SetXMin(fMinFrequency);
182  spline->SetXMax(fMaxFrequency);
183  //GainVariation* gainVarResult = CreateGainVariation(spline, spectrum->GetNBins(), spectrum->GetRangeMin(), spectrum->GetRangeMax());
184 
185  delete [] xVals;
186  delete [] yVals;
187 
188  newData.SetSpline(spline, iComponent);
189  //newData->SetGainVariation(gainVarResult, iComponent);
190  }
191  KTINFO(gvlog, "Completed gain variation calculation for " << nComponents);
192 
193  return true;
