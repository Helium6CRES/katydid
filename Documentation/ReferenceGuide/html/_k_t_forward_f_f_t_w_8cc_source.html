127  bool KTForwardFFTW::InitializeFFT(KTForwardFFTW::State intendedState, unsigned timeSize)
128  {
129  if (intendedState == kNone)
130  {
131  KTERROR(fftwlog, "Cannot initialize FFT for state <" << intendedState << ">");
132  return false;
133  }
134 
135  if (timeSize == 0) timeSize = fTimeSize;
136 
137  // Set the time size whether or not it's different from fTimeSize, since the frequency size might not be
138  // right for the intended state.
139  SetTimeSizeForState(timeSize, intendedState);
140 
141  // fTransformFlag is guaranteed to be valid in the Set method.
142  KTDEBUG(fftwlog, "Transform flag: " << fTransformFlag);
143  TransformFlagMap::const_iterator iter = fTransformFlagMap.find(fTransformFlag);
144  unsigned transformFlag = iter->second;
145 
146  // allocate the input and output arrays if they're not there already
147  if (! AllocateArrays(intendedState))
148  {
149  KTERROR(fftwlog, "Unable to allocate arrays");
150  return false;
151  }
152 
153  if (fUseWisdom)
154  {
155  KTDEBUG(fftwlog, "Reading wisdom from file <" << fWisdomFilename << ">");
156  if (fftw_import_wisdom_from_filename(fWisdomFilename.c_str()) == 0)
157  {
158  KTWARN(fftwlog, "Unable to read FFTW wisdom from file <" << fWisdomFilename << ">");
159  }
160  }
161 
162  InitializeMultithreaded();
163 
164  if (intendedState == kR2C)
165  {
166  KTDEBUG(fftwlog, "Creating R2C plan: " << fTimeSize << " time bins; forward FFT");
167  // No FFTW_PRESERVE_INPUT, since the input array contents are replaced for each FFT
168  fForwardPlan = fftw_plan_dft_r2c_1d(fTimeSize, fRInputArray, fOutputArray, transformFlag);
169  // deleting arrays to save space
170  // input array is required; output array is not needed
171  KTDEBUG(fftwlog, "Freeing output array");
172  fftw_free(fOutputArray);
173  fOutputArray = NULL;
174  }
175  else if (intendedState == kC2C)
176  {
177  KTDEBUG(fftwlog, "Creating C2C plan: " << fTimeSize << " time bins; forward FFT");
178  // Add FFTW_PRESERVE_INPUT so that the input array content is not destroyed during the FFT
179  fForwardPlan = fftw_plan_dft_1d(fTimeSize, fCInputArray, fOutputArray, FFTW_FORWARD, transformFlag | FFTW_PRESERVE_INPUT);
180  // deleting arrays to save space
181  FreeArrays();
182  }
183  else // intendedState == kRasC2C
184  {
185  KTDEBUG(fftwlog, "Creating RasC2C plan: " << fTimeSize << " time bins; forward FFT");
186  // No FFTW_PRESERVE_INPUT, since the input array contents are replaced for each FFT
187  fForwardPlan = fftw_plan_dft_1d(fTimeSize, fCInputArray, fOutputArray, FFTW_FORWARD, transformFlag);
188  // deleting arrays to save space
189  // input array not required for C2C, but is for kRasC2C; output array not needed in either case
190  KTDEBUG(fftwlog, "Freeing output array");
191  fftw_free(fOutputArray);
192  fOutputArray = NULL;
193  }
194 
195  if (fForwardPlan != NULL)
196  {
197  fIsInitialized = true;
198  if (fUseWisdom)
199  {
200  if (fftw_export_wisdom_to_filename(fWisdomFilename.c_str()) == 0)
201  {
202  KTWARN(fftwlog, "Unable to write FFTW wisdom to file <" << fWisdomFilename << ">");
203  }
204  }
205  KTDEBUG(fftwlog, "FFTW plan created; Initialization complete.");
206  }
207  else
208  {
209  fIsInitialized = false;
210  KTERROR(fftwlog, "Unable to create the forward FFT plan! FFT is not initialized.");;
211  return false;
212  }
213 
214  fState = intendedState;
215  return true;
216  }

266

270  if (nextInput == NULL)

272  KTERROR(fftwlog, "Incorrect time series type: time series did not cast to KTTimeSeriesReal.");
273  return false;

275

281  return false;

284  newData.SetSpectrum(nextResult, iComponent);
285  }
