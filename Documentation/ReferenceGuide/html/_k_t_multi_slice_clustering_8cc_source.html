537  KTDEBUG(sclog, "adding unassigned fb clusters as new clusters");
538  for (FreqBinClusters::const_iterator fbIt = freqBinClusters.begin(); fbIt != freqBinClusters.end(); ++fbIt)
539  {
540  //KTDEBUG(sclog, "next fb cluster");
541  if (! fbIt->fAddedToActiveCluster)
542  {
543  //KTDEBUG(sclog, " adding new active cluster");
544  newCluster.fPoints.clear();
545  for (SetOfDiscriminatedPoints::const_iterator fbPointsIt = fbIt->fPoints.begin(); fbPointsIt != fbIt->fPoints.end(); ++fbPointsIt)
546  {
547  newPoint.fTimeBin = this->fTimeBin;
548  newPoint.fFreqBin = fbPointsIt->first;
549  newPoint.fAmplitude = fbPointsIt->second.fOrdinate;
550  newCluster.fPoints.insert(newPoint);
551  }
552  newCluster.fEndMinFreqPoint = fbIt->fFirstPoint;
553  newCluster.fEndMaxFreqPoint = fbIt->fLastPoint;
554  //KTDEBUG(sclog, " range is " << newCluster.fEndMinFreqPoint << " - " << newCluster.fEndMaxFreqPoint);

556  fActiveClusters[component].push_back(newCluster);
557  }
558  }
559  //KTDEBUG(sclog, "almost complete: " << fAlmostCompleteClusters[0].size() << " active: " << fActiveClusters[0].size());
560 
561  // Last step: add this frequency spectrum into the pre-cluster spectra list; remove the oldest item in the list if necessary
562  fPreClusterSpectra[component].push_back(spectrumPtr);
563  while (fPreClusterSpectra[component].size() > fNFramingTimeBins)
564  {
565  fPreClusterSpectra[component].pop_front();
566  }

568  return completeClusters; //CompleteInactiveClusters(component);

581  Nymph::KTDataPtr KTMultiSliceClustering::CreateDataFromCluster(const Cluster& cluster)
582  {
583 #ifndef NDEBUG
584  // if in debug mode, put together a string to print
585  std::stringstream printStream;
586  printStream << "Creating data object # " << fDataCount << '\n';
587 #endif
588 
589  ++fDataCount;
590 
591  Nymph::KTDataPtr data(new Nymph::KTData());
592 
593  KTWaterfallCandidateData& wfcData = data->Of< KTWaterfallCandidateData >();
594  wfcData.SetComponent(cluster.fDataComponent);
595 
596  wfcData.SetStartRecordNumber(cluster.fPoints.begin()->fHeaderPtr->GetStartRecordNumber());
597  wfcData.SetStartSampleNumber(cluster.fPoints.begin()->fHeaderPtr->GetStartSampleNumber());
598  wfcData.SetEndRecordNumber(cluster.fPoints.rbegin()->fHeaderPtr->GetEndRecordNumber());
599  wfcData.SetEndSampleNumber(cluster.fPoints.rbegin()->fHeaderPtr->GetEndSampleNumber());
600 
601  unsigned firstTimeBin = cluster.FirstTimeBin();
602  unsigned lastTimeBin = cluster.LastTimeBin();
603 
604  int firstTimeBinWithFrame = (int)firstTimeBin - (int)fNFramingTimeBins;
605  int lastTimeBinWithFrame = (int)lastTimeBin + (int)fNFramingTimeBins;
606 
607 #ifndef NDEBUG
608  printStream << "\tTime bin range: " << firstTimeBin << " - " << lastTimeBin << "; with frame: " << firstTimeBinWithFrame << " - " << lastTimeBinWithFrame << '\n';
609 #endif
610 
611  double ftbSumOfWeights = 0., ftbWeightedSum = 0.;
612  double ltbSumOfWeights = 0., ltbWeightedSum = 0.;
613 
614  SetOfPoints::const_iterator it = cluster.fPoints.begin();
615  unsigned firstFreqBin = it->fFreqBin;
616  unsigned lastFreqBin = firstFreqBin;
617 
618  if (it->fTimeBin == firstTimeBin)
619  {
620  ftbSumOfWeights += it->fAmplitude;
621  ftbWeightedSum += it->fSpectrumPtr->GetBinCenter(it->fFreqBin) * it->fAmplitude;
622  }
623  if (it->fTimeBin == lastTimeBin)
624  {
625  ltbSumOfWeights += it->fAmplitude;
626  ltbWeightedSum += it->fSpectrumPtr->GetBinCenter(it->fFreqBin) * it->fAmplitude;
627  }
628 
629  for (++it; it != cluster.fPoints.end(); ++it)
630  {
631  if (it->fFreqBin < firstFreqBin) firstFreqBin = it->fFreqBin;
632  if (it->fFreqBin > lastFreqBin) lastFreqBin = it->fFreqBin;
633 
634  if (it->fTimeBin == firstTimeBin)
635  {
636  ftbSumOfWeights += it->fAmplitude;
637  ftbWeightedSum += it->fSpectrumPtr->GetBinCenter(it->fFreqBin) * it->fAmplitude;
638  }
639  if (it->fTimeBin == lastTimeBin)
640  {
641  ltbSumOfWeights += it->fAmplitude;
642  ltbWeightedSum += it->fSpectrumPtr->GetBinCenter(it->fFreqBin) * it->fAmplitude;
643  }
644  }
645 
646  int firstFreqBinWithFrame = (int)firstFreqBin - (int)fNFramingFreqBins;
647  int lastFreqBinWithFrame =(int) lastFreqBin + (int)fNFramingFreqBins;
648 
649 #ifndef NDEBUG
650  printStream << "\tFreq bin range: " << firstFreqBin << " - " << lastFreqBin << "; with frame: " << firstFreqBinWithFrame << " - " << lastFreqBinWithFrame << '\n';
651 #endif
652 
653  unsigned firstFreqBinToUse = firstFreqBinWithFrame >= 0 ? (unsigned)firstFreqBinWithFrame : 0;
654  unsigned lastFreqBinToUse = lastFreqBinWithFrame < (int)(cluster.fPoints.begin()->fSpectrumPtr->size()) ? (unsigned)lastFreqBinWithFrame : cluster.fPoints.begin()->fSpectrumPtr->size() - 1;
655  unsigned freqBinOffset = unsigned((int)firstFreqBinToUse - firstFreqBinWithFrame);
656 
657  double timeOverlapFrac = cluster.fPoints.begin()->fHeaderPtr->GetNonOverlapFrac();
658  double timeBinWidthWithOverlap = cluster.fPoints.begin()->fHeaderPtr->GetSliceLength();
659  double timeBinWidth = timeBinWidthWithOverlap * timeOverlapFrac;
660  double freqBinWidth = cluster.fPoints.begin()->fSpectrumPtr->GetBinWidth();
661  unsigned nTimeBins = lastTimeBin - firstTimeBin + 1;
662  unsigned nFreqBins = lastFreqBin - firstFreqBin + 1;
663 
664  unsigned nTimeBinsWithFrame = nTimeBins + 2 * fNFramingTimeBins;
665  unsigned nFreqBinsWithFrame = nFreqBins + 2 * fNFramingFreqBins;
666 
667 #ifndef NDEBUG
668  printStream << "\tDimensions with frame: " << nTimeBinsWithFrame << " time bins and " << nFreqBinsWithFrame << " freq bins; cluster dimensions are " << nTimeBins << " by " << nFreqBins << '\n';
669  printStream << "\tTime bin width: " << timeBinWidthWithOverlap << " s (raw) --- " << timeBinWidth << " s (accounting for overlap)\n";
670  printStream << "\tFreq bin width: " << freqBinWidth << " Hz\n";
671 #endif
672 
673  //KTWARN(sclog, "loading spectrum pointers; expecting " << nTimeBinsWithFrame << " spectra");
674  // Create a vector of pointers to spectra, where each component is for a single time bin
675  vector< boost::shared_ptr< KTFrequencySpectrumPolar > > spectra(nTimeBinsWithFrame);
676  // Pre-cluster frame bins
677  unsigned frameBin = fNFramingTimeBins - cluster.fPreClusterSpectra.size();
678  for (ListOfSpectra::const_iterator losIt = cluster.fPreClusterSpectra.begin(); losIt != cluster.fPreClusterSpectra.end(); ++losIt)
679  {
680  spectra[frameBin] = *losIt;
681  //KTDEBUG(sclog, "put spectrum in frame bin " << frameBin << ": " << bool(spectra[frameBin]));
682  ++frameBin;
683  }
684  // Cluster bins
685  for (SetOfPoints::const_iterator it = cluster.fPoints.begin(); it != cluster.fPoints.end(); ++it)
686  {
687  if (! it->fSpectrumPtr)
688  {
689  KTWARN(sclog, "Spectrum pointer is NULL! " << it->fSpectrumPtr.get());
690  }
691  spectra[it->fTimeBin - firstTimeBin + fNFramingTimeBins] = it->fSpectrumPtr;
692  //KTDEBUG(sclog, "put spectrum in frame bin " << it->fTimeBin - firstTimeBin + fNFramingTimeBins << ": " << bool(spectra[it->fTimeBin - firstTimeBin + fNFramingTimeBins]));
693  }
694  // Post-cluster frame bins
695  frameBin = lastTimeBin - firstTimeBin + 1 + fNFramingTimeBins;
696  for (ListOfSpectra::const_iterator losIt = cluster.fPostClusterSpectra.begin(); losIt != cluster.fPostClusterSpectra.end(); ++losIt)
697  {
698  spectra[frameBin] = *losIt;
699  //KTDEBUG(sclog, "put spectrum in frame bin " << frameBin << ": " << bool(spectra[frameBin]));
700  ++frameBin;
701  }
702 
703  // this is the amount of time missing on each end of a cluster if the length of the cluster is calculated with the timeBinWidth, which is the non-overlap fraction of the slice length
704  double endTimeBinShift = 0.5 * (1. - timeOverlapFrac) * timeBinWidthWithOverlap;
705 
706  wfcData.SetTimeInRun(cluster.fPoints.begin()->fHeaderPtr->GetTimeInRun());
707  wfcData.SetFirstSliceNumber(cluster.fPoints.begin()->fHeaderPtr->GetSliceNumber());
708  wfcData.SetLastSliceNumber(cluster.fPoints.rbegin()->fHeaderPtr->GetSliceNumber());
709  wfcData.SetTimeLength(timeBinWidth * double(nTimeBins) + 2. * endTimeBinShift);
710  wfcData.SetMeanStartFrequency(ftbWeightedSum / ftbSumOfWeights);
711  wfcData.SetMeanEndFrequency(ltbWeightedSum / ltbSumOfWeights);
712  wfcData.SetFrequencyWidth(freqBinWidth * double(nFreqBins));
713 
714  unsigned i = 0;
715  for (; ! spectra[i]; ++i); // get the first spectrum that is definitely there (since sometimes spectra are missing)
716  if (i != spectra.size())
717  {
718  wfcData.SetMinimumFrequency(spectra[i]->GetBinLowEdge(firstFreqBin));
719  wfcData.SetMaximumFrequency(spectra[i]->GetBinLowEdge(lastFreqBin) + freqBinWidth);
720  }
721 #ifndef NDEBUG
722  printStream << "\tTime in run: " << wfcData.GetTimeInRun() << " s\n";
723  printStream << "\tTime length: " << wfcData.GetTimeLength() << " s\n";
724  printStream << "\tFirst slice number: " << wfcData.GetFirstSliceNumber() << '\n';
725  printStream << "\tLast slice number: " << wfcData.GetLastSliceNumber() << '\n';
726  printStream << "\tMean start freq: " << wfcData.GetMeanStartFrequency() << " Hz\n";
727  printStream << "\tMean end freq: " << wfcData.GetMeanEndFrequency() << " Hz\n";
728  printStream << "\tFreq width: " << wfcData.GetFrequencyWidth() << " Hz\n";
729  printStream << "\tMinimum frequency: " << wfcData.GetMinimumFrequency() << " Hz\n";
730  printStream << "\tMaximum frequency: " << wfcData.GetMaximumFrequency() << " Hz\n";
731 #endif
732 
733 
734  double tfStartTime = wfcData.GetTimeInRun() + endTimeBinShift - double(fNFramingTimeBins) * timeBinWidth;
735  double tfStartFreq = wfcData.GetMinimumFrequency() - double(fNFramingTimeBins) * freqBinWidth;
736  KTTimeFrequency* tf = new KTTimeFrequencyPolar(nTimeBinsWithFrame, tfStartTime, tfStartTime + double(nTimeBinsWithFrame)*timeBinWidth, nFreqBinsWithFrame, tfStartFreq, tfStartFreq + double(nFreqBinsWithFrame)*freqBinWidth);
737  //KTDEBUG(sclog, "tf freq dims: " << tfStartFreq << " - " << tfStartFreq + double(nFreqBinsWithFrame)*freqBinWidth);
738  for (int iTBin=firstTimeBinWithFrame; iTBin <= lastTimeBinWithFrame; ++iTBin)
739  {
740  int spectrumNum = iTBin - firstTimeBinWithFrame;
741  if (spectra[spectrumNum])
742  {
743  //KTDEBUG(sclog, " have spectrum for spectrumNum = " << spectrumNum);
744  for (unsigned iFBin=firstFreqBinToUse; iFBin <= lastFreqBinToUse; ++iFBin)
745  {
746  //KTDEBUG(sclog, " setting point at (" << spectrumNum << ", " << iFBin-firstFreqBinToUse << "), aka (" << iTBin << ", " << iFBin << "); length of spectrum: " << spectra[spectrumNum]->size());
747  tf->SetPolar(spectrumNum, iFBin - firstFreqBinToUse + freqBinOffset, spectra[spectrumNum]->GetAbs(iFBin), spectra[spectrumNum]->GetArg(iFBin));
748  }
749  }
750  /*
751  else
752  {
753  KTDEBUG(sclog, " NO SPECTRUM for spectrumNum = " << spectrumNum);
754  }
755  */
756  }
757  wfcData.SetCandidate(tf);
758 
759 #ifndef NDEBUG
760  KTDEBUG(sclog, printStream.str());
761 #endif
762 
763  return data;
764  }
