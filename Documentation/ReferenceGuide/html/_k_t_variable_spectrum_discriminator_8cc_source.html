332  bool KTVariableSpectrumDiscriminator::DiscriminateSpectrum(const KTFrequencySpectrumPolar* spectrum, const KTSpline* spline, KTDiscriminatedPoints1DData& newData, unsigned component)
333  {
334  if (spectrum == NULL)
335  {
336  KTERROR(sdlog, "Frequency spectrum pointer (component " << component << ") is NULL!");
337  return false;
338  }
339 
340  unsigned nBins = fMaxBin - fMinBin + 1;
341  double binWidth = spectrum->GetBinWidth();
342  double freqMin = spectrum->GetBinLowEdge(fMinBin);
343  double freqMax = spectrum->GetBinLowEdge(fMaxBin) + spectrum->GetBinWidth();
344  KTSpline::Implementation* splineImp = spline->Implement(nBins, freqMin, freqMax);
345 
346  //************
347  // SNR mode
348  //************
349  if (fThresholdMode == eSNR_Amplitude || fThresholdMode == eSNR_Power)
350  {
351  double thresholdMult = 0.;
352  if (fThresholdMode == eSNR_Amplitude)
353  {
354  // SNR = P_signal / P_noise = (A_signal / A_noise)^2, A_noise = mean
355  thresholdMult = sqrt(fSNRThreshold);
356  KTDEBUG(sdlog, "Discriminator threshold multiplier for component " << component << " set at <" << thresholdMult << "> (SNR-amplitude mode)");
357  }
358  else
359  {
360  // SNR = P_signal / P_noise, P_noise = mean
361  thresholdMult = fSNRThreshold;
362  KTDEBUG(sdlog, "Discriminator threshold multiplier for component " << component << " set at <" << thresholdMult << "> (SNR-power mode)");
363  }
364 
365  // loop over bins, checking against the threshold
366  double threshold, value;
367 #pragma omp parallel for private(value)
368  for (unsigned iBin=fMinBin; iBin<=fMaxBin; ++iBin)
369  {
370  value = (*spectrum)(iBin).abs();
371  threshold = thresholdMult * (*splineImp)(iBin - fMinBin);
372  if (value >= threshold)
373  newData.AddPoint(iBin, KTDiscriminatedPoints1DData::Point(binWidth * ((double)iBin + 0.5), value, threshold), component);
374  }
375  }
376  //**************
377  // Sigma mode
378  //**************
379  else if (fThresholdMode == eSigma)
380  {
381  double sigmaNorm = 1. / double(nBins - 1);
382  double sigma = 0., diff;
383 #pragma omp parallel for private(diff) reduction(+:sigma)
384  for (unsigned iBin=fMinBin; iBin<=fMaxBin; ++iBin)
385  {
386  diff = (*spectrum)(iBin).abs() - (*splineImp)(iBin - fMinBin);
387  sigma += diff * diff;
388  }
389  sigma = sqrt(sigma * sigmaNorm);
390 
391  double thresholdAdd = fSigmaThreshold * sigma;
392  KTDEBUG(sdlog, "Discriminator threshold diff for component " << component << " set at <" << thresholdAdd << "> (Sigma mode)");
393 
394  // loop over bins, checking against the threshold
395  double value, threshold;
396 #pragma omp parallel for private(value)
397  for (unsigned iBin=fMinBin; iBin<=fMaxBin; ++iBin)
398  {
399  value = (*spectrum)(iBin).abs();
400  threshold = thresholdAdd + (*splineImp)(iBin - fMinBin);
401  if (value >= threshold)
402  newData.AddPoint(iBin, KTDiscriminatedPoints1DData::Point(binWidth * ((double)iBin + 0.5), value, threshold), component);
403  }
404  }
405 
406  spline->AddToCache(splineImp);
407 
408  return true;
409
