116  bool KTMultiPeakEventBuilder::FindMultiPeakTracks()
117  {
118  KTPROG(tclog, "collecting lines into multi-peak tracks");
119  // loop over components
120  unsigned component = 0;
121  for (vector< TrackSet >::const_iterator compIt = fCompTracks.begin(); compIt != fCompTracks.end(); ++compIt)
122  {
123  KTINFO(tclog, "Doing component: " << compIt - fCompTracks.begin() + 1 << "/" << fCompTracks.size());
124  fMPTracks[component].clear();
125 
126  // loop over individual tracks
127  TrackSetCIt trackIt = compIt->begin();
128  if (trackIt == compIt->end()) continue;
129 
130  list< MultiPeakTrackRef > activeTrackRefs;
131 
132  int trackCount = 0;
133  while (trackIt != compIt->end())
134  {
135  KTINFO(tclog, "considering track (" << ++trackCount << "/" << compIt->size() << ")");
136  // loop over active track refs
137  list< MultiPeakTrackRef >::iterator mptrIt = activeTrackRefs.begin();
138  bool trackHasBeenAdded = false; // this will allow us to check all of the track refs for whether they're still active, even after adding the track to a ref
139  int activeTrackCount = 0;
140  while (mptrIt != activeTrackRefs.end())
141  {
142  KTINFO(tclog, "checking active track (" << ++activeTrackCount << "/" << activeTrackRefs.size() << ")" );
143  double deltaStartT = trackIt->GetStartTimeInRunC() - mptrIt->fMeanStartTimeInRunC;
144 
145  // check to see if this track ref should no longer be active
146  if (deltaStartT > fSidebandTimeTolerance)
147  {
148  KTDEBUG(tclog, "this track ref should no longer be active");
149  // there's no way this track, or any following it in the set, will match in time
150  fMPTracks[component].insert(*mptrIt);
151  mptrIt = activeTrackRefs.erase(mptrIt); // this results in mptrIt being one element past the one that was erased
152  KTDEBUG(tclog, "there are now " << fMPTracks[component].size() << " completed MPTracks");
153  }
154  else
155  {
156  double deltaEndT = trackIt->GetEndTimeInRunC() - mptrIt->fMeanEndTimeInRunC;
157  // check if this track should be added to this track ref
158  if ( !trackHasBeenAdded &&
159  (fabs(deltaStartT) <= fSidebandTimeTolerance || fabs(deltaEndT) < fSidebandTimeTolerance)
160  )
161  {
162  // then this track matches this track ref
163  mptrIt->InsertTrack(trackIt);
164  trackHasBeenAdded = true;
165  if (!(fabs(deltaStartT) <= fSidebandTimeTolerance && fabs(deltaEndT) < fSidebandTimeTolerance))
166  {
167  mptrIt->fUnknownEventTopology = true;
168  }
169  }
170  ++mptrIt; // only increment if we haven't removed one
171  }
172  } // while loop over active track refs
173  if (! trackHasBeenAdded) //track didn't match anything, create a new MPTrack
174  {
175  activeTrackRefs.push_back(MultiPeakTrackRef());
176  activeTrackRefs.rbegin()->InsertTrack(trackIt);
177  activeTrackRefs.rbegin()->fAcquisitionID = trackIt->GetAcquisitionID();
178  trackHasBeenAdded = true;
179  }
180  ++trackIt;
181  } // while loop over tracks
182 
183  // now that we're done with tracks, all active track refs are finished as well
184  list< MultiPeakTrackRef >::iterator mptrIt = activeTrackRefs.begin();
185  while (mptrIt != activeTrackRefs.end())
186  {
187  fMPTracks[component].insert(*mptrIt);
188  mptrIt = activeTrackRefs.erase(mptrIt); // this results in mptrIt being one element past the one that was erased
189  }
190 
191  ++component;
192  } // for loop over components
193 
194  return true;
195  }

197  bool KTMultiPeakEventBuilder::FindEvents()
198  {
199  KTPROG(tclog, "combining multi-peak tracks into events");
200 
201  // we're unpacking all components into a unified set of events, so this goes outside the loop
202  typedef std::set< double > TrackEndsType;
203  typedef std::pair< Nymph::KTDataPtr, TrackEndsType > ActiveEventType;
204  std::vector< ActiveEventType > activeEvents;
205 
206  for (unsigned iComponent = 0; iComponent < fMPTracks.size(); ++iComponent)
207  { // loop over components
208  KTINFO(tclog, "Doing component: (" << iComponent + 1 << "/" << fCompTracks.size() << ")");
209  if (fMPTracks[iComponent].empty())
210  { // if the component has no tracks, skip it
211  continue;
212  }
213  std::set< MultiPeakTrackRef, MTRComp >::iterator trackIt = fMPTracks[iComponent].begin();
214 
215  int countMPTracks = 0;
216  while (trackIt != fMPTracks[iComponent].end())
217  { // loop over new Multi-Peak Tracks
218  KTINFO(tclog, "placing MPTrack (" << ++countMPTracks << "/" << fMPTracks[iComponent].size() << ")");
219  int trackAssigned = -1; // keep track of which event the track when into
220 
221  for (std::vector< ActiveEventType >::iterator eventIt=activeEvents.begin(); eventIt != activeEvents.end();)
222  { // loop over active events and add this track to one
223  KTDEBUG(tclog, "checking active event (" << eventIt - activeEvents.begin() + 1 << "/" << activeEvents.size() << ")");
224  bool incrementEventIt = true;
225  if ( trackIt->fMeanStartTimeInRunC - fJumpTimeTolerance > *(eventIt->second.rbegin()) )
226  { // if the event's last end is earlier than this track's start, the event is done
227  KTDEBUG(tclog, "event no longer active");
228  eventIt->first->Of< KTMultiTrackEventData >().ProcessTracks();
229  fCandidates.insert(eventIt->first);
230  eventIt = activeEvents.erase(eventIt);
231  incrementEventIt = false;
232  continue;
233  }
234  for (TrackEndsType::iterator endTimeIt=eventIt->second.begin(); endTimeIt != eventIt->second.end();)
235  { // loop over track ends to test against
236  if ( trackIt->fMeanEndTimeInRunC - *endTimeIt < fJumpTimeTolerance )
237  { // if this track head matches the tail of a track in this event, add it
238  KTDEBUG(tclog, "track matched this active event");
239  if (trackAssigned == -1)
240  { // If this track hasn't been added to any event, add to this one
241  trackAssigned = eventIt - activeEvents.begin();
242 
243  KTMultiTrackEventData& thisEvent = eventIt->first->Of< KTMultiTrackEventData >();
244  thisEvent.AddTracks(trackIt->fTrackRefs);
245  if (trackIt->fUnknownEventTopology)
246  {
247  thisEvent.SetUnknownEventTopology(true);
248  }
249  thisEvent.ProcessTracks();
250  eventIt->second.insert( trackIt->fMeanEndTimeInRunC );
251  }
252  else
253  { // if this track is already in an event, merge this event into that one (NOTE: this is weird)
254  std::vector< ActiveEventType >::iterator firstEventLoc = activeEvents.begin();
255  std::advance( firstEventLoc, trackAssigned);
256  KTMultiTrackEventData& firstEvent = firstEventLoc->first->Of< KTMultiTrackEventData >();
257  KTMultiTrackEventData& thisEvent = eventIt->first->Of< KTMultiTrackEventData >();
258  thisEvent.SetUnknownEventTopology(true);
259  //for (unsigned iLine = 0; iLine < thisEvent.GetNTracks(); ++iLine)
260  firstEvent.AddTracks(thisEvent.GetTracksSet());
261  /*for (TrackSetCIt eventTrackIt=thisEvent.GetTracksBegin(); eventTrackIt != thisEvent.GetTracksEnd(); eventTrackIt++)
262  {
263  firstEvent.AddTracks(eventTrackIt);//thisEvent.GetTrack(iLine));
264  KTERROR("don't know how to deal with EventSequenceID here!!!")
265  }*/
266  firstEvent.ProcessTracks();
267  for (TrackEndsType::const_iterator endpointIt=eventIt->second.begin(); endpointIt != eventIt->second.end(); ++endpointIt)
268  {
269  firstEventLoc->second.insert(*endpointIt);
270  }
271  eventIt = activeEvents.erase(eventIt);
272  incrementEventIt = false;
273  }
274  break; // this track already matched the event, don't keep checking
275  }
276  ++endTimeIt;
277  } // for loop over end times
278  if (incrementEventIt)
279  { // don't increment if we removed this active event from the vector
280  ++eventIt;
281  }
282  } // for loop over active events
283  if (trackAssigned == -1)
284  { // if no event matched then create one
285  KTINFO(tclog, "track not matched, creating new event");
286  ++fDataCount;
287  Nymph::KTDataPtr data(new Nymph::KTData());
288  ActiveEventType new_event(data, TrackEndsType());
289  KTMultiTrackEventData& event = new_event.first->Of< KTMultiTrackEventData >();
290  event.SetComponent(iComponent);
291  event.SetAcquisitionID(trackIt->fAcquisitionID);
292  event.SetEventID(fDataCount);
293 
294  event.AddTracks(trackIt->fTrackRefs);
295  if (trackIt->fUnknownEventTopology)
296  {
297  event.SetUnknownEventTopology(true);
298  }
299  event.ProcessTracks();
300  new_event.second.insert( trackIt->fMeanEndTimeInRunC );
301  activeEvents.push_back(new_event);
302  }
303  else
304  {
305  KTINFO(tclog, "track assigned to event " << trackAssigned);
306  }
307  ++trackIt;
308  } // while loop over tracks
309  } // for loop over components
310 
311  // tracks are done so any remaining active events are done
312  std::vector< ActiveEventType >::iterator eventIt=activeEvents.begin();
313  while (eventIt != activeEvents.end())
314  {
315  eventIt->first->Of< KTMultiTrackEventData >().ProcessTracks();
316  fCandidates.insert(eventIt->first);
317  eventIt = activeEvents.erase(eventIt);
318  }
319 
320  // emit event signals
321  for (std::set< Nymph::KTDataPtr >::const_iterator dataIt=fCandidates.begin(); dataIt != fCandidates.end(); ++dataIt)
322  {
323  fEventSignal(*dataIt);
324  }
325  // clear everything since we've emitted these events
326  fCandidates.clear();
327  for (unsigned iComponent = 0; iComponent < fMPTracks.size(); ++iComponent)
328  {
329  fCompTracks[iComponent].clear();
330  }
331 
332  return true;
333  }
