220  void KTWignerVille::CalculateACF(Buffer::iterator data1It, const Buffer::iterator& data2End, unsigned /*iWindow*/)
221  {
222  // data2It will be decremented before it's used
223  Buffer::iterator data2It = data2End + fWindowSize;
224 
225  double t1_real;
226  double t1_imag;
227  double t2_real;
228  double t2_imag;
229 
230  //KTERROR(wvlog, "iWindow = " << iWindow);
231 
232  for (unsigned fftBin = 0; fftBin < fWindowSize; fftBin++)
233  {
234  // decrement data2It first so that it doesn't run past begin() on the first window
235  --data2It;
236  t1_real = data1It->real();
237  t1_imag = data1It->imag();
238  t2_real = data2It->real();
239  t2_imag = data2It->imag();
240  (*fInputArray)(fftBin)[0] = t1_real * t2_real + t1_imag * t2_imag;
241  (*fInputArray)(fftBin)[1] = t1_imag * t2_real - t1_real * t2_imag;
242  //KTWARN(wvlog, " " << fftBin << " -- " << t1_real << " " << t1_imag << " -- " << t2_real << " " << t2_imag << " -- " << (*fInputArray)(fftBin)[0] << " " << (*fInputArray)(fftBin)[1]);
243  ++data1It;
244  }
245 
246  if (fUseWindowFunction)

248  fWindower->ApplyWindow(fInputArray);

250
