79  Bool_t KTWignerVille::Configure(const KTPStoreNode* node)
80  {
81  const KTPStoreNode* childNode = node->GetChild("complex-fftw");
82  if (childNode != NULL)
83  {
84  fFFT->Configure(childNode);
85  }
86 
87  SetInputDataName(node->GetData< string >("input-data-name", fInputDataName));
88  SetOutputDataName(node->GetData< string >("output-data-name", fOutputDataName));
89 
90  KTPStoreNode::csi_pair itPair = node->EqualRange("wv-pair");
91  for (KTPStoreNode::const_sorted_iterator citer = itPair.first; citer != itPair.second; citer++)
92  {
93  string pairString(citer->second.get_value< string >());
94  UInt_t first = 0, second = 0;
95  Bool_t parsed = phrase_parse(pairString.begin(), pairString.end(),
96  (int_[ref(first)=boost::spirit::qi::_1] >> ',' >> int_[ref(second) = boost::spirit::qi::_1]),
97  space);
98  if (! parsed)
99  {
100  KTWARN(wvlog, "Unable to parse WV pair: " << pairString);
101  continue;
102  }
103  KTINFO(wvlog, "Adding WV pair " << first << ", " << second);
104  this->AddPair(KTWVPair(first, second));
105  }
106 
107  return true;

110  KTFrequencySpectrumDataFFTW* KTWignerVille::TransformData(const KTTimeSeriesData* data)
111  {
112  if (fPairs.empty())
113  {
114  KTWARN(wvlog, "No Wigner-Ville pairs specified; no transforms performed.");
115  return NULL;
116  }
117 
118  // cast all time series into KTTimeSeriesFFTW
119  vector< const KTTimeSeriesFFTW* > timeSeries(data->GetNTimeSeries());
120  for (UInt_t iTS=0; iTS < timeSeries.size(); iTS++)
121  {
122  timeSeries[iTS] = dynamic_cast< const KTTimeSeriesFFTW* >(data->GetTimeSeries(iTS));
123  if (timeSeries[iTS] == NULL)
124  {
125  KTERROR(wvlog, "Time series " << iTS << " did not cast to a const KTTimeSeriesFFTW*. No transforms performed.");
126  return NULL;
127  }
128  }
129 
130  KTFrequencySpectrumDataFFTW* newData = new KTFrequencySpectrumDataFFTW(fPairs.size());
131 
132  // Do WV transform for each pair
133  UInt_t iPair = 0;
134  for (PairVector::const_iterator pairIt = fPairs.begin(); pairIt != fPairs.end(); pairIt++)
135  {
136  UInt_t firstChannel = (*pairIt).first;
137  UInt_t secondChannel = (*pairIt).second;
138 
139  CrossMultiplyToInputArray(timeSeries[firstChannel], timeSeries[secondChannel], 0);
140 
141  KTFrequencySpectrumFFTW* newSpectrum = fFFT->Transform(fInputArray);
142  newSpectrum->SetRange(0.5 * newSpectrum->GetRangeMin(), 0.5 * newSpectrum->GetRangeMax());
143 
144  newData->SetSpectrum(newSpectrum, iPair);
145  iPair++;
146  }
147 
148  newData->SetBundle(data->GetBundle());
149  newData->SetName(fOutputDataName);
150 
151  fWVSignal(newData);
152 
153  return newData;
154 
155  }
