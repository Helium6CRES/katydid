80  void KTBasicROOTTypeWriterTransform::WriteFrequencySpectrumDataPolar(Nymph::KTDataPtr data)
81  {
82  if (! data) return;
83 
84  uint64_t sliceNumber = data->Of<KTSliceHeader>().GetSliceNumber();
85 
86  KTFrequencySpectrumDataPolar& fsData = data->Of<KTFrequencySpectrumDataPolar>();
87  unsigned nComponents = fsData.GetNComponents();
88 
89  if (! fWriter->OpenAndVerifyFile()) return;
90 
91  for (unsigned iChannel=0; iChannel<nComponents; iChannel++)
92  {
93  const KTFrequencySpectrumPolar* spectrum = fsData.GetSpectrumPolar(iChannel);
94  if (spectrum != NULL)
95  {
96  stringstream conv;
97  conv << "histFSpolar_" << sliceNumber << "_" << iChannel;
98  string histName;
99  conv >> histName;
100  TH1D* powerSpectrum = KT2ROOT::CreateMagnitudeHistogram(spectrum, histName);
101  powerSpectrum->SetDirectory(fWriter->GetFile());
102  powerSpectrum->Write();
103  KTDEBUG(publog, "Histogram <" << histName << "> written to ROOT file");
104  }
105  }
106  return;
107  }
