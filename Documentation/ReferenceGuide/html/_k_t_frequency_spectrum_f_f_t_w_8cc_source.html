87  KTFrequencySpectrumFFTW& KTFrequencySpectrumFFTW::AnalyticAssociate()
88  {
89  // Note: the data storage array is accessed directly, so the FFTW data storage format is used.
90  // Nyquist bin(s) and negative frequency bins are set to 0 (from size/2 to the end of the array)
91  // DC bin stays as is (array position 0).
92  // Positive frequency bins are multiplied by 2 (from array position 1 to size/2).
93  UInt_t nBins = size();
94  UInt_t nyquistPos = nBins / 2; // either the sole nyquist bin (if even # of bins) or the first of the two (if odd # of bins; bins are sequential in the array).
95 #pragma omp parallel for
96  for (UInt_t arrayPos=1; arrayPos<nyquistPos; arrayPos++)
97  {
98  fData[arrayPos][0] = fData[arrayPos][0] * 2.;
99  fData[arrayPos][1] = fData[arrayPos][1] * 2.;
100  }
101 #pragma omp parallel for
102  for (UInt_t arrayPos=nyquistPos; arrayPos<nBins; arrayPos++)
103  {
104  fData[arrayPos][0] = 0.;
105  fData[arrayPos][1] = 0.;
106  }
107  return *this;
108  }

308  TH1D* KTFrequencySpectrumFFTW::CreatePowerDistributionHistogram(const std::string& name) const
309  {
310  UInt_t nBins = size();
311  Double_t tMaxMag = -1.;
312  Double_t tMinMag = 1.e9;
313  Double_t value;
314  Double_t scaling = 1. / KTPowerSpectrum::GetResistance();
315  for (UInt_t iBin=0; iBin<nBins; iBin++)
316  {
317  // order doesn't matter, so use fData[iBin] to access values
318  value = (fData[iBin][0] * fData[iBin][0] + fData[iBin][1] * fData[iBin][1]) * scaling;
319  if (value < tMinMag) tMinMag = value;
320  if (value > tMaxMag) tMaxMag = value;
321  }
322  if (tMinMag < 1. && tMaxMag > 1.) tMinMag = 0.;
323  TH1D* hist = new TH1D(name.c_str(), "Power Distribution", 100, tMinMag*0.95, tMaxMag*1.05);
324  for (UInt_t iBin=0; iBin<nBins; iBin++)
325  {
326  // order matters, so use (*this)() to access values
327  value = (*this)(iBin)[0] * (*this)(iBin)[0] + (*this)(iBin)[1] * (*this)(iBin)[1];
328  hist->Fill(value * scaling);
329  }
330  hist->SetXTitle("Power (W)");
331  return hist;
332  }
