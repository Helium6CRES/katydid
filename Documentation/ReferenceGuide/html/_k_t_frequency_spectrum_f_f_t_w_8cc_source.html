117  KTFrequencySpectrumFFTW& KTFrequencySpectrumFFTW::AnalyticAssociate()
118  {
119  // This is only valid if the original signal is Real only (not complex)
120  // Note: the data storage array is accessed directly, so the FFTW data storage format is used.
121  // Nyquist bin(s) and negative frequency bins are set to 0 (from size/2 to the end of the array)
122  // DC bin stays as is (array position 0).
123  // Positive frequency bins are multiplied by 2 (from array position 1 to size/2).
124  unsigned nBins = size();
125  unsigned nyquistPos = nBins / 2; // either the sole nyquist bin (if even # of bins) or the first of the two (if odd # of bins; bins are sequential in the array).
126 #pragma omp parallel for
127  for (unsigned arrayPos=1; arrayPos<nyquistPos; arrayPos++)
128  {
129  fData[arrayPos][0] = fData[arrayPos][0] * 2.;
130  fData[arrayPos][1] = fData[arrayPos][1] * 2.;
131  }
132 #pragma omp parallel for
133  for (unsigned arrayPos=nyquistPos; arrayPos<nBins; arrayPos++)
134  {
135  fData[arrayPos][0] = 0.;
136  fData[arrayPos][1] = 0.;
137  }
138  return *this;
139  }

168  KTPowerSpectrum* KTFrequencySpectrumFFTW::CreatePowerSpectrum() const
169  {
170  // This function creates a power spectrum that runs from the smallest to the largest absolute frequency.
171  // It can handle frequency ranges that do or don't cross DC, and that are symmetric or asymmetric.
172  // If the frequency range does not cross DC, the power spectrum range will not either.
173  // If the frequency range crosses DC, the power spectrum range will run from DC to the maximum absolute frequency
174  // In this case negative-frequency bins are added to positive-frequency bins.
175 
176  double maxFreq = std::max(fabs(GetRangeMin()), fabs(GetRangeMax()));
177  double minFreq = -0.5 * GetBinWidth();
178  unsigned nBins = (maxFreq - minFreq) / GetBinWidth();
179  if (GetRangeMax() < 0. || GetRangeMin() > 0.)
180  {
181  minFreq = std::min(fabs(GetRangeMin()), fabs(GetRangeMax()));
182  nBins = size();
183  }
184 
185  KTPowerSpectrum* newPS = new KTPowerSpectrum(nBins, minFreq, maxFreq);
186  for (unsigned iBin = 0; iBin < nBins; ++iBin) (*newPS)(iBin) = 0.;
187 
188  int dcBin = FindBin(0.);
189  // default case: dcBin >= 0 && dcBin < size()
190  int firstPosFreqBin = dcBin;
191  int lastPosFreqBin = size();
192  int firstNegFreqBin = 0;
193  int lastNegFreqBin = dcBin;
194  if (dcBin >= (int)size())
195  {
196  firstPosFreqBin = size(); // lastPosFreqBin = size();
197  lastNegFreqBin = size(); // firstNEgFreqBin = 0
198  }
199  else if (dcBin < 0)
200  {
201  firstPosFreqBin = 0; // lastPosFreqBin = size();
202  firstNegFreqBin = dcBin; // lastNegFreqBin = dcBin;
203  }
204 
205  double scaling = 1. / KTPowerSpectrum::GetResistance() / (double)GetNTimeBins();
206 
207  double valueImag, valueReal;
208 #pragma omp parallel for private(valueReal, valueImag)
209  for (unsigned iBin = firstPosFreqBin; iBin < lastPosFreqBin; ++iBin)
210  {
211  valueReal = (*this)(iBin)[0];
212  valueImag = (*this)(iBin)[1];
213  (*newPS)(iBin) = (valueReal * valueReal + valueImag * valueImag) * scaling;
214  }
215 #pragma omp parallel for private(valueReal, valueImag)
216  for (unsigned iBin = firstNegFreqBin; iBin < lastNegFreqBin; ++iBin)
217  {
218  valueReal = (*this)(iBin)[0];
219  valueImag = (*this)(iBin)[1];
220  (*newPS)(iBin) = (valueReal * valueReal + valueImag * valueImag) * scaling;
221  }
222 
223  return newPS;
224  }
