364  Nymph::KTDataPtr KTRSAMatReader::HatchNextSlice()
365  {
366  // IMPORTANT:
367  // Updated: KTRSAMatReader::HatchNextSlice is currently capable of reading MAT files containing multiple records, *as long as they have
368  // the same size* - that is, MAT files of identical sizes that were concatenated;
369  // it still cannot read multiple channels
370 
371  // Initialize output
372  Nymph::KTDataPtr newData(new Nymph::KTData());
373 
374  // ********************************************************* //
375  // Check whether we still have enough data to fill up slice //
376  // ********************************************************* //
377  // fSamplesRead is used as a pointer to where we are in the file;
378  // It gets incremented by the fSliceSize at the end of reading a slice;
379  // So here we must:
380  // (1) test if there is still enough data in the file to fill the slice,
381  // and if not, exit;
382  // (2) test if there is still enough data in the record to fill the slice,
383  // and if not, increment it to the start of the next record.
384 
385  // Check if you still have enough data in the file
386  if (fSamplesRead + fSliceSize > fSamplesPerFile)
387  {
388  KTWARN(eggreadlog, "End of mat file reached");
389  KTDEBUG(eggreadlog, "fSamplesRead: " << fSamplesRead << "; fSliceSize: " << fSliceSize << "; fRecordSize: " << fRecordSize << "; fSamplesPerFile: " << fSamplesPerFile);
390 
391  // Return Empty Pointer
392  return Nymph::KTDataPtr();
393  }
394 
395  // Create Slice Header
396  KTSliceHeader& sliceHeader = newData->Of< KTSliceHeader >().SetNComponents(fHeader.GetNChannels());
397  sliceHeader.SetIsNewAcquisition(false);
398 
399  // If this slice if going to cross over a record boundary, then
400  // increment the record number, set slice to 0, move fSamplesRead
401  // to the start of the next record, and return empty data
402  // (you don't want a slice with data from 2 records...)
403  if ((fSamplesRead - fRecordsRead * fRecordSize + fSliceSize) > fRecordSize)
404  {
405  KTINFO(eggreadlog, "End of Record reached");
406  KTDEBUG(eggreadlog, "fRecordsRead: " << fRecordsRead << "; fSamplesRead: " << fSamplesRead << "; fSliceSize: " << fSliceSize << "; fRecordSize: " << fRecordSize << "; fSamplesPerFile: " << fSamplesPerFile);
407 
408  ++fRecordsRead;
409  fSamplesRead = fRecordsRead * fRecordSize;
410  sliceHeader.SetIsNewAcquisition(true);
411  }
412 
413  // ********************************** //
414  // Fill out slice header information //
415  // ********************************** //
416  if (fSliceNumber == 0) sliceHeader.SetIsNewAcquisition(true);
417 
418  // Slice Header Variables
419  sliceHeader.SetSampleRate(fHeader.GetAcquisitionRate());
420  sliceHeader.SetRawSliceSize(fSliceSize);
421  sliceHeader.SetSliceSize(fSliceSize);
422  sliceHeader.CalculateBinWidthAndSliceLength();
423  sliceHeader.SetNonOverlapFrac((double) fStride / (double) fSliceSize);

425  sliceHeader.SetTimeInAcq(GetTimeInAcq());
426  sliceHeader.SetSliceNumber(fSliceNumber);

446  (*newSliceComplex)(iBin)[1] = double(dataImag[iBin + fSamplesRead]);

449  fSamplesRead = fSamplesRead + fSliceSize;
