36 // Method to determine whether a line intersects a bin with finite size
37 // The line extends over a finite segment from (x1,y1) to (x2,y2)
38 // The bin has lower-left corner (xx1,yy1) and upper-right corner (xx2,yy2)
39 // It follows that the lower-right corner is (xx2,yy1) and the upper-left corner is (xx1,yy2)
40 bool lineIntersects( double x1, double y1, double x2, double y2, double xx1, double yy1, double xx2, double yy2 )
41 {
42  // If the x or y coordinates of the line are entirely outside the bin range, we can immediately return false
43  if( x2 <= xx1 || x1 >= xx2 )
44  return false;
45  if( y2 <= yy1 || y1 >= yy2 )
46  return false;
47 
48  double m = (y2 - y1) / (x2 - x1); // slope
49  double yLeft = y1 + m * (xx1 - x1); // y-position of the line at xx1
50  double yRight = y1 + m * (xx2 - x1); // y-position of the line at xx2
51 
52  if( m > 0 )
53  {
54  // Positive slope
55  // We check the upper-left corner and the lower-right corner
56  if( yLeft >= yy2 || yRight <= yy1 )
57  return false;
58  else
59  return true;
60  }
61  else
62  {
63  // Negative slope
64  // We check the lower-left corner and the upper-right corner
65  if( yLeft <= yy1 || yRight >= yy2 )
66  return false;
67  else
68  return true;
69  }
70 }
