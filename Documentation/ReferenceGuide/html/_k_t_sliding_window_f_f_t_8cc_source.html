165  KTPhysicalArray< 1, KTFrequencySpectrum* >* KTSlidingWindowFFT::Transform(const KTTimeSeries* data) const
166  {
167  if (fWindowFunction->GetSize() < data->size())
168  {
169  UInt_t windowShift = fWindowFunction->GetSize() - GetEffectiveOverlap();
170  UInt_t nWindows = (data->size() - fWindowFunction->GetSize()) / windowShift + 1;
171  UInt_t nTimeBinsNotUsed = data->size() - (nWindows - 1) * windowShift + fWindowFunction->GetSize();
172  Double_t timeMin = 0.;
173  Double_t timeMax = ((nWindows - 1) * windowShift + fWindowFunction->GetSize()) * fWindowFunction->GetBinWidth();
174  KTPhysicalArray< 1, KTFrequencySpectrum* >* newSpectra = new KTPhysicalArray< 1, KTFrequencySpectrum* >(data->size(), timeMin, timeMax);
175  UInt_t windowStart = 0;
176  for (UInt_t iWindow = 0; iWindow < nWindows; iWindow++)
177  {
178  copy(data->begin() + windowStart, data->begin() + windowStart + fWindowFunction->GetSize(), fInputArray);
179  fftw_execute(fFTPlan);
180  (*newSpectra)[iWindow] = ExtractTransformResult();
181  // emit a signal that the FFT was performed, for any connected slots
182  fSingleFFTSignal(iWindow, (*newSpectra)[iWindow]);
183  windowStart += windowShift;
184  }
185  KTINFO(fftlog_sw, "FFTs complete; windows used: " << nWindows << "; time bins not used: " << nTimeBinsNotUsed);
186  return newSpectra;
187  }
188 
189  KTERROR(fftlog_sw, "Window size is larger than time data: " << fWindowFunction->GetSize() << " > " << data->size() << "\n" <<
190  "No transform was performed!");
191  throw(std::length_error("Window size is larger than time data"));
192  return NULL;
193  }

195  KTFrequencySpectrum* KTSlidingWindowFFT::ExtractTransformResult() const
196  {
197  UInt_t freqSize = this->GetFrequencySize();

199 
200  Double_t tempReal, tempImag;
