85  bool KTTSGenerator::Run()
86  {
87  // Create, signal, and destroy the egg header
88  KTEggHeader* newHeader = CreateEggHeader();
89 
90  KTDEBUG(genlog, "Created header:\n" << *newHeader);
91 
92  fHeaderSignal(newHeader);
93  delete newHeader;
94 
95  // Loop over slices
96  // The local copy of the data shared pointer is created and destroyed in each iteration of the loop
97  for (fSliceCounter = 0; fSliceCounter < fNSlices; ++fSliceCounter)
98  {
99  Nymph::KTDataPtr newData = CreateNewData();
100 
101  if (! AddSliceHeader(*newData.get()))
102  {
103  KTERROR(genlog, "Something went wrong while adding the slice header");
104  return false;
105  }
106 
107  if (! AddEmptySlice(*newData.get()))
108  {
109  KTERROR(genlog, "Something went wrong while adding the empty slices");
110  return false;
111  }
112 
113  if (! newData->Has< KTTimeSeriesData >())
114  {
115  KTERROR(genlog, "New data does not contain time-series data!");
116  return false;
117  }
118 
119  GenerateTS(newData->Of< KTTimeSeriesData >());
120 
121  fDataSignal(newData);
122  }
123 
124  fDoneSignal();
125 
126  KTProcSummary* summary = new KTProcSummary();
127  summary->SetNSlicesProcessed(fSliceCounter);
128  summary->SetNRecordsProcessed((unsigned)ceil(double(fSliceCounter * fSliceSize) / fRecordSize));
129  summary->SetIntegratedTime(double(fSliceCounter * fSliceSize) * fBinWidth);
130  fSummarySignal(summary);
131  delete summary;
132 
133  return true;
134  }

185  bool KTTSGenerator::AddSliceHeader(Nymph::KTData& data) const
186  {
187  KTSliceHeader& sliceHeader = data.Of< KTSliceHeader >().SetNComponents(1);
188  sliceHeader.SetSampleRate(1. / fBinWidth);
189  sliceHeader.SetSliceSize(fSliceSize);
190  sliceHeader.SetRawSliceSize(fSliceSize);
191  sliceHeader.CalculateBinWidthAndSliceLength();
192  sliceHeader.SetTimeInRun(double(fSliceCounter * fSliceSize) * fBinWidth);
193  sliceHeader.SetSliceNumber(fSliceCounter);
194 
195  for (unsigned iComponent = 0; iComponent < fNChannels; ++iComponent)
196  {
197  sliceHeader.SetTimeStamp((uint64_t)(sliceHeader.GetTimeInRun() * (double)NSEC_PER_SEC), iComponent); // TODO: change this to 1e3 when switch to usec is made
198  sliceHeader.SetAcquisitionID(0);
199  sliceHeader.SetRecordID(0);
200  }
201 
202  KTDEBUG(genlog, "Filled out slice header:\n"
203  << "\tSample rate: " << sliceHeader.GetSampleRate() << " Hz\n"
204  << "\tSlice size: " << sliceHeader.GetSliceSize() << '\n'
205  << "\tBin width: " << sliceHeader.GetBinWidth() << " s\n"
206  << "\tSlice length: " << sliceHeader.GetSliceLength() << " s\n"
207  << "\tTime in run: " << sliceHeader.GetTimeInRun() << " s\n"
208  << "\tSlice number: " << sliceHeader.GetSliceNumber() << '\n'
209  << "\tStart record number: " << sliceHeader.GetStartRecordNumber() << '\n'
210  << "\tStart sample number: " << sliceHeader.GetStartSampleNumber());
211 
212  return true;
213  }
