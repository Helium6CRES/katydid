132  void KTSingleChannelDAC::Initialize()
133  {
134  if (! fShouldRunInitialize)
135  {
136  return;
137  }
138 
139  KTDEBUG(egglog_scdac, "Initializing single-channel DAC");
140 
141  if (! fVoltages.empty())
142  {
143  fVoltages.clear();
144  }
145 
146  unsigned levelDivisor = 1;
147 
148  scarab::dig_calib_params params;
149  // The bit adjustment is the number of bits that will be ignored, starting at the LSB.
150  // This will be used to account for both left-alignment of the data, if that's the case and if the bit depth is smaller than the data type size,
151  // and any simulated reduction in the bit depth that's requested by the user.
152  // Levels are skipped according to the bit adjustment: levelRepeat = 2^bitAdjustment
153  // The level skipping is done by repeating the same voltage for levelRepeat levels.
154  unsigned bitAdjustment = 0;
155  unsigned dataTypeBitSize = fDataTypeSize * 8;
156  if (fBitAlignment == sBitsAlignedLeft)
157  {
158  bitAdjustment += dataTypeBitSize - fNBits;
159  }
160  if (fBitDepthMode == kReducing)
161  {
162  bitAdjustment += fNBits - fEmulatedNBits;
163  //get_calib_params(fEmulatedNBits, sizeof(uint64_t), fVoltageOffset, fVoltageRange, fBitAlignment == sBitsAlignedRight, &params);
164  //levelDivisor = 1 << (fNBits - fEmulatedNBits);
165  //TODO: for reducing the bit depth, we currently don't account for a separately specified DAC gain
166  }
167  unsigned levelRepeat = pow( 2, bitAdjustment );
168 
169  unsigned bitsForVoltageArray = fBitAlignment == sBitsAlignedLeft ? dataTypeBitSize : fNBits;
170  if (fDACGain < 0)
171  {
172  get_calib_params(bitsForVoltageArray, sizeof(uint64_t), fVoltageOffset, fVoltageRange, fBitAlignment == sBitsAlignedRight, &params);
173  fDACGain = params.dac_gain;
174  }
175  else
176  {
177  get_calib_params2(bitsForVoltageArray, sizeof(uint64_t), fVoltageOffset, fVoltageRange, fDACGain, fBitAlignment == sBitsAlignedRight, &params);
178  }
179 
180  if (fBitDepthMode == kIncreasing)
181  {
182  if (fNBits >= fEmulatedNBits)
183  {
184  KTERROR(egglog_scdac, "Increasing-bit-depth mode was indicated, but emulated bits (" << fEmulatedNBits << ") <= actual bits (" << fNBits << ")");
185  return;
186  }
187  unsigned additionalBits = fEmulatedNBits - fNBits;
188  fOversamplingBins = pow(2, 2 * additionalBits);
189  fOversamplingScaleFactor = 1. / pow(2, additionalBits);
190  }
191 
192  KTDEBUG(egglog_scdac, "Assigning voltages with:\n" <<
193  "\tData type size: " << fDataTypeSize << " bytes\n" <<
194  "\tDigitizer bits: " << fNBits << '\n' <<
195  "\tBit depth mode: " << fBitDepthMode << '\n' <<
196  "\tBit alignment: " << fBitAlignment << '\n' <<
197  "\tVoltage levels: " << (1 << fNBits) << '\n' <<
198  "\tEmulated bits: " << fEmulatedNBits << '\n' <<
199  "\tLevel divisor: " << levelDivisor << '\n' <<
200  "\tReduced levels: " << params.levels << '\n' <<
201  "\tVoltage range: " << params.v_range << '\n' <<
202  "\tVoltage offset: " << params.v_offset << " V\n" <<
203  "\tDAC gain: " << params.dac_gain << " V\n" <<
204  "\tOversampling bins: " << fOversamplingBins << '\n' <<
205  "\tOversampling scale factor: " << fOversamplingScaleFactor << '\n' <<
206  "\tBit adjustment: " << bitAdjustment << '\n' <<
207  "\tLevel repeat period: " << levelRepeat << '\n');
208 
209  // calculating the voltage conversion
210  fVoltages.resize(params.levels);
211 
212  if (fDigitizedDataFormat == sDigitizedS)
213  {
214  fIntLevelOffset = params.levels / 2;
215  double valueHoldPos = scarab::d2a< int64_t, double >(0, &params);
216  double valueHoldNeg = valueHoldPos;
217  KTDEBUG( egglog_scdac, "Calculating voltage conversion for signed integers; integer level offset: " << fIntLevelOffset );
218  fVoltages[fIntLevelOffset] = valueHoldPos;
219  unsigned levelSkipCounter = 1;
220  for (int64_t level = 1; level < fIntLevelOffset; ++level)
221  {
222  if (levelSkipCounter == 0)
223  {
224  valueHoldPos = scarab::d2a< int64_t, double >(level, &params);
225  valueHoldNeg = scarab::d2a< int64_t, double >(-level, &params);
226  }
227  fVoltages[fIntLevelOffset + level] = valueHoldPos;
228  fVoltages[fIntLevelOffset - level] = valueHoldNeg;
229  KTWARN( egglog_scdac, "level <" << fIntLevelOffset + level << "> = " << fVoltages[fIntLevelOffset + level] );
230  KTWARN( egglog_scdac, "level <" << fIntLevelOffset - level << "> = " << fVoltages[fIntLevelOffset - level] );
231  ++levelSkipCounter;
232  if (levelSkipCounter == levelRepeat) levelSkipCounter = 0;
233  }
234  if (levelSkipCounter == 0)
235  {
236  valueHoldNeg = scarab::d2a< int64_t, double >(-fIntLevelOffset, &params);
237  }
238  fVoltages[0] = valueHoldNeg;
239  }
240  else //(fDigitizedDataFormat == sDigitizedUS)
241  {
242  fIntLevelOffset = 0;
243  unsigned levelSkipCounter = 0;
244  double valueHold = 0.;
245  KTDEBUG( egglog_scdac, "Calculating voltage conversion for unsigned integers up to " << params.levels << "; integer level offset: " << fIntLevelOffset );
246  for (uint64_t level = 0; level < params.levels; ++level)
247  {
248  if (levelSkipCounter == 0) valueHold = scarab::d2a< uint64_t, double >(level, &params);
249  fVoltages[level] = valueHold;
250  KTWARN(egglog_scdac, "level <" << level << "> = voltage " << fVoltages[level]);
251  ++levelSkipCounter;
252  if (levelSkipCounter == levelRepeat) levelSkipCounter = 0;
253  }
254  }
255 
256  // setting the convert function
257  if (fTimeSeriesType == kFFTWTimeSeries)
258  {
259  if (fBitDepthMode != kIncreasing)
260  {
261  if (fDigitizedDataFormat == sDigitizedS)
262  {
263  fConvertTSFunc = &KTSingleChannelDAC::ConvertSignedToFFTW;
264  KTDEBUG(egglog_scdac, "Convert function set to signed int --> FFTW");
265  }
266  else //(fDigitizedDataFormat == sDigitizedUS)
267  {
268  fConvertTSFunc = &KTSingleChannelDAC::ConvertUnsignedToFFTW;
269  KTDEBUG(egglog_scdac, "Convert function set to unsigned int --> FFTW");
270  }
271  }
272  else //(fBitDepthMode == kIncreasing)
273  {
274  if (fDigitizedDataFormat == sDigitizedS)
275  {
276  KTDEBUG(egglog_scdac, "Convert function set to signed int --> FFTW oversampled");
277  fConvertTSFunc = &KTSingleChannelDAC::ConvertSignedToFFTWOversampled;
278  }
279  else //(fDigitizedDataFormat == sDigitizedUS)
280  {
281  KTDEBUG(egglog_scdac, "Convert function set to unsigned int --> FFTW oversampled");
282  fConvertTSFunc = &KTSingleChannelDAC::ConvertUnsignedToFFTWOversampled;
283  }
284  }
285  }
286  else //(fTimeSeriesType == kRealTimeSeries)
287  {
288  if (fBitDepthMode != kIncreasing)
289  {
290  if (fDigitizedDataFormat == sDigitizedS)
291  {
292  KTDEBUG(egglog_scdac, "Convert function set to signed int --> real");
293  fConvertTSFunc = &KTSingleChannelDAC::ConvertSignedToReal;
294  }
295  else //(fDigitizedDataFormat == sDigitizedUS)
296  {
297  KTDEBUG(egglog_scdac, "Convert function set to unsigned int --> real");
298  fConvertTSFunc = &KTSingleChannelDAC::ConvertUnsignedToReal;
299  }
300  }
301  else //(fBitDepthMode == kIncreasing)
302  {
303  if (fDigitizedDataFormat == sDigitizedS)
304  {
305  KTDEBUG(egglog_scdac, "Convert function set to signed int --> real oversampled");
306  fConvertTSFunc = &KTSingleChannelDAC::ConvertSignedToRealOversampled;
307  }
308  else //(fDigitizedDataFormat == sDigitizedUS)
309  {
310  KTDEBUG(egglog_scdac, "Convert function set to unsigned int --> real oversampled");
311  fConvertTSFunc = &KTSingleChannelDAC::ConvertUnsignedToRealOversampled;
312  }
