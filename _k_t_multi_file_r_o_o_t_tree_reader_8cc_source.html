109  {

117  }
118 
119  KTDataPtr newData(new KTData());
120  for (deque< DataType >::const_iterator dtIt = fDataTypes.begin(); dtIt != fDataTypes.end(); dtIt++)

123

134  return false;

136  (*(dtIt->fSignal))(newData);

183  bool KTMultiFileROOTTreeReader::AppendAmpDistData(TTree* tree, KTData& appendToData)
184  {
185  // Determine the number of components and the number of frequency bins
186  unsigned nComponents = 0;
187  unsigned nFreqBins = 0;
188  while (true)
189  {
190  stringstream cut;
191  cut << "Component == " << nComponents;
192  unsigned nEntriesForComponent = (unsigned)tree->GetEntries(cut.str().c_str());
193  if (nEntriesForComponent == 0)
194  {
195  nComponents++; // advance this by 1 to be the number of components, not the last component number
196  break;
197  }
198  if (nEntriesForComponent > nFreqBins)
199  {
200  nFreqBins = nEntriesForComponent;
201  }
202  }
203 
204  // Create data structures
205  TAmplitudeDistributionData ampDistData;
206  tree->SetBranchAddress("Component", &ampDistData.fComponent);
207  tree->SetBranchAddress("FreqBin", &ampDistData.fFreqBin);
208  tree->SetBranchAddress("Distribution", &ampDistData.fDistribution);
209 
210  KTDEBUG(inlog, "Initializing new set of amplitude distributions, with " << nComponents << " components and " << nFreqBins << "frequency bins");
211  KTAmplitudeDistribution& ampDist = appendToData.Of< KTAmplitudeDistribution >();
212  ampDist.InitializeNull(nComponents, nFreqBins);
213 
214  // Read in the data
215  unsigned nEntries = (unsigned)tree->GetEntries();
216  for (unsigned iEntry=0; iEntry < nEntries; iEntry++)
217  {
218  tree->GetEntry(iEntry);
219 
220  unsigned nDistBins = (unsigned)ampDistData.fDistribution->GetNbinsX();
221  ampDist.InitializeADistribution(ampDistData.fComponent, ampDistData.fFreqBin, nDistBins, ampDistData.fDistribution->GetXaxis()->GetXmin(), ampDistData.fDistribution->GetXaxis()->GetXmax());
222  for (unsigned iDistBin = 0; iDistBin < nDistBins; iDistBin++)
223  {
224  ampDist.SetDistValue(ampDistData.fDistribution->GetBinContent(iDistBin+1), ampDistData.fFreqBin, iDistBin, ampDistData.fComponent);
225  }
226  }
227 
228 
229  KTDEBUG(inlog, "some success message!");
230 
231  return true;
232  }
