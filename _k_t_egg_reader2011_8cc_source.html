168

170  if (nodeDigitizer == NULL)
171  {
172  KTERROR(eggreadlog, "No digitizer node");
173  return NULL;
174  }
175 
176  attr = nodeDigitizer->first_attribute("rate");
177  if (attr == NULL)
178  {

183

188  return NULL;

190

197

199

202  << "\tTime Stamp Size: " << fHeaderInfo.fTimeStampSize << '\n'
203  << "\tRecord Size: " << fHeaderInfo.fRecordSize << '\n'
